### 1.6 后端核心代码 - 智能匹配算法

```java
package com.andiantong.order.service.impl;

import com.andiantong.electrician.entity.Electrician;
import com.andiantong.electrician.mapper.ElectricianMapper;
import com.andiantong.order.dto.CreateOrderDTO;
import com.andiantong.order.entity.Order;
import com.andiantong.order.mapper.OrderMapper;
import com.andiantong.order.service.MatchService;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.data.redis.core.RedisTemplate;
import org.springframework.stereotype.Service;

import java.util.*;
import java.util.concurrent.TimeUnit;
import java.util.stream.Collectors;

@Slf4j
@Service
@RequiredArgsConstructor
public class MatchServiceImpl implements MatchService {

    private final ElectricianMapper electricianMapper;
    private final OrderMapper orderMapper;
    private final RedisTemplate<String, Object> redisTemplate;

    private static final double SEARCH_RADIUS_KM = 5.0;
    private static final double EARTH_RADIUS_KM = 6371.0;
    private static final double W1_DISTANCE = 0.4;
    private static final double W2_RATING = 0.35;
    private static final double W3_LEVEL = 0.25;
    private static final int MAX_PUSH_COUNT = 10;

    @Override
    public List<Electrician> matchElectricians(CreateOrderDTO dto) {
        double userLat = dto.getLatitude();
        double userLng = dto.getLongitude();

        List<Electrician> nearbyElectricians = electricianMapper
                .findOnlineWithinRadius(userLat, userLng, SEARCH_RADIUS_KM);

        if (nearbyElectricians.isEmpty()) {
            log.warn("No electricians found within {}km of ({}, {})",
                    SEARCH_RADIUS_KM, userLat, userLng);
            return Collections.emptyList();
        }

        List<Map.Entry<Electrician, Double>> scoredList = nearbyElectricians.stream()
                .map(e -> {
                    double distanceScore = calculateDistanceScore(
                            userLat, userLng, e.getLatitude(), e.getLongitude());
                    double ratingScore = calculateRatingScore(e.getRating());
                    double levelScore = calculateLevelScore(e.getLevel());
                    double totalScore = W1_DISTANCE * distanceScore
                            + W2_RATING * ratingScore
                            + W3_LEVEL * levelScore;
                    return Map.entry(e, totalScore);
                })
                .sorted(Map.Entry.<Electrician, Double>comparingByValue().reversed())
                .collect(Collectors.toList());

        List<Electrician> result = scoredList.stream()
                .limit(MAX_PUSH_COUNT)
                .map(Map.Entry::getKey)
                .collect(Collectors.toList());

        String cacheKey = "order:match:" + dto.getUserId();
        redisTemplate.opsForValue().set(cacheKey, result, 30, TimeUnit.MINUTES);

        return result;
    }

    private double calculateDistanceScore(double lat1, double lng1,
                                          double lat2, double lng2) {
        double distance = haversineDistance(lat1, lng1, lat2, lng2);
        return Math.max(0, 1.0 - (distance / SEARCH_RADIUS_KM));
    }

    private double haversineDistance(double lat1, double lng1,
                                     double lat2, double lng2) {
        double dLat = Math.toRadians(lat2 - lat1);
        double dLng = Math.toRadians(lng2 - lng1);
        double a = Math.sin(dLat / 2) * Math.sin(dLat / 2)
                + Math.cos(Math.toRadians(lat1)) * Math.cos(Math.toRadians(lat2))
                * Math.sin(dLng / 2) * Math.sin(dLng / 2);
        double c = 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1 - a));
        return EARTH_RADIUS_KM * c;
    }

    private double calculateRatingScore(double rating) {
        return rating / 5.0;
    }

    private double calculateLevelScore(Integer level) {
        if (level == null) return 0.0;
        return level / 4.0;
    }
}
```

```java
package com.andiantong.order.websocket;

import com.alibaba.fastjson.JSON;
import com.andiantong.order.dto.LocationDTO;
import lombok.extern.slf4j.Slf4j;
import org.springframework.stereotype.Component;

import javax.websocket.*;
import javax.websocket.server.PathParam;
import javax.websocket.server.ServerEndpoint;
import java.io.IOException;
import java.util.concurrent.ConcurrentHashMap;

@Slf4j
@Component
@ServerEndpoint("/ws/location/{orderId}")
public class LocationWebSocket {

    private static final ConcurrentHashMap<String, Session> SESSION_MAP
            = new ConcurrentHashMap<>();

    @OnOpen
    public void onOpen(Session session, @PathParam("orderId") String orderId) {
        SESSION_MAP.put(orderId, session);
        log.info("WebSocket connected for order: {}", orderId);
    }

    @OnMessage
    public void onMessage(String message, @PathParam("orderId") String orderId) {
        try {
            LocationDTO location = JSON.parseObject(message, LocationDTO.class);
            String userId = location.getUserId();

            Session userSession = SESSION_MAP.get("user_" + userId);
            if (userSession != null && userSession.isOpen()) {
                userSession.getAsyncRemote().sendText(message);
            }

            log.debug("Location update for order {}: lat={}, lng={}",
                    orderId, location.getLatitude(), location.getLongitude());
        } catch (Exception e) {
            log.error("Error processing location message for order: {}", orderId, e);
        }
    }

    @OnClose
    public void onClose(@PathParam("orderId") String orderId) {
        SESSION_MAP.remove(orderId);
        log.info("WebSocket closed for order: {}", orderId);
    }

    @OnError
    public void onError(Session session, Throwable error) {
        log.error("WebSocket error", error);
    }

    public static void sendMessage(String orderId, String message) {
        Session session = SESSION_MAP.get(orderId);
        if (session != null && session.isOpen()) {
            try {
                session.getBasicRemote().sendText(message);
            } catch (IOException e) {
                log.error("Error sending WebSocket message", e);
            }
        }
    }
}
```

```typescript
import React, { useState, useEffect } from 'react';
import { View, Text, ScrollView, Image } from '@tarojs/components';
import Taro from '@tarojs/taro';
import { getServiceList, getBannerList } from '../../api/home';
import { ServiceItem, BannerItem } from '../../types/home';
import './index.scss';

const HomePage: React.FC = () => {
  const [services, setServices] = useState<ServiceItem[]>([]);
  const [banners, setBanners] = useState<BannerItem[]>([]);

  useEffect(() => {
    loadHomeData();
  }, []);

  const loadHomeData = async () => {
    try {
      const [serviceRes, bannerRes] = await Promise.all([
        getServiceList(),
        getBannerList()
      ]);
      setServices(serviceRes.data);
      setBanners(bannerRes.data);
    } catch (error) {
      Taro.showToast({ title: '加载失败', icon: 'none' });
    }
  };

  const handleServiceClick = (service: ServiceItem) => {
    switch (service.type) {
      case 'emergency':
        Taro.navigateTo({ url: '/pages/repair/index?type=emergency' });
        break;
      case 'inspect':
        Taro.navigateTo({ url: '/pages/repair/index?type=inspect' });
        break;
      case 'install':
        Taro.navigateTo({ url: '/pages/repair/index?type=install' });
        break;
      case 'smart':
        Taro.navigateTo({ url: '/pages/repair/index?type=smart' });
        break;
      default:
        Taro.navigateTo({ url: '/pages/repair/index' });

  };
```

```typescript
const FAULT_TYPES: FaultType[] = [

const RepairPage: React.FC = () => {
  const [selectedType, setSelectedType] = useState<number>(0);
  const [description, setDescription] = useState<string>('');
  const [images, setImages] = useState<string[]>([]);
  const [address, setAddress] = useState<string>('');
  const [submitting, setSubmitting] = useState<boolean>(false);

  const handleChooseImage = () => {
    if (images.length >= 3) {
      Taro.showToast({ title: '最多上传3张照片', icon: 'none' });
      return;
    }
    Taro.chooseImage({
      count: 3 - images.length,
      sizeType: ['compressed'],
      sourceType: ['album', 'camera'],
      success: async (res) => {
        for (const tempPath of res.tempFilePaths) {
          try {
            const url = await uploadImage(tempPath);
            setImages(prev => [...prev, url]);
          } catch (error) {
            Taro.showToast({ title: '图片上传失败', icon: 'none' });
          }
        }
      }
    });
  };

  const handleRemoveImage = (index: number) => {
    setImages(prev => prev.filter((_, i) => i !== index));
  };

  const handleSubmit = async () => {
    if (!selectedType) {
      Taro.showToast({ title: '请选择故障类型', icon: 'none' });
      return;
    }
    if (!description.trim()) {
      Taro.showToast({ title: '请描述故障情况', icon: 'none' });
      return;
    }
    if (!address.trim()) {
      Taro.showToast({ title: '请选择服务地址', icon: 'none' });
      return;
    }

    setSubmitting(true);
    try {
      const params: CreateOrderParams = {
        faultTypeId: selectedType,
        description: description.trim(),
        images: images,
        address: address.trim(),
        serviceTime: new Date().toISOString()
      };
      const result = await createOrder(params);
      Taro.showToast({ title: '报修成功', icon: 'success' });
      setTimeout(() => {
        Taro.navigateTo({ url: `/pages/order/detail?id=${result.data.id}` });
      }, 1500);
    } catch (error) {
      Taro.showToast({ title: '报修失败，请重试', icon: 'none' });
    } finally {
      setSubmitting(false);
    }
  };

  return (
    <View className="repair-page">
      <View className="fault-type-section">
        <Text className="section-label">故障类型</Text>
        <View className="fault-type-grid">
          {FAULT_TYPES.map(type => (
            <View
              key={type.id}
              className={`fault-type-item ${selectedType === type.id ? 'active' : ''}`}
              onClick={() => setSelectedType(type.id)}
            >
              <Text className="type-name">{type.name}</Text>
              <Text className="type-desc">{type.desc}</Text>
            </View>
          ))}
        </View>
      </View>

      <View className="description-section">
        <Text className="section-label">故障描述</Text>
        <Input
          className="description-input"
          type="text"
          placeholder="请描述您的用电故障情况"
          value={description}
          onInput={e => setDescription(e.detail.value)}
          maxLength={200}
        />
      </View>

      <View className="image-section">
        <Text className="section-label">故障照片（最多3张）</Text>
        <View className="image-list">
          {images.map((img, index) => (
            <View key={index} className="image-item">
              <Image className="preview-image" src={img} />
              <View className="remove-btn" onClick={() => handleRemoveImage(index)}>
                <Text>×</Text>
              </View>
            </View>
          ))}
          {images.length < 3 && (
            <View className="add-image-btn" onClick={handleChooseImage}>
              <Text>+</Text>
            </View>
          )}
        </View>
      </View>

      <View className="address-section">
        <Text className="section-label">服务地址</Text>
        <Input
```

```java
            addPoints(userId, points, "订单消费返积分", "ORDER");
        }
    }

    @Override
    @Transactional(rollbackFor = Exception.class)
    public void addInvitePoints(Long userId) {
        addPoints(userId, INVITE_POINTS, "邀请好友奖励", "INVITE");
    }

    @Override
    @Transactional(rollbackFor = Exception.class)
    public void addReviewPoints(Long userId) {
        addPoints(userId, REVIEW_POINTS, "评价订单奖励", "REVIEW");
    }

    @Override
    @Transactional(rollbackFor = Exception.class)
    public void addSharePoints(Long userId) {
        String lockKey = POINTS_LOCK_PREFIX + "share:" + userId;
        Boolean locked = redisTemplate.opsForValue()
                .setIfAbsent(lockKey, "1", 24, TimeUnit.HOURS);
        if (Boolean.FALSE.equals(locked)) {
            log.info("Share points already awarded today for user: {}", userId);
            return;
        }
        addPoints(userId, SHARE_POINTS, "分享活动奖励", "SHARE");
    }

    @Override
    @Transactional(rollbackFor = Exception.class)
    public void exchangeGoods(Long userId, ExchangeDTO dto) {
        String lockKey = POINTS_LOCK_PREFIX + "exchange:" + userId;
        Boolean locked = redisTemplate.opsForValue()
                .setIfAbsent(lockKey, "1", 10, TimeUnit.SECONDS);
        if (Boolean.FALSE.equals(locked)) {
            throw new BusinessException("操作过于频繁，请稍后再试");
        }

        try {
            User user = userMapper.selectById(userId);
            if (user == null) {
                throw new BusinessException("用户不存在");
            }

            Goods goods = goodsMapper.selectById(dto.getGoodsId());
            if (goods == null) {
                throw new BusinessException("商品不存在");
            }

            if (goods.getStock() <= 0) {

            }

                throw new BusinessException("积分不足");

            log.info("User {} exchanged goods {} for {} points",
                    userId, goods.getId(), goods.getPointsPrice());
        } finally {
            redisTemplate.delete(lockKey);
        }
    }

            throw new BusinessException("用户不存在");
```

```java
package com.andiantong.ai.service.impl;

import com.andiantong.ai.dto.DiagnosisDTO;
import com.andiantong.ai.entity.FaultRule;
import com.andiantong.ai.mapper.FaultRuleMapper;
import com.andiantong.ai.service.DiagnosisService;
import com.andiantong.ai.vo.DiagnosisVO;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.data.redis.core.RedisTemplate;
import org.springframework.stereotype.Service;

import java.util.*;
import java.util.concurrent.TimeUnit;
import java.util.regex.Matcher;
import java.util.regex.Pattern;
import java.util.stream.Collectors;

@Slf4j
@Service
@RequiredArgsConstructor
public class DiagnosisServiceImpl implements DiagnosisService {

    private final FaultRuleMapper faultRuleMapper;
    private final RedisTemplate<String, Object> redisTemplate;

    private static final String RULES_CACHE_KEY = "ai:fault:rules";

    @Override
    public DiagnosisVO diagnose(DiagnosisDTO dto) {
        String description = dto.getDescription().toLowerCase();

        List<FaultRule> rules = getFaultRules();

        List<Map.Entry<FaultRule, Integer>> matchedRules = rules.stream()
                .map(rule -> {
                    int score = calculateMatchScore(description, rule);
                    return Map.entry(rule, score);
                })
                .filter(entry -> entry.getValue() > 0)
                .sorted(Map.Entry.<FaultRule, Integer>comparingByValue().reversed())
                .collect(Collectors.toList());

        if (matchedRules.isEmpty()) {
            DiagnosisVO result = new DiagnosisVO();
            result.setFaultType("其他问题");
            result.setConfidence(0.3);
            result.setSuggestion("建议联系专业电工上门检查");
            result.setEstimatedCost(299);
            return result;
        }

        FaultRule bestMatch = matchedRules.get(0).getKey();
        int bestScore = matchedRules.get(0).getValue();
        double confidence = Math.min(0.95, bestScore / 10.0);

        DiagnosisVO result = new DiagnosisVO();
        result.setFaultType(bestMatch.getFaultType());
        result.setConfidence(confidence);
        result.setSuggestion(bestMatch.getSuggestion());
        result.setEstimatedCost(bestMatch.getEstimatedCost());
        result.setUrgencyLevel(bestMatch.getUrgencyLevel());

        if (matchedRules.size() > 1) {
            List<DiagnosisVO.AlternativeDiagnosis> alternatives = matchedRules.stream()
                    .skip(1)
                    .limit(3)
                    .map(entry -> {
                        DiagnosisVO.AlternativeDiagnosis alt = new DiagnosisVO.AlternativeDiagnosis();
                        alt.setFaultType(entry.getKey().getFaultType());
                        alt.setConfidence(Math.min(0.9, entry.getValue() / 10.0));
                        return alt;
                    })
                    .collect(Collectors.toList());
            result.setAlternatives(alternatives);
        }

        log.info("Diagnosis result: type={}, confidence={}, cost={}",
                result.getFaultType(), result.getConfidence(),
                result.getEstimatedCost());

        return result;
    }

    private int calculateMatchScore(String description, FaultRule rule) {
        int score = 0;
        String[] keywords = rule.getKeywords().split(",");

        for (String keyword : keywords) {
            String trimmedKeyword = keyword.trim().toLowerCase();
            if (trimmedKeyword.isEmpty()) continue;

            if (description.contains(trimmedKeyword)) {
                score += rule.getWeight() != null ? rule.getWeight() : 1;
            }

            try {
                Pattern pattern = Pattern.compile(trimmedKeyword);
                Matcher matcher = pattern.matcher(description);
                if (matcher.find()) {
                    score += 2;
                }
            } catch (Exception ignored) {
            }
        }

        return score;
    }

    @SuppressWarnings("unchecked")
    private List<FaultRule> getFaultRules() {
        Object cached = redisTemplate.opsForValue().get(RULES_CACHE_KEY);
        if (cached != null) {
            return (List<FaultRule>) cached;
        }

        List<FaultRule> rules = faultRuleMapper.selectList(null);
        redisTemplate.opsForValue().set(RULES_CACHE_KEY, rules, 1, TimeUnit.HOURS);
        return rules;
    }
}
```

```java
public class CommonUtil {

    private static final SecureRandom RANDOM = new SecureRandom();
    private static final String CHARS = "0123456789ABCDEFGHIJKLMNOPQRSTUVWXYZ";

    public static String generateOrderNo() {
        SimpleDateFormat sdf = new SimpleDateFormat("yyyyMMddHHmmss");
        String timestamp = sdf.format(new Date());
        String random = generateRandomString(6);
        return "ADT" + timestamp + random;
    }

    public static String generateRandomString(int length) {
        StringBuilder sb = new StringBuilder(length);
        for (int i = 0; i < length; i++) {
            sb.append(CHARS.charAt(RANDOM.nextInt(CHARS.length())));
        }
        return sb.toString();
    }

    public static String generateUUID() {
        return UUID.randomUUID().toString().replace("-", "");
    }

    public static String formatDateTime(LocalDateTime dateTime) {
        if (dateTime == null) return "";
        return dateTime.format(DateTimeFormatter.ofPattern("yyyy-MM-dd HH:mm:ss"));
    }

    public static String formatDate(LocalDateTime dateTime) {
        if (dateTime == null) return "";
        return dateTime.format(DateTimeFormatter.ofPattern("yyyy-MM-dd"));
    }

    public static String maskPhone(String phone) {
        if (phone == null || phone.length() < 7) return phone;
        return phone.substring(0, 3) + "****" + phone.substring(7);
    }

    public static String maskIdCard(String idCard) {
        if (idCard == null || idCard.length() < 8) return idCard;
        return idCard.substring(0, 4) + "**********"
                + idCard.substring(idCard.length() - 4);
    }

    public static boolean isValidPhone(String phone) {
        if (phone == null) return false;
        return phone.matches("^1[3-9]\\d{9}$");
```

```java
package com.andiantong.common.util;

import io.jsonwebtoken.Claims;
import io.jsonwebtoken.Jwts;
import io.jsonwebtoken.SignatureAlgorithm;
import io.jsonwebtoken.security.Keys;
import org.springframework.beans.factory.annotation.Value;
import org.springframework.stereotype.Component;

import java.security.Key;
import java.util.Date;
import java.util.HashMap;
import java.util.Map;

@Component
public class JwtUtil {

    @Value("${jwt.secret:andiantong-secret-key-2026-secure}")
    private String secret;

    @Value("${jwt.expiration:86400000}")
    private Long expiration;

    private Key getSigningKey() {
        byte[] keyBytes = secret.getBytes();
        return Keys.hmacShaKeyFor(keyBytes);
    }

    public String generateToken(Long userId, String userType) {
        Map<String, Object> claims = new HashMap<>();
        claims.put("userId", userId);
        claims.put("userType", userType);

        return Jwts.builder()
                .setClaims(claims)
                .setSubject(String.valueOf(userId))
                .setIssuedAt(new Date())
                .setExpiration(new Date(System.currentTimeMillis() + expiration))
                .signWith(getSigningKey(), SignatureAlgorithm.HS256)
                .compact();
    }

    public Claims parseToken(String token) {
        return Jwts.parserBuilder()
                .setSigningKey(getSigningKey())
                .build()
                .parseClaimsJws(token)
                .getBody();
    }

    public Long getUserId(String token) {
        Claims claims = parseToken(token);
        return Long.valueOf(claims.getSubject());
    }

    public String getUserType(String token) {
        Claims claims = parseToken(token);
        return (String) claims.get("userType");
    }

    public boolean isTokenExpired(String token) {
        try {
            Claims claims = parseToken(token);
            return claims.getExpiration().before(new Date());
        } catch (Exception e) {
            return true;
        }
    }

    public boolean validateToken(String token) {
        try {
            parseToken(token);
            return !isTokenExpired(token);
        } catch (Exception e) {
            return false;
        }
    }
}
```

### 2.7 前端核心代码 - API请求封装

```typescript
import Taro from '@tarojs/taro';
import { BASE_URL } from '../config';

interface RequestConfig {
  url: string;
  method?: 'GET' | 'POST' | 'PUT' | 'DELETE';
  data?: any;
  header?: Record<string, string>;
  needAuth?: boolean;
}

interface ApiResponse<T = any> {
  code: number;
  message: string;
  data: T;
}

const getToken = (): string => {
  return Taro.getStorageSync('token') || '';
};

const request = async <T = any>(config: RequestConfig): Promise<ApiResponse<T>> => {
  const {
    url,
    method = 'GET',
    data,
    header = {},
    needAuth = true
  } = config;

  if (needAuth) {
    const token = getToken();
    if (token) {
      header['Authorization'] = `Bearer ${token}`;
    }
  }

  header['Content-Type'] = header['Content-Type'] || 'application/json';

  try {
    const response = await Taro.request({
      url: `${BASE_URL}${url}`,
      method,
      data,
      header
    });

    const { statusCode, data: resData } = response;

    if (statusCode === 200) {
      if (resData.code === 0) {
        return resData;
      } else if (resData.code === 401) {
        Taro.removeStorageSync('token');
        Taro.redirectTo({ url: '/pages/login/index' });
        throw new Error('登录已过期，请重新登录');
      } else {
        throw new Error(resData.message || '请求失败');
      }
    } else if (statusCode === 401) {
      Taro.removeStorageSync('token');
      Taro.redirectTo({ url: '/pages/login/index' });
      throw new Error('登录已过期');
    } else {
      throw new Error(`请求失败：${statusCode}`);
    }
  } catch (error: any) {
    if (error.message) {
      Taro.showToast({ title: error.message, icon: 'none' });
    }
    throw error;
  }
};

export const get = <T = any>(url: string, data?: any, needAuth = true) =>
  request<T>({ url, method: 'GET', data, needAuth });

export const post = <T = any>(url: string, data?: any, needAuth = true) =>
  request<T>({ url, method: 'POST', data, needAuth });

export const put = <T = any>(url: string, data?: any, needAuth = true) =>
  request<T>({ url, method: 'PUT', data, needAuth });

export const del = <T = any>(url: string, data?: any, needAuth = true) =>
  request<T>({ url, method: 'DELETE', data, needAuth });

export default request;
```
