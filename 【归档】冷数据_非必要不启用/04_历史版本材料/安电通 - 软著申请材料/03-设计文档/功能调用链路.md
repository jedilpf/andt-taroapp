```typescript
// 前端 - publishTask
const publishTask = async () => {
  const formData = {
    title: title.value,
    description: desc.value,
    reward: parseFloat(reward.value),
    address: address.value,
    images: imageUrls
  }
  
  try {
```

```java
// 后端 - TaskController.publish
@PostMapping("/publish")
public Result<TaskVO> publish(@RequestBody @Valid PublishTaskDTO dto,
                               @RequestHeader("Authorization") String token) {
    Long publisherId = JwtUtil.getUserId(token);
    Task task = taskService.publish(dto, publisherId);
    return Result.success(convertToVO(task));
}
```

**完整调用链路：**

```mermaid
sequenceDiagram

    participant A as api/task.ts

    U->>D: 5. 确认接单
    
    rect rgb(240, 248, 255)
        Note over D,R: 前端请求
        D->>A: 6. acceptTask(taskId)
        A->>R: 7. POST /task/{taskId}/accept
        R->>R: 8. 添加Token
    end
    
    R->>G: 9. 发送请求
    G->>G: 10. 验证Token，提取acceptorId
    G->>TS: 11. 转发请求
    TS->>C: 12. TaskController.accept(taskId)
    C->>C: 13. 从Token获取acceptorId
    
    rect rgb(240, 255, 240)
        Note over C,DB: 业务处理
        C->>SV: 14. taskService.accept(taskId, acceptorId)
        SV->>M: 15. getById(taskId)
        M->>DB: 16. SELECT * WHERE id=?
        DB-->>M: 17. Task记录
        M-->>SV: 18. Task对象
        
        alt 任务状态检查
            SV->>SV: 19. 检查status=PENDING
            alt status != PENDING
                SV-->>C: throw Exception("任务已被接单")
            else status == PENDING
                SV->>SV: 20. 检查不是发布者自己
                alt publisherId == acceptorId
                    SV-->>C: throw Exception("不能接自己的任务")
                else 可以接单
                    SV->>SV: 21. task.setAcceptorId(acceptorId)
                    SV->>SV: 22. task.setStatus("ACCEPTED")
                    SV->>SV: 23. task.setAcceptTime(now)
                    SV->>M: 24. updateById(task)
                    M->>DB: 25. UPDATE t_task SET status=?, acceptor_id=?, accept_time=?
                    DB-->>M: 26. 影响行数=1
                end
            end
        end
    end
    
    rect rgb(255, 250, 240)
        Note over SV,OS: 创建订单（可选）
        SV->>OS: 27. Feign调用createOrder(taskId, acceptorId)
        OS-->>SV: 28. 订单信息
    end
    
    M-->>SV: 29. 更新成功
    SV-->>C: 30. Task对象
    C-->>TS: 31. Result.success(true)
    TS-->>G: 32. JSON响应
    G-->>R: 33. 返回
    R-->>A: 34. success
    A-->>D: 35. 完成
    
    D->>D: 36. showToast('接单成功')
    D->>D: 37. 更新任务状态显示
    D->>D: 38. 隐藏"接单"按钮
    D-->>U: 39. 显示"进行中"
```

| **派单** | `POST /api/dispatch/auto` | 触发智能派单算法，匹配附近电工 | `{orderId: "123", loc: {lat, lng}}` |
| **推送** | `WebSocket` / `SSE` | **关键**：向用户端实时推送订单状态变更 (接单、到达、完成) | Event: `ORDER_UPDATE` <br/> Payload: `{status: "ARRIVED"}` |
| **动作** | `POST /api/order/{id}/timeline` | 记录关键节点 (如用户点击开始检测)，用于审计 | `{action: "USER_START_SCAN"}` |
| **报告** | `GET /api/report/{id}` | 获取详细的检测报告数据（评分、隐患项、建议） | Response: `{score: 78, issues: [...]}` |
| **物料** | `GET /api/product/materials` | 获取当前可用的整改物料清单及价格 | Response: `[{id: 1, name: "漏保", price: 85}]` |

> **注意**：目前前端 `InspectionFlow.tsx` 使用 `setTimeout` 模拟了 Socket 推送过程。对接时，需将 `useEffect` 中的定时器逻辑替换为 `useWebSocket` 或轮询逻辑。

## 通用链路说明

### 请求流程标准步骤

**前端 → 后端：**

1. **用户操作** → 触发事件
2. **页面组件** → 调用API方法
3. **API封装** → 调用request
4. **Request工具** → 添加Token、Headers
5. **发送请求** → Gateway
6. **Gateway** → 路由匹配、认证
7. **微服务** → Controller接收
8. **Controller** → 参数校验、调用Service
9. **Service** → 业务逻辑、调用Mapper
10. **Mapper** → 执行SQL
11. **MySQL** → 返回数据

**后端 → 前端：**

12. **Mapper** → 返回Entity
13. **Service** → 返回业务对象
14. **Controller** → 转换为VO、包装Result
15. **微服务** → 返回JSON
16. **Gateway** → 转发响应
17. **Request** → 响应拦截、错误处理
18. **API封装** → 返回业务数据
19. **页面组件** → 更新状态、渲染UI
20. **用户** → 看到结果

## 错误处理链路

### 统一错误处理流程

```mermaid
sequenceDiagram
    participant P as 页面
    participant R as Request
    participant G as Gateway
    participant S as Service
    participant EH as ExceptionHandler
    
    P->>R: 1. 发起请求
    R->>G: 2. 转发
    G->>S: 3. 调用服务
    
    alt 业务异常
        S->>S: 4. throw BusinessException
        S->>EH: 5. 捕获异常
        EH->>EH: 6. 构造错误响应
        EH-->>G: 7. {"code": 1002, "message": "..."}
        G-->>R: 8. 返回错误
        R->>R: 9. 检查code != 0
        R->>R: 10. showToast(message)
        R-->>P: 11. throw error
        P->>P: 12. catch处理
    else Token过期
        G->>G: 13. 验证Token失败
        G-->>R: 14. 401 Unauthorized
        R->>R: 15. 清除Token
        R->>R: 16. 跳转登录页
    else 网络错误
        R->>R: 17. catch网络异常
        R->>R: 18. showToast('网络错误')
    end
```
