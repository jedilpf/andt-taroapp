安电通数据库设计

未来申活（上海）数字科技有限公司

一、数据库规划

1.1 数据库策略

库名说明
andiantong安电通业务主库，包含所有业务表

1.2 表设计原则

  • 主键使用BIGINT自增
  • 必须有created_at、updated_at字段
  • 状态字段使用TINYINT枚举
  • 外键关系通过索引维护，不建物理外键
  • 字段添加comment注释

二、用户模块

2.1 用户表 (user)

CREATE TABLE `user` ( `id` BIGINT NOT NULL AUTO_INCREMENT COMMENT '用户ID', `phone` VARCHAR(11) NOT NULL COMMENT '手机号', `password` VARCHAR(64) NOT NULL COMMENT '密码(BCrypt加密)', `nickname` VARCHAR(50) DEFAULT NULL COMMENT '昵称', `avatar` VARCHAR(255) DEFAULT NULL COMMENT '头像URL', `gender` TINYINT DEFAULT 0 COMMENT '性别:0未知/1男/2女', `role` TINYINT NOT NULL DEFAULT 1 COMMENT '角色:1普通用户/2电工', `status` TINYINT NOT NULL DEFAULT 1 COMMENT '状态:0禁用/1正常', `openid` VARCHAR(64) DEFAULT NULL COMMENT '微信openid', `created_at` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间', `updated_at` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间', PRIMARY KEY (`id`), UNIQUE KEY `uk_phone` (`phone`), UNIQUE KEY `uk_openid` (`openid`), KEY `idx_role` (`role`), KEY `idx_status` (`status`) ) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='用户表';

2.2 电工表 (electrician)

CREATE TABLE `electrician` ( `id` BIGINT NOT NULL AUTO_INCREMENT COMMENT '电工ID', `user_id` BIGINT NOT NULL COMMENT '关联用户ID', `name` VARCHAR(50) NOT NULL COMMENT '姓名', `cert_no` VARCHAR(30) DEFAULT NULL COMMENT '电工证号', `cert_level` TINYINT DEFAULT 1 COMMENT '等级:1初级/2中级/3高级', `service_area` VARCHAR(200) DEFAULT NULL COMMENT '服务区域(JSON)', `skills` VARCHAR(500) DEFAULT NULL COMMENT '技能标签(JSON)', `rating` DECIMAL(3,2) DEFAULT 5.00 COMMENT '评分', `order_count` INT DEFAULT 0 COMMENT '接单数', `income_total` DECIMAL(12,2) DEFAULT 0.00 COMMENT '累计收入', `status` TINYINT NOT NULL DEFAULT 1 COMMENT '状态:0禁用/1空闲/2工作中', `verified` TINYINT DEFAULT 0 COMMENT '认证状态:0未认证/1已认证', `created_at` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP, `updated_at` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP, PRIMARY KEY (`id`), UNIQUE KEY `uk_user_id` (`user_id`), KEY `idx_status` (`status`), KEY `idx_rating` (`rating`), KEY `idx_verified` (`verified`) ) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='电工表';

2.3 电工位置表 (electrician_location)

CREATE TABLE `electrician_location` ( `id` BIGINT NOT NULL AUTO_INCREMENT, `elec_id` BIGINT NOT NULL COMMENT '电工ID', `latitude` DECIMAL(10,6) NOT NULL COMMENT '纬度', `longitude` DECIMAL(10,6) NOT NULL COMMENT '经度', `updated_at` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP, PRIMARY KEY (`id`), UNIQUE KEY `uk_elec_id` (`elec_id`), KEY `idx_location` (`latitude`, `longitude`) ) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='电工位置表';

三、订单模块

3.1 订单表 (order)

CREATE TABLE `order` ( `id` BIGINT NOT NULL AUTO_INCREMENT COMMENT '订单ID', `order_no` VARCHAR(32) NOT NULL COMMENT '订单号', `user_id` BIGINT NOT NULL COMMENT '用户ID', `elec_id` BIGINT DEFAULT NULL COMMENT '电工ID', `service_type` TINYINT NOT NULL COMMENT '服务类型:1维修/2安装/3安检', `service_name` VARCHAR(50) NOT NULL COMMENT '服务名称', `description` VARCHAR(500) DEFAULT NULL COMMENT '问题描述', `address_id` BIGINT NOT NULL COMMENT '地址ID', `address_text` VARCHAR(200) NOT NULL COMMENT '地址文本', `contact_phone` VARCHAR(11) NOT NULL COMMENT '联系电话', `appointment_time` DATETIME DEFAULT NULL COMMENT '预约时间', `amount` DECIMAL(10,2) NOT NULL COMMENT '订单金额', `status` TINYINT NOT NULL DEFAULT 0 COMMENT '状态:-1取消/0待支付/10已支付/20已接单/30服务中/40已完成/50已评价', `cancel_reason` VARCHAR(200) DEFAULT NULL COMMENT '取消原因', `created_at` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP, `updated_at` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP, PRIMARY KEY (`id`), UNIQUE KEY `uk_order_no` (`order_no`), KEY `idx_user_id` (`user_id`), KEY `idx_elec_id` (`elec_id`), KEY `idx_status` (`status`), KEY `idx_created_at` (`created_at`) ) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='订单表';

3.2 订单状态枚举

值状态说明
-1已取消订单被取消
0待支付刚创建，等待支付
10已支付支付成功，等待电工接单
20已接单电工已接单，等待服务
30服务中电工正在服务
40已完成服务完成，等待评价
50已评价用户已评价，订单结束

3.3 订单日志表 (order_log)

CREATE TABLE `order_log` ( `id` BIGINT NOT NULL AUTO_INCREMENT, `order_id` BIGINT NOT NULL COMMENT '订单ID', `action` VARCHAR(50) NOT NULL COMMENT '操作类型', `from_status` TINYINT DEFAULT NULL COMMENT '原状态', `to_status` TINYINT DEFAULT NULL COMMENT '新状态', `operator_id` BIGINT DEFAULT NULL COMMENT '操作人ID', `operator_type` TINYINT DEFAULT 1 COMMENT '操作人类型:1用户/2电工/3系统', `remark` VARCHAR(200) DEFAULT NULL COMMENT '备注', `created_at` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP, PRIMARY KEY (`id`), KEY `idx_order_id` (`order_id`), KEY `idx_created_at` (`created_at`) ) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='订单日志表';

3.4 订单评价表 (order_review)

CREATE TABLE `order_review` ( `id` BIGINT NOT NULL AUTO_INCREMENT, `order_id` BIGINT NOT NULL COMMENT '订单ID', `user_id` BIGINT NOT NULL COMMENT '用户ID', `elec_id` BIGINT NOT NULL COMMENT '电工ID', `rating` TINYINT NOT NULL COMMENT '评分:1-5', `content` VARCHAR(500) DEFAULT NULL COMMENT '评价内容', `tags` VARCHAR(200) DEFAULT NULL COMMENT '评价标签(JSON)', `created_at` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP, PRIMARY KEY (`id`), UNIQUE KEY `uk_order_id` (`order_id`), KEY `idx_elec_id` (`elec_id`), KEY `idx_rating` (`rating`) ) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='订单评价表';

3.5 支付表 (payment)

CREATE TABLE `payment` ( `id` BIGINT NOT NULL AUTO_INCREMENT, `payment_no` VARCHAR(32) NOT NULL COMMENT '支付流水号', `order_id` BIGINT NOT NULL COMMENT '订单ID', `order_no` VARCHAR(32) NOT NULL COMMENT '订单号', `user_id` BIGINT NOT NULL COMMENT '用户ID', `amount` DECIMAL(10,2) NOT NULL COMMENT '支付金额', `channel` TINYINT NOT NULL COMMENT '支付渠道:1微信/2支付宝', `status` TINYINT NOT NULL DEFAULT 0 COMMENT '状态:0待支付/1已支付/2退款中/3已退款', `transaction_id` VARCHAR(64) DEFAULT NULL COMMENT '第三方交易号', `refund_no` VARCHAR(32) DEFAULT NULL COMMENT '退款单号', `refund_amount` DECIMAL(10,2) DEFAULT NULL COMMENT '退款金额', `refund_reason` VARCHAR(200) DEFAULT NULL COMMENT '退款原因', `created_at` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP, `updated_at` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP, PRIMARY KEY (`id`), UNIQUE KEY `uk_payment_no` (`payment_no`), KEY `idx_order_id` (`order_id`), KEY `idx_user_id` (`user_id`), KEY `idx_status` (`status`) ) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='支付表';

四、商品模块

4.1 商品分类表 (category)

CREATE TABLE `category` ( `id` BIGINT NOT NULL AUTO_INCREMENT, `name` VARCHAR(50) NOT NULL COMMENT '分类名称', `parent_id` BIGINT DEFAULT 0 COMMENT '父分类ID', `level` TINYINT DEFAULT 1 COMMENT '层级', `sort` INT DEFAULT 0 COMMENT '排序', `icon` VARCHAR(255) DEFAULT NULL COMMENT '图标', `status` TINYINT DEFAULT 1 COMMENT '状态:0禁用/1正常', `created_at` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP, `updated_at` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP, PRIMARY KEY (`id`), KEY `idx_parent_id` (`parent_id`), KEY `idx_sort` (`sort`) ) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='商品分类表';

4.2 商品表 (goods)

CREATE TABLE `goods` ( `id` BIGINT NOT NULL AUTO_INCREMENT, `category_id` BIGINT NOT NULL COMMENT '分类ID', `name` VARCHAR(100) NOT NULL COMMENT '商品名称', `description` VARCHAR(500) DEFAULT NULL COMMENT '商品描述', `price` DECIMAL(10,2) NOT NULL COMMENT '售价', `original_price` DECIMAL(10,2) DEFAULT NULL COMMENT '原价', `stock` INT DEFAULT 0 COMMENT '库存', `sales` INT DEFAULT 0 COMMENT '销量', `main_image` VARCHAR(255) NOT NULL COMMENT '主图', `images` VARCHAR(1000) DEFAULT NULL COMMENT '图片列表(JSON)', `unit` VARCHAR(10) DEFAULT '件' COMMENT '单位', `status` TINYINT DEFAULT 1 COMMENT '状态:0下架/1上架', `sort` INT DEFAULT 0 COMMENT '排序', `created_at` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP, `updated_at` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP, PRIMARY KEY (`id`), KEY `idx_category_id` (`category_id`), KEY `idx_status` (`status`), KEY `idx_sales` (`sales`), KEY `idx_sort` (`sort`) ) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='商品表';

4.3 购物车表 (cart)

CREATE TABLE `cart` ( `id` BIGINT NOT NULL AUTO_INCREMENT, `user_id` BIGINT NOT NULL COMMENT '用户ID', `goods_id` BIGINT NOT NULL COMMENT '商品ID', `quantity` INT NOT NULL DEFAULT 1 COMMENT '数量', `selected` TINYINT DEFAULT 1 COMMENT '选中状态:0否/1是', `created_at` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP, `updated_at` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP, PRIMARY KEY (`id`), UNIQUE KEY `uk_user_goods` (`user_id`, `goods_id`), KEY `idx_user_id` (`user_id`) ) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='购物车表';

五、公共模块

5.1 用户地址表 (address)

CREATE TABLE `address` ( `id` BIGINT NOT NULL AUTO_INCREMENT, `user_id` BIGINT NOT NULL COMMENT '用户ID', `name` VARCHAR(50) NOT NULL COMMENT '联系人', `phone` VARCHAR(11) NOT NULL COMMENT '电话', `province` VARCHAR(50) NOT NULL COMMENT '省', `city` VARCHAR(50) NOT NULL COMMENT '市', `district` VARCHAR(50) NOT NULL COMMENT '区', `detail` VARCHAR(200) NOT NULL COMMENT '详细地址', `latitude` DECIMAL(10,6) DEFAULT NULL COMMENT '纬度', `longitude` DECIMAL(10,6) DEFAULT NULL COMMENT '经度', `is_default` TINYINT DEFAULT 0 COMMENT '默认地址:0否/1是', `created_at` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP, `updated_at` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP, PRIMARY KEY (`id`), KEY `idx_user_id` (`user_id`), KEY `idx_is_default` (`is_default`) ) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='用户地址表';

5.2 系统配置表 (config)

CREATE TABLE `config` ( `id` BIGINT NOT NULL AUTO_INCREMENT, `key` VARCHAR(50) NOT NULL COMMENT '配置键', `value` VARCHAR(500) NOT NULL COMMENT '配置值', `description` VARCHAR(200) DEFAULT NULL COMMENT '描述', `created_at` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP, `updated_at` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP, PRIMARY KEY (`id`), UNIQUE KEY `uk_key` (`key`) ) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='系统配置表';

5.3 消息通知表 (message)

CREATE TABLE `message` ( `id` BIGINT NOT NULL AUTO_INCREMENT, `user_id` BIGINT NOT NULL COMMENT '接收用户ID', `type` TINYINT NOT NULL COMMENT '类型:1系统/2订单/3支付/4活动', `title` VARCHAR(100) NOT NULL COMMENT '标题', `content` VARCHAR(500) NOT NULL COMMENT '内容', `data` VARCHAR(500) DEFAULT NULL COMMENT '附加数据(JSON)', `is_read` TINYINT DEFAULT 0 COMMENT '已读:0否/1是', `created_at` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP, PRIMARY KEY (`id`), KEY `idx_user_id` (`user_id`), KEY `idx_is_read` (`is_read`), KEY `idx_created_at` (`created_at`) ) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='消息通知表';

5.4 服务定价表 (service_price)

CREATE TABLE `service_price` ( `id` BIGINT NOT NULL AUTO_INCREMENT, `service_type` TINYINT NOT NULL COMMENT '服务类型:1维修/2安装/3安检', `service_name` VARCHAR(50) NOT NULL COMMENT '服务名称', `base_price` DECIMAL(10,2) NOT NULL COMMENT '基础价格', `unit` VARCHAR(10) DEFAULT '次' COMMENT '计价单位', `description` VARCHAR(200) DEFAULT NULL COMMENT '服务说明', `sort` INT DEFAULT 0 COMMENT '排序', `status` TINYINT DEFAULT 1 COMMENT '状态:0禁用/1正常', `created_at` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP, `updated_at` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP, PRIMARY KEY (`id`), KEY `idx_service_type` (`service_type`), KEY `idx_status` (`status`) ) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='服务定价表';

六、广告模块

6.1 广告位表 (ad_position)

CREATE TABLE `ad_position` ( `id` BIGINT NOT NULL AUTO_INCREMENT, `code` VARCHAR(50) NOT NULL COMMENT '位置编码(splash/home_banner/order_complete)', `name` VARCHAR(50) NOT NULL COMMENT '位置名称', `type` TINYINT NOT NULL COMMENT '类型:1开屏/2横幅/3插屏', `width` INT DEFAULT NULL COMMENT '推荐宽度(px)', `height` INT DEFAULT NULL COMMENT '推荐高度(px)', `max_count` INT DEFAULT 1 COMMENT '最大广告数', `status` TINYINT DEFAULT 1 COMMENT '状态:0禁用/1启用', `created_at` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP, `updated_at` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP, PRIMARY KEY (`id`), UNIQUE KEY `uk_code` (`code`) ) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='广告位表';

6.2 广告表 (ad)

CREATE TABLE `ad` ( `id` BIGINT NOT NULL AUTO_INCREMENT, `position_id` BIGINT NOT NULL COMMENT '广告位ID', `title` VARCHAR(100) NOT NULL COMMENT '广告标题', `image_url` VARCHAR(255) NOT NULL COMMENT '图片URL', `video_url` VARCHAR(255) DEFAULT NULL COMMENT '视频URL(开屏广告)', `link_url` VARCHAR(255) DEFAULT NULL COMMENT '跳转链接', `link_type` TINYINT DEFAULT 1 COMMENT '链接类型:1页面/2商品/3活动/4外部', `duration` INT DEFAULT 5 COMMENT '展示时长(秒)', `sort` INT DEFAULT 0 COMMENT '排序', `start_time` DATETIME NOT NULL COMMENT '开始时间', `end_time` DATETIME NOT NULL COMMENT '结束时间', `status` TINYINT DEFAULT 1 COMMENT '状态:0禁用/1启用', `show_count` BIGINT DEFAULT 0 COMMENT '展示次数', `click_count` BIGINT DEFAULT 0 COMMENT '点击次数', `created_at` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP, `updated_at` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP, PRIMARY KEY (`id`), KEY `idx_position_id` (`position_id`), KEY `idx_status` (`status`), KEY `idx_time` (`start_time`, `end_time`) ) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='广告表';

6.3 广告日志表 (ad_log)

CREATE TABLE `ad_log` ( `id` BIGINT NOT NULL AUTO_INCREMENT, `ad_id` BIGINT NOT NULL COMMENT '广告ID', `user_id` BIGINT DEFAULT NULL COMMENT '用户ID', `action` TINYINT NOT NULL COMMENT '行为:1展示/2点击', `device_id` VARCHAR(64) DEFAULT NULL COMMENT '设备ID', `ip` VARCHAR(50) DEFAULT NULL COMMENT 'IP地址', `created_at` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP, PRIMARY KEY (`id`), KEY `idx_ad_id` (`ad_id`), KEY `idx_created_at` (`created_at`) ) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='广告日志表';

八、索引设计原则

8.1 索引类型

类型使用场景示例
主键每表必建PRIMARY KEY (id)
唯一索引防重复数据UNIQUE KEY uk_phone
普通索引查询条件KEY idx_status
联合索引多条件查询KEY idx_user_status
外键索引关联查询KEY idx_user_id

8.2 索引设计规则

  • 选择区分度高的字段
  • 联合索引遵循最左匹配
  • 避免冗余索引
  • 定期分析慢查询优化索引
