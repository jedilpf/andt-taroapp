三、API接入方案

3.1 Claude API接入示例（后端）

后端服务封装

// src/services/ai/claude.service.ts
import Anthropic from '@anthropic-ai/sdk';

export class ClaudeService {
  private client: Anthropic;

  constructor() {
    this.client = new Anthropic({
      apiKey: process.env.ANTHROPIC_API_KEY,
    });
  }

  async generateCode(prompt: string, context?: string): Promise<string> {
    const response = await this.client.messages.create({
      model: 'claude-3-5-sonnet-20241022',
      max_tokens: 4096,
      system: `你是一位专业的全栈开发工程师，擅长React、TypeScript、Node.js。
               请根据用户需求生成高质量的代码，包含适当的注释和类型定义。`,
      messages: [
        {
          role: 'user',
          content: context
            ? `背景信息：\n${context}\n\n需求：\n${prompt}`
            : prompt,
        },
      ],
    });

    return response.content[0].type === 'text'
      ? response.content[0].text
      : '';
  }
}

错误处理与重试

export abstract class AIClientBase {
  protected async withRetry<T>(
    fn: () => Promise<T>,
    maxRetries: number = 3,
    delay: number = 1000
  ): Promise<T> {
    let lastError: Error | null = null;

    for (let i = 0; i < maxRetries; i++) {
      try {
        return await fn();
      } catch (error: any) {
        lastError = error;

        // 速率限制，等待后重试
        if (error.status === 429) {
          const retryAfter = error.headers?.['retry-after']
            ? parseInt(error.headers['retry-after']) * 1000
            : delay * (i + 1);
          await this.sleep(retryAfter);
          continue;
        }

        // 服务端错误，重试
        if (error.status >= 500) {
          await this.sleep(delay * (i + 1));
          continue;
        }

        // 客户端错误，直接抛出
        throw error;
      }
    }

    throw lastError;
  }
}

3.2 国内API接入示例

通义千问接入

// src/services/ai/qwen.service.ts
export class QwenService extends AIClientBase {
  private apiKey: string;
  private baseUrl = 'https://dashscope.aliyuncs.com/api/v1/services/aigc/text-generation/generation';

  constructor() {
    super();
    this.apiKey = process.env.QWEN_API_KEY!;
  }

  async chat(
    messages: QwenMessage[],
    model: 'qwen-max' | 'qwen-plus' | 'qwen-turbo' = 'qwen-plus'
  ): Promise<string> {
    return this.withRetry(async () => {
      const response = await fetch(this.baseUrl, {
        method: 'POST',
        headers: {
          'Authorization': `Bearer ${this.apiKey}`,
          'Content-Type': 'application/json',
        },
        body: JSON.stringify({
          model,
          input: { messages },
          parameters: {
            result_format: 'text',
            max_tokens: 2048,
            temperature: 0.7,
          },
        }),
      });

      if (!response.ok) {
        throw new Error(`通义千问API错误: ${response.status}`);
      }

      const data = await response.json();
      return data.output.text;
    });
  }
}

DeepSeek接入

// src/services/ai/deepseek.service.ts
export class DeepSeekService extends AIClientBase {
  private apiKey: string;
  private baseUrl = 'https://api.deepseek.com/v1/chat/completions';

  constructor() {
    super();
    this.apiKey = process.env.DEEPSEEK_API_KEY!;
  }

  async generateCode(
    prompt: string,
    language: string = 'typescript'
  ): Promise<string> {
    return this.withRetry(async () => {
      const response = await fetch(this.baseUrl, {
        method: 'POST',
        headers: {
          'Authorization': `Bearer ${this.apiKey}`,
          'Content-Type': 'application/json',
        },
        body: JSON.stringify({
          model: 'deepseek-coder',
          messages: [
            {
              role: 'system',
              content: `你是一位专业的${language}开发工程师。请生成高质量、可维护的代码。`,
            },
            { role: 'user', content: prompt },
          ],
          max_tokens: 4096,
          temperature: 0.3,
        }),
      });

      const data = await response.json();
      return data.choices[0].message.content;
    });
  }
}

3.3 统一AI服务网关

// src/services/ai/ai-gateway.service.ts
export class AIGatewayService {
  private claude: ClaudeService;
  private qwen: QwenService;
  private deepseek: DeepSeekService;

  // 成本预算控制
  private dailyBudget = 100; // 元/天
  private dailySpent = 0;

  // 智能路由
  async process(task: AITask): Promise<string> {
    // 检查预算
    if (this.dailySpent >= this.dailyBudget) {
      throw new Error('今日AI预算已用完，请明天再试');
    }

    // 选择服务商
    const provider = task.provider || this.selectProvider(task);

    try {
      const result = await this.executeByProvider(provider, task);
      this.recordUsage(provider, task);
      return result;
    } catch (error) {
      // 降级策略
      return this.fallback(task, provider, error);
    }
  }

  private selectProvider(task: AITask): AIProvider {
    const strategy: Record<AITaskType, AIProvider> = {
      code: 'deepseek',     // 代码生成用DeepSeek，性价比高
      chat: 'qwen',         // 对话用通义千问，中文好
      review: 'claude',     // 代码审查用Claude，质量高
      translate: 'qwen',    // 翻译用通义千问
      vision: 'doubao',     // 图像理解用豆包
    };

    return strategy[task.type];
  }
}

4.3 预算监控代码

// src/services/ai/budget-monitor.ts
export class BudgetMonitor {
  private dailyBudget: number;
  private monthlyBudget: number;
  private usageLog: UsageEntry[] = [];

  constructor(daily: number, monthly: number) {
    this.dailyBudget = daily;
    this.monthlyBudget = monthly;
  }

  checkBudget(): { allowed: boolean; reason?: string } {
    const todaySpent = this.getTodaySpent();
    const monthSpent = this.getMonthSpent();

    if (todaySpent >= this.dailyBudget) {
      return { allowed: false, reason: '今日预算已用完' };
    }

    if (monthSpent >= this.monthlyBudget) {
      return { allowed: false, reason: '本月预算已用完' };
    }

    return { allowed: true };
  }

  getUsageReport(): UsageReport {
    return {
      today: this.getTodaySpent(),
      month: this.getMonthSpent(),
      dailyBudget: this.dailyBudget,
      monthlyBudget: this.monthlyBudget,
      dailyUsageRate: this.getTodaySpent() / this.dailyBudget,
      monthlyUsageRate: this.getMonthSpent() / this.monthlyBudget,
    };
  }
}

5.1 敏感数据处理

数据分类

级别 数据类型 处理原则
绝密 用户密码、支付密钥 禁止发送给AI
机密 身份证号、银行卡号 脱敏后可发送
敏感 手机号、地址 最小化发送
普通 商品信息、评论内容 可正常发送

脱敏处理代码

// src/utils/data-mask.ts
export class DataMasker {
  // 手机号脱敏：138****8888
  static maskPhone(phone: string): string {
    return phone.replace(/(\d{3})\d{4}(\d{4})/, '$1****$2');
  }

  // 身份证脱敏：310***********1234
  static maskIdCard(idCard: string): string {
    return idCard.replace(/(.{3}).*(.{4})/, '$1***********$2');
  }

  // 银行卡脱敏：6222****1234
  static maskBankCard(card: string): string {
    return card.replace(/(.{4}).*(.{4})/, '$1****$2');
  }

  // 敏感词过滤
  static filterSensitiveWords(text: string): string {
    const sensitivePatterns = [
      /密码[：:]\s*\S+/g,
      /账号[：:]\s*\d+/g,
      /身份证[：:]\s*\d+/g,
    ];

    let filtered = text;
    for (const pattern of sensitivePatterns) {
      filtered = filtered.replace(pattern, '[已脱敏]');
    }

    return filtered;
  }
}

服务商选择策略

// src/services/ai/compliance-router.ts
export class ComplianceRouter {
  // 根据数据敏感度选择服务商
  static selectProvider(data: any): AIProvider {
    const sensitivity = this.assessSensitivity(data);

    switch (sensitivity) {
      case 'high':
        // 高敏感数据：只用国内服务
        return 'qwen';
      case 'medium':
        // 中敏感数据：优先国内，可降级
        return 'qwen';
      case 'low':
        // 低敏感数据：可用任何服务
        return 'deepseek';
      default:
        return 'qwen';
    }
  }

  private static assessSensitivity(data: any): 'high' | 'medium' | 'low' {
    const dataStr = JSON.stringify(data).toLowerCase();

    // 检测高敏感信息
    if (this.containsIdCard(dataStr) || this.containsBankCard(dataStr)) {
      return 'high';
    }

    // 检测中敏感信息
    if (this.containsPhone(dataStr) || this.containsAddress(dataStr)) {
      return 'medium';
    }

    return 'low';
  }
}

B. 常见问题

Q: Claude和GPT-4怎么选？

A: 代码开发选Claude，复杂推理选GPT-4，日常对话选DeepSeek。

Q: 如何降低API成本？

A: 1) 使用缓存 2) 精简Prompt 3) 选择合适模型 4) 批量处理 5) 设置预算限制

Q: 数据出境怎么办？

A: 优先使用国内AI服务商（通义千问、DeepSeek、豆包），用户敏感数据脱敏后发送。

Q: API调用失败怎么办？

A: 实现重试机制和降级策略，配置多服务商互备。
