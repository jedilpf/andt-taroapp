#### 错误处理与重试

```typescript
// src/services/ai/ai-client.base.ts
export abstract class AIClientBase {
  protected async withRetry<T>(
    fn: () => Promise<T>,
    maxRetries: number = 3,
    delay: number = 1000
  ): Promise<T> {
    let lastError: Error | null = null;

    for (let i = 0; i < maxRetries; i++) {
      try {
        return await fn();
      } catch (error: any) {
        lastError = error;

        // 速率限制，等待后重试
        if (error.status === 429) {
          const retryAfter = error.headers?.['retry-after'] 
            ? parseInt(error.headers['retry-after']) * 1000 
            : delay * (i + 1);
          await this.sleep(retryAfter);
          continue;
        }

        // 服务端错误，重试
        if (error.status >= 500) {
          await this.sleep(delay * (i + 1));
          continue;
        }

        // 客户端错误，直接抛出
        throw error;
      }
    }

    throw lastError;
  }

  private sleep(ms: number): Promise<void> {
    return new Promise(resolve => setTimeout(resolve, ms));
  }
}
```

```typescript
// src/services/ai/qwen.service.ts
import { AIClientBase } from './ai-client.base';

interface QwenMessage {
  role: 'system' | 'user' | 'assistant';
  content: string;
}

export class QwenService extends AIClientBase {
  private apiKey: string;
  private baseUrl = 'https://dashscope.aliyuncs.com/api/v1/services/aigc/text-generation/generation';

  constructor() {
    super();
    this.apiKey = process.env.QWEN_API_KEY!;
  }

  async chat(
    messages: QwenMessage[],
    model: 'qwen-max' | 'qwen-plus' | 'qwen-turbo' = 'qwen-plus'
  ): Promise<string> {
    return this.withRetry(async () => {
      const response = await fetch(this.baseUrl, {
        method: 'POST',
        headers: {
          'Authorization': `Bearer ${this.apiKey}`,
          'Content-Type': 'application/json',
        },
        body: JSON.stringify({
          model,
          input: {
            messages,
          },
          parameters: {
            result_format: 'text',
            max_tokens: 2048,
            temperature: 0.7,
          },
        }),
      });

      if (!response.ok) {
        throw new Error(`通义千问API错误: ${response.status}`);
      }

      const data = await response.json();
      return data.output.text;
    });
  }

  // 流式响应
  async *chatStream(
    messages: QwenMessage[],
    model: 'qwen-max' | 'qwen-plus' | 'qwen-turbo' = 'qwen-plus'
  ): AsyncGenerator<string> {
    const response = await fetch(this.baseUrl, {
      method: 'POST',
      headers: {
        'Authorization': `Bearer ${this.apiKey}`,
        'Content-Type': 'application/json',
        'X-DashScope-SSE': 'enable',
      },
      body: JSON.stringify({
        model,
        input: { messages },
        parameters: {
          incremental_output: true,
        },
      }),
    });

    const reader = response.body?.getReader();
    if (!reader) return;

    const decoder = new TextDecoder();
    while (true) {
      const { done, value } = await reader.read();
      if (done) break;

      const chunk = decoder.decode(value);
      const lines = chunk.split('\n').filter(line => line.startsWith('data:'));

      for (const line of lines) {
        const data = JSON.parse(line.slice(5));
        if (data.output?.text) {
          yield data.output.text;
        }
      }
    }
  }
}
```

### 3.3 统一AI服务网关

```typescript
// src/services/ai/ai-gateway.service.ts
import { ClaudeService } from './claude.service';
import { QwenService } from './qwen.service';
import { DeepSeekService } from './deepseek.service';

type AIProvider = 'claude' | 'qwen' | 'deepseek' | 'doubao';
type AITaskType = 'code' | 'chat' | 'review' | 'translate' | 'vision';

interface AITask {
  type: AITaskType;
  prompt: string;
  context?: string;
  provider?: AIProvider;
}

export class AIGatewayService {
  private claude: ClaudeService;
  private qwen: QwenService;
  private deepseek: DeepSeekService;

  // 成本预算控制
  private dailyBudget = 100; // 元/天
  private dailySpent = 0;

  constructor() {
    this.claude = new ClaudeService();
    this.qwen = new QwenService();
    this.deepseek = new DeepSeekService();
  }

  // 智能路由
  async process(task: AITask): Promise<string> {
    // 检查预算
    if (this.dailySpent >= this.dailyBudget) {
      throw new Error('今日AI预算已用完，请明天再试');
    }

    // 选择服务商
    const provider = task.provider || this.selectProvider(task);

    try {
      const result = await this.executeByProvider(provider, task);
      this.recordUsage(provider, task);
      return result;
    } catch (error) {
      // 降级策略
      return this.fallback(task, provider, error);
    }
  }

  private selectProvider(task: AITask): AIProvider {
    // 根据任务类型选择最优服务商
    const strategy: Record<AITaskType, AIProvider> = {
      code: 'deepseek',     // 代码生成用DeepSeek，性价比高
      chat: 'qwen',         // 对话用通义千问，中文好
      review: 'claude',     // 代码审查用Claude，质量高
      translate: 'qwen',    // 翻译用通义千问
      vision: 'doubao',     // 图像理解用豆包
    };

    return strategy[task.type];
  }

  private async executeByProvider(
    provider: AIProvider,
    task: AITask
  ): Promise<string> {
    switch (provider) {
      case 'claude':
        return this.claude.generateCode(task.prompt, task.context);
      case 'qwen':
        return this.qwen.chat([
          { role: 'user', content: task.prompt },
        ]);
      case 'deepseek':
        return this.deepseek.generateCode(task.prompt);
      default:
        throw new Error(`不支持的AI服务商: ${provider}`);
    }
  }

  // 降级策略
  private async fallback(
    task: AITask,
    failedProvider: AIProvider,
    error: any
  ): Promise<string> {
    console.error(`${failedProvider}调用失败:`, error.message);

    // 降级优先级
    const fallbackOrder: Record<AITaskType, AIProvider[]> = {
      code: ['qwen', 'claude'],
      chat: ['deepseek', 'qwen'],
      review: ['deepseek', 'qwen'],
      translate: ['deepseek', 'claude'],
      vision: ['qwen'],
    };

    const fallbacks = fallbackOrder[task.type].filter(p => p !== failedProvider);

    for (const provider of fallbacks) {
      try {
        console.log(`尝试降级到${provider}`);
        return await this.executeByProvider(provider, task);
      } catch (err) {
        console.error(`${provider}也失败:`, err);
        continue;
      }
    }

    throw new Error('所有AI服务商均不可用，请稍后重试');
  }

  private recordUsage(provider: AIProvider, task: AITask): void {
    // 记录使用情况，用于成本分析
    const costs: Record<AIProvider, number> = {
      claude: 0.02,   // 美元/千token
      qwen: 0.01,     // 人民币/千token
      deepseek: 0.001, // 人民币/千token
      doubao: 0.008,  // 人民币/千token
    };

    // 简化计算，实际应按真实token数
    this.dailySpent += costs[provider] * 10;
  }
}
```

```typescript
// 1. 缓存层
class AICache {
  private cache = new Map<string, CacheEntry>();

  async get(prompt: string): Promise<string | null> {
    const key = this.hash(prompt);
    const entry = this.cache.get(key);

    if (entry && !this.isExpired(entry)) {
      return entry.response;
    }

    return null;
  }

  async set(prompt: string, response: string): void {
    const key = this.hash(prompt);
    this.cache.set(key, {
      response,
      timestamp: Date.now(),
    });
  }

  private hash(prompt: string): string {
    // 简化哈希，实际应用MD5
    return prompt.slice(0, 64);
  }

  private isExpired(entry: CacheEntry): boolean {
    const TTL = 24 * 60 * 60 * 1000; // 24小时
    return Date.now() - entry.timestamp > TTL;
  }
}
```

```typescript
export class BudgetMonitor {
  private dailyBudget: number;
  private monthlyBudget: number;
  private usageLog: UsageEntry[] = [];

  constructor(daily: number, monthly: number) {
    this.dailyBudget = daily;
    this.monthlyBudget = monthly;
  }

  checkBudget(): { allowed: boolean; reason?: string } {
    const todaySpent = this.getTodaySpent();
    const monthSpent = this.getMonthSpent();

    if (todaySpent >= this.dailyBudget) {
      return { allowed: false, reason: '今日预算已用完' };
    }

    if (monthSpent >= this.monthlyBudget) {
      return { allowed: false, reason: '本月预算已用完' };
    }

    return { allowed: true };
  }

  recordUsage(usage: UsageEntry): void {
    this.usageLog.push({
      ...usage,
      timestamp: Date.now(),
    });

    // 持久化存储
    this.persistLog();
  }

  private getTodaySpent(): number {
    const today = new Date().setHours(0, 0, 0, 0);
    return this.usageLog
      .filter(e => e.timestamp >= today)
      .reduce((sum, e) => sum + e.cost, 0);
  }

  private getMonthSpent(): number {
    const monthStart = new Date();
    monthStart.setDate(1);
    monthStart.setHours(0, 0, 0, 0);

    return this.usageLog
      .filter(e => e.timestamp >= monthStart.getTime())
      .reduce((sum, e) => sum + e.cost, 0);
  }

  getUsageReport(): UsageReport {
    return {
      today: this.getTodaySpent(),
      month: this.getMonthSpent(),
      dailyBudget: this.dailyBudget,
      monthlyBudget: this.monthlyBudget,
      dailyUsageRate: this.getTodaySpent() / this.dailyBudget,
      monthlyUsageRate: this.getMonthSpent() / this.monthlyBudget,
    };
  }
}
```

```typescript
// src/utils/data-mask.ts
export class DataMasker {
  // 手机号脱敏：138****8888
  static maskPhone(phone: string): string {
    return phone.replace(/(\d{3})\d{4}(\d{4})/, '$1****$2');
  }

  // 身份证脱敏：310***********1234
  static maskIdCard(idCard: string): string {
    return idCard.replace(/(.{3}).*(.{4})/, '$1***********$2');
  }

  // 银行卡脱敏：6222****1234
  static maskBankCard(card: string): string {
    return card.replace(/(.{4}).*(.{4})/, '$1****$2');
  }

  // 地址脱敏：保留区级
  static maskAddress(address: string): string {
    // 提取区级以上信息
    const match = address.match(/^(.*?区)/);
    return match ? `${match[1]}***` : '***';
  }

  // 敏感词过滤
  static filterSensitiveWords(text: string): string {
    const sensitivePatterns = [
      /密码[：:]\s*\S+/g,
      /账号[：:]\s*\d+/g,
      /身份证[：:]\s*\d+/g,
    ];

    let filtered = text;
    for (const pattern of sensitivePatterns) {
      filtered = filtered.replace(pattern, '[已脱敏]');
    }

    return filtered;
  }

  // 发送给AI前的完整处理
  static prepareForAI(data: any): string {
    const jsonStr = JSON.stringify(data);
    return this.filterSensitiveWords(jsonStr);
  }
}
```

```typescript
// src/services/ai/compliance-router.ts
export class ComplianceRouter {
  // 根据数据敏感度选择服务商
  static selectProvider(data: any): AIProvider {
    const sensitivity = this.assessSensitivity(data);

    switch (sensitivity) {
      case 'high':
        // 高敏感数据：只用国内服务
        return 'qwen';
      case 'medium':
        // 中敏感数据：优先国内，可降级
        return 'qwen';
      case 'low':
        // 低敏感数据：可用任何服务
        return 'deepseek';
      default:
        return 'qwen';
    }
  }

  private static assessSensitivity(data: any): 'high' | 'medium' | 'low' {
    const dataStr = JSON.stringify(data).toLowerCase();

    // 检测高敏感信息
    if (this.containsIdCard(dataStr) || this.containsBankCard(dataStr)) {
      return 'high';
    }

    // 检测中敏感信息
    if (this.containsPhone(dataStr) || this.containsAddress(dataStr)) {
      return 'medium';
    }

    return 'low';
  }

  private static containsIdCard(text: string): boolean {
    return /\d{17}[\dXx]/.test(text);
  }

  private static containsBankCard(text: string): boolean {
    return /\d{16,19}/.test(text);
  }

  private static containsPhone(text: string): boolean {
    return /1[3-9]\d{9}/.test(text);
  }

  private static containsAddress(text: string): boolean {
    const addressKeywords = ['省', '市', '区', '县', '街道', '小区'];
    return addressKeywords.some(kw => text.includes(kw));
  }
}
```
