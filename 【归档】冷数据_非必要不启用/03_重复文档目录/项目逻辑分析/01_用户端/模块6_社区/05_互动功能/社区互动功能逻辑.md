```typescript
// 点赞状态管const useLikeState = (videoId: number) => {
  const [liked, setLiked] = useState(false);
  const [likeCount, setLikeCount] = useState(0);
  const [animating, setAnimating] = useState(false);

  // 点赞/取消点赞
  const toggleLike = async () => {
    try {
      if (liked) {
        // 取消点赞
        await communityService.unlikeVideo(videoId);
        setLiked(false);
        setLikeCount(prev => prev - 1);
      } else {
        // 点赞
        await communityService.likeVideo(videoId);
        setLiked(true);
        setLikeCount(prev => prev + 1);
        
        // 触发动画
        setAnimating(true);
        setTimeout(() => setAnimating(false), 300);
      }
    } catch (error) {
      showToast('操作失败，请重试');
    }
  };

  // 双击点赞
  const handleDoubleTap = () => {
    if (!liked) {
      toggleLike();
    }
  };

  return {
    liked,
    likeCount,
    animating,
    toggleLike,
    handleDoubleTap
  };
};
```

```typescript
// 分享到微const shareToWechat = () => {
  // 调用微信SDK
  if (window.wx) {
    window.wx.shareAppMessage({
      title: video.title,
      desc: video.description,
      link: window.location.href,
      imgUrl: video.poster
    });
  } else {
    showToast('请在微信中打开');
  }
};

// 分享到朋友圈
const shareToMoments = () => {
  if (window.wx) {
    window.wx.shareTimeline({
      title: video.title,
      link: window.location.href,
      imgUrl: video.poster
    });
  } else {
    showToast('请在微信中打开');
  }
};

// 下载视频
const downloadVideo = async () => {
  if (!video.videoUrl) {
    showToast('暂无可下载的视频');
    return;
  }
  
  try {
    showToast('开始下..');
    const response = await fetch(video.videoUrl);
    const blob = await response.blob();
    const url = URL.createObjectURL(blob);
    
    const a = document.createElement('a');
    a.href = url;
    a.download = `${video.title}.mp4`;
    a.click();
    
    URL.revokeObjectURL(url);
    showToast('下载完成');
  } catch (error) {
    showToast('下载失败，请重试');
  }
};

// 复制链接
const copyLink = () => {
  const link = window.location.href;
  navigator.clipboard.writeText(link).then(() => {
    showToast('链接已复);
  }).catch(() => {
    showToast('复制失败，请手动复制');
  });
};
```

### 8.2 限制检```typescript
// 频率限制服务
class RateLimitService {
  private limits: Map<string, number[]> = new Map();

  check(action: string, userId: string): boolean {
    const key = `${action}:${userId}`;
    const now = Date.now();
    const limit = RATE_LIMITS.find(l => l.action === action);
    
    if (!limit) return true;

    // 获取历史记录
    let history = this.limits.get(key) | [];
    
    // 过滤过期记录
    history = history.filter(time => now - time < limit.window * 1000);
    
    // 检查是否超if (history.length >= limit.limit) {
      showToast(limit.message);
      return false;
    }
    
    // 记录本次操作
    history.push(now);
    this.limits.set(key, history);
    
    return true;
  }
}

const rateLimitService = new RateLimitService();
