## 五、内容推荐策

### 5.1 推荐算法

```typescript
interface RecommendationParams {
  // 用户兴趣
  userInterests: string[];
  
  // 浏览历史
  viewHistory: string[];
  
  // 互动行为
  interactions: {
    likes: string[];
    comments: string[];
    shares: string[];
  };
  
  // 地理位置
  location: {
    province: string;
    city: string;
    district: string;
  };
  
  // 时间因素
  timeContext: {
    hour: number;
    dayOfWeek: number;
  };
}

// 推荐分数计算
const calculateScore = (video: Video, params: RecommendationParams): number => {
  let score = 0;
  
  // 兴趣匹配 (权重: 0.3)
  score += calculateInterestScore(video, params.userInterests) * 0.3;
  
  // 热度分数 (权重: 0.25)
  score += calculatePopularityScore(video) * 0.25;
  
  // 地理相关(权重: 0.2)
  score += calculateLocationScore(video, params.location) * 0.2;
  
  // 时效(权重: 0.15)
  score += calculateFreshnessScore(video) * 0.15;
  
  // 多样(权重: 0.1)
  score += calculateDiversityScore(video, params.viewHistory) * 0.1;
  
  return score;
};
```
