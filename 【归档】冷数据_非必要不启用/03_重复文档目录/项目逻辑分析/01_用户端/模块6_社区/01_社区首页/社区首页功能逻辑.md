### 3.3 视频播放

```typescript
// 视频播放组件
const VideoPlayer: React.FC<{ video: Video }> = ({ video }) => {
  const videoRef = useRef<HTMLVideoElement>(null);
  const [isPlaying, setIsPlaying] = useState(true);
  const [hasError, setHasError] = useState(false);

  // 自动播放
  useEffect(() => {
    if (videoRef.current && video.videoUrl) {
      videoRef.current.play().catch(() => {
        // 自动播放失败，显示封setHasError(true);
      });
    }
  }, [video.id]);

  // 错误处理 - 回退到静态背const handleError = () => {
    console.warn(`Video load failed for ${video.videoUrl}`);
    setHasError(true);
  };

  // 播放控制
  const togglePlay = () => {
    if (videoRef.current) {
      if (isPlaying) {
        videoRef.current.pause();
      } else {
        videoRef.current.play();
      }
      setIsPlaying(!isPlaying);
    }
  };

  return (
    <div className="absolute inset-0">
      {video.videoUrl && !hasError (
        <video
          ref={videoRef}
          src={video.videoUrl}
          poster={video.poster}
          autoPlay
          muted
          loop
          playsInline
          onError={handleError}
          className="w-full h-full object-cover"
        />
      ) : (
        // 静态背景回退
        <StaticBackground video={video} />
      )}
    </div>
  );
};
```

### 4.2 无限滚动

```typescript
// 无限滚动Hook
const useInfiniteScroll = (callback: () => void, hasMore: boolean) => {
  const observerRef = useRef<IntersectionObserver>();

  useEffect(() => {
    const options = {
      root: null,
      rootMargin: '100px',
      threshold: 0.1
    };

    observerRef.current = new IntersectionObserver((entries) => {
      if (entries[0].isIntersecting && hasMore) {
        callback();
      }
    }, options);

    return () => observerRef.current.disconnect();
  }, [callback, hasMore]);

  return observerRef;
};

// 使用示例
const loadMoreRef = useInfiniteScroll(() => {
  if (!loading && hasMore) {
    setPage(page + 1);
    loadVideos(activeCategory, page + 1);
  }
}, hasMore);
```

// 点赞状态更const handleLike = (videoId: number) => {
  const newLikedVideos = new Set(likedVideos);
  
  if (newLikedVideos.has(videoId)) {
    newLikedVideos.delete(videoId);
    // 调用取消点赞API

// 网络状态检const useNetworkStatus = () => {
  const [isOnline, setIsOnline] = useState(navigator.onLine);

  useEffect(() => {
    const handleOnline = () => setIsOnline(true);
    const handleOffline = () => setIsOnline(false);

    window.addEventListener('online', handleOnline);
    window.addEventListener('offline', handleOffline);

    return () => {
      window.removeEventListener('online', handleOnline);
      window.removeEventListener('offline', handleOffline);

### 8.2 内存管理

```typescript
// 视频缓存管理
const videoCacheManager = {
  // 最大缓存数maxCacheSize: 10,
  
  // 缓存队列
  cacheQueue: new Map(),
  
  // 添加缓存
  addCache: (videoId: number, videoElement: HTMLVideoElement) => {
    if (videoCacheManager.cacheQueue.size >= videoCacheManager.maxCacheSize) {
      // 移除最早的缓存
      const firstKey = videoCacheManager.cacheQueue.keys().next().value;
      videoCacheManager.cacheQueue.delete(firstKey);
    }
    videoCacheManager.cacheQueue.set(videoId, videoElement);
  },
  
  // 清理缓存
  clearCache: () => {
    videoCacheManager.cacheQueue.clear();
  }
};
```
