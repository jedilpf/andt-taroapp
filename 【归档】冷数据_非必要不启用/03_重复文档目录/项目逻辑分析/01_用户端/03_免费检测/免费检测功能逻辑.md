**自动重派逻辑**:
```typescript
interface DispatchConfig {
  maxRetryCount: number;      // 最大重派次retryInterval: number;      // 重派间隔（秒expandRadius: boolean;      // 是否扩大派单范围
  radiusMultiplier: number;   // 半径扩大倍数
}

const DISPATCH_CONFIGS: Record<string, DispatchConfig> = {
  FIRST_INSPECTION: {
    maxRetryCount: 0,         // 不重派，直接取消
    retryInterval: 0,
    expandRadius: false,
    radiusMultiplier: 1
  },
  PAID_INSPECTION: {
    maxRetryCount: 2,         // 重派2retryInterval: 300,       // 5分钟间隔
    expandRadius: true,
    radiusMultiplier: 1.5     // 扩大50%
  },
  URGENT_INSPECTION: {
    maxRetryCount: 3,         // 重派3retryInterval: 180,       // 3分钟间隔
    expandRadius: true,
    radiusMultiplier: 2       // 扩大100%
  }
};

const handleOrderTimeout = async (orderId: string) => {
  const order = await getOrder(orderId);
  const config = DISPATCH_CONFIGS[order.type];
  
  if (order.retryCount >= config.maxRetryCount) {
    // 达到最大重派次数，取消订单
    await cancelOrder(orderId, 'TIMEOUT');
    await notifyUser(order.userId, {
      title: '订单已取,
      content: '暂无电工接单，订单已自动取消',
      actions: ['重新预约', '联系客服']
    });
    
    // 首次检测用户给予赔if (order.isFirstInspection) {
      await sendCompensation(order.userId, {
        type: 'COUPON',
        amount: 10,
        reason: '首次检测响应超});
    }
  } else {
    // 重派订单
    const newRadius = config.expandRadius 
      order.radius * config.radiusMultiplier 
      : order.radius;
    
    await retryDispatch(orderId, {
      radius: newRadius,
      retryCount: order.retryCount + 1
    });
    
    await notifyUser(order.userId, {
      title: '正在重新派单',
      content: `暂无电工接单，已为您扩大范围重新派单（第${order.retryCount + 1}次）`
    });
  }
};
```

interface TimeoutMetrics {
  totalOrders: number;
  timeoutOrders: number;
  timeoutRate: number;
  avgResponseTime: number;
  retrySuccessRate: number;
}

const calculateTimeoutMetrics = async (
  startDate: Date,
  endDate: Date
): Promise<TimeoutMetrics> => {
  const orders = await getOrders({ startDate, endDate });
  
  const timeoutOrders = orders.filter(o => o.status === 'TIMEOUT');
  const retrySuccessOrders = orders.filter(
    o => o.retryCount > 0 && o.status === 'COMPLETED'
  );
  
  return {
    totalOrders: orders.length,
    timeoutOrders: timeoutOrders.length,
    timeoutRate: timeoutOrders.length / orders.length,
    avgResponseTime: calculateAvg(orders.map(o => o.responseTime)),
    retrySuccessRate: retrySuccessOrders.length / timeoutOrders.length
  };
};
```

#### 技术实```typescript
interface PointsRule {
  action: string;
  points: number;
  type: 'FIXED' | 'PERCENTAGE';
  limit: {
    daily: number;      // 每日上限
    total: number;      // 总上};
}

const POINTS_RULES: PointsRule[] = [
  {
    action: 'FIRST_INSPECTION',
    points: 50,
    type: 'FIXED',
    limit: { daily: 50, total: 50 }
  },
  {
    action: 'NORMAL_INSPECTION',
    points: 10,
    type: 'FIXED',
    limit: { daily: 30, total: null }
  },
  {
    action: 'RECTIFICATION_ORDER',
    points: 1,  // 消费金额×1
    type: 'PERCENTAGE',
    limit: { daily: null, total: null }
  },
  {
    action: 'INVITE_FRIEND',
    points: 100,
    type: 'FIXED',
    limit: { daily: 500, total: null }
  }
];

const calculatePointsDiscount = (
  orderAmount: number,
  userPoints: number
): {
  usedPoints: number;
  discountAmount: number;
  finalAmount: number;
} => {
  // 最高可抵扣积分 = 订单金额 × 100
  const maxUsablePoints = Math.floor(orderAmount * 100);
  
  // 实际使用积分 = min(用户积分, 最高可抵扣积分)
  const usedPoints = Math.min(userPoints, maxUsablePoints);
  
  // 抵扣金额 = 使用积分 / 100
  const discountAmount = usedPoints / 100;
  
  // 最终支付金= 订单金额 - 抵扣金额
  // 最低支付.01
  const finalAmount = Math.max(0.01, orderAmount - discountAmount);
  
  return {
    usedPoints,
    discountAmount,
    finalAmount
  };
};

const awardPoints = async (
  userId: string,
  action: string,
  amount: number
): Promise<number> => {
  const rule = POINTS_RULES.find(r => r.action === action);
  if (!rule) return 0;
  
  let points = 0;
  if (rule.type === 'FIXED') {
    points = rule.points;
  } else if (rule.type === 'PERCENTAGE' && amount) {
    points = Math.floor(amount * rule.points);
  }
  
  // 检查每日上if (rule.limit.daily) {
    const todayPoints = await getTodayPoints(userId, action);
    if (todayPoints >= rule.limit.daily) {
      points = 0;
    } else {
      points = Math.min(points, rule.limit.daily - todayPoints);
    }
  }
  
  // 检查总上if (rule.limit.total) {
    const totalPoints = await getTotalPoints(userId, action);
    if (totalPoints >= rule.limit.total) {
      points = 0;
    }
  }
  
  if (points > 0) {
    await addPoints(userId, points, action);
  }
  
  return points;
};
```

GET /api/points/balance

  totalPoints: number;
  availablePoints: number;
  frozenPoints: number;
  expiringPoints: {

  pageSize: number;

    type: 'EARN' | 'USE';

    action: string;
    description: string;
