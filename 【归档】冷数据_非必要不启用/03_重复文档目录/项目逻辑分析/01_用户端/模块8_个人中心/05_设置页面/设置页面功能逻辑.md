// 设置页面中的第三方绑const ThirdPartyBinding = () => {
  const [bindings, setBindings] = useState<BindingState>({
    wechat: true,
    alipay: false,
    apple: false,
  });
  
  const [unbindingKey, setUnbindingKey] = useState<keyof BindingState | null>(null);
  
  // 绑定/解绑操作
  const handleBind = async (key: keyof BindingState) => {
    if (bindings[key]) {
      // 解绑确认
      setUnbindingKey(key);
    } else {
      // 发起绑定
      try {
        const result = await bindThirdParty(key);
        if (result.success) {
          setBindings(prev => ({ ...prev, [key]: true }));
          showToast('绑定成功');
        }
      } catch (error) {
        showToast('绑定失败，请重试');
      }
    }
  };
  
  // 确认解绑
  const confirmUnbind = async () => {
    if (!unbindingKey) return;
    
    try {
      await unbindThirdParty(unbindingKey);
      setBindings(prev => ({ ...prev, [unbindingKey]: false }));
      showToast('解绑成功');
    } catch (error) {
      showToast('解绑失败，请重试');

      setUnbindingKey(null);

### 4.2 密码修改组件

```typescript
// 修改密码页面
const ChangePassword = () => {
  const [step, setStep] = useState<'verify' | 'newPassword' | 'success'>('verify');
  const [currentPassword, setCurrentPassword] = useState('');
  const [newPassword, setNewPassword] = useState('');
  const [confirmPassword, setConfirmPassword] = useState('');
  const [loading, setLoading] = useState(false);
  
  // 验证当前密码
  const verifyCurrentPassword = async () => {
    if (!currentPassword) {
      showToast('请输入当前密);
      return;
    }
    
    setLoading(true);
    try {
      await verifyPassword(currentPassword);
      setStep('newPassword');
    } catch (error) {
      showToast('密码错误，请重试');
    } finally {
      setLoading(false);
    }
  };
  
  // 设置新密const setNewPasswordHandler = async () => {
    // 验证新密if (!validatePassword(newPassword)) {
      showToast('密码格式不正);
      return;
    }
    
    if (newPassword !== confirmPassword) {
      showToast('两次输入的密码不一);
      return;
    }
    
    setLoading(true);
    try {
      await updatePassword(currentPassword, newPassword);
      setStep('success');
    } catch (error) {
      showToast('修改失败，请重试');
    } finally {
      setLoading(false);
    }
  };
  
  // 密码强度计算
  const getPasswordStrength = (password: string) => {
    let strength = 0;
    if (password.length >= 8) strength++;
    if (/[a-z]/.test(password)) strength++;
    if (/[A-Z]/.test(password)) strength++;
    if (/[0-9]/.test(password)) strength++;
    if (/[^a-zA-Z0-9]/.test(password)) strength++;
    
    if (strength <= 2) return { level: ', color: 'red' };
    if (strength <= 3) return { level: ', color: 'yellow' };
    return { level: ', color: 'green' };
  };
  
  return (
    <div className="p-4">
      {step === 'verify' && (
        <div className="space-y-4">
          <h3 className="text-lg font-bold">验证身份</h3>
          <input
            type="password"
            value={currentPassword}
            onChange={e => setCurrentPassword(e.target.value)}
            placeholder="请输入当前密className="w-full p-3 bg-gray-50 rounded-xl"
          />
          <button
            onClick={verifyCurrentPassword}
            disabled={loading}
            className="w-full py-4 bg-blue-500 text-white rounded-xl font-bold"
          >
            {loading '验证..' : '下一}
          </button>
        </div>
      )}
      
      {step === 'newPassword' && (
        <div className="space-y-4">
          <h3 className="text-lg font-bold">设置新密/h3>
          
          <div>
            <input
              type="password"
              value={newPassword}
              onChange={e => setNewPassword(e.target.value)}
              placeholder="请输入新密码"
              className="w-full p-3 bg-gray-50 rounded-xl"
            />
            {newPassword && (
              <div className="flex items-center mt-2">
                <span className="text-gray-500 text-sm">密码强度/span>
                <span className={`text-${getPasswordStrength(newPassword).color}-500 text-sm font-bold`}>
                  {getPasswordStrength(newPassword).level}
                </span>
              </div>
            )}
          </div>
          
          <input
            type="password"
            value={confirmPassword}
            onChange={e => setConfirmPassword(e.target.value)}
            placeholder="请确认新密码"
            className="w-full p-3 bg-gray-50 rounded-xl"
          />
          
          <div className="bg-blue-50 p-4 rounded-xl">
            <p className="text-blue-600 text-sm font-bold mb-2">密码要求</p>
            <ul className="text-blue-500 text-xs space-y-1">
              <li>至少8个字/li>
              <li>包含大小写字母和数字</li>
              <li>建议包含特殊字符</li>
            </ul>
          </div>
          
          <button
            onClick={setNewPasswordHandler}
            disabled={loading}
            className="w-full py-4 bg-blue-500 text-white rounded-xl font-bold"
          >
            {loading '修改..' : '确认修改'}
          </button>
        </div>
      )}
      
      {step === 'success' && (
        <div className="text-center py-8">
          <div className="w-20 h-20 bg-green-100 rounded-full flex items-center justify-center mx-auto mb-4">
            <Check size={40} className="text-green-500" />
          </div>
          <h3 className="text-xl font-bold mb-2">修改成功</h3>
          <p className="text-gray-500 mb-6">请使用新密码重新登录</p>
          <button
            onClick={() => navigate('/login')}
            className="w-full py-4 bg-blue-500 text-white rounded-xl font-bold"
          >
            重新登录
          </button>
        </div>
      )}
    </div>
  );
```
