### 5.2 位置服务

```typescript
// 位置服务API
const LocationService = {
  // 获取当前位置
  getCurrentLocation: (): Promise<{ lat: number; lng: number }> => {
    return new Promise((resolve, reject) => {
      if (!navigator.geolocation) {
        reject(new Error('浏览器不支持定位'));
        return;
      }
      
      navigator.geolocation.getCurrentPosition(
        position => {
          resolve({
            lat: position.coords.latitude,
            lng: position.coords.longitude,
          });
        },
        error => {
          reject(error);
        },
        {
          enableHighAccuracy: true,
          timeout: 10000,
          maximumAge: 0,
        }
      );
    });
  },
  
  // 逆地理编reverseGeocode: async (lat: number, lng: number): Promise<string> => {
    const response = await fetch(
      `https://restapi.amap.com/v3/geocode/regeokey=${AMAP_KEY}&location=${lng},${lat}`
    );
    const data = await response.json();
    return data.regeocode.formatted_address;
  },
  
  // POI搜索
  searchPOI: async (keyword: string): Promise<POI[]> => {
    const response = await fetch(
      `https://restapi.amap.com/v3/place/textkey=${AMAP_KEY}&keywords=${keyword}`
    );
    const data = await response.json();
    return data.pois;
  },
};
```

### 7.3 地址验证

```typescript
// 地址验证规则
const validateAddress = (data: AddressFormData): ValidationResult => {
  const errors: string[] = [];
  
  // 姓名验证
  if (!data.name | data.name.trim().length === 0) {
    errors.push('请输入联系人姓名');
  } else if (data.name.length > 20) {
    errors.push('姓名不能超过20个字);
  }
  
  // 电话验证
  const phoneRegex = /^1[3-9]\d{9}$/;
  if (!data.phone) {
    errors.push('请输入联系电);
  } else if (!phoneRegex.test(data.phone)) {
    errors.push('请输入正确的手机);
  }
  
  // 地址验证
  if (!data.address | data.address.trim().length === 0) {
    errors.push('请选择地址');
  } else if (data.address.length > 100) {
    errors.push('地址不能超过100个字);
  }
  
  // 门牌号验if (data.detail && data.detail.length > 50) {
    errors.push('门牌号不能超0个字);
  }
  
  return {
    valid: errors.length === 0,
    errors,
  };
};
```
