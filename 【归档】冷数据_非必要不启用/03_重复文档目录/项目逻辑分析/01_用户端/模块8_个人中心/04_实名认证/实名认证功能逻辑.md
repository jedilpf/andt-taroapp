// 认证流程状interface VerificationState {

// 步骤1：输入基本信const Step1BasicInfo = ({ 
  data, 
  onNext 
}: StepProps) => {
  const [name, setName] = useState(data.realName);
  const [idNumber, setIdNumber] = useState(data.idNumber);
  
  const validateAndNext = () => {
    // 验证姓名
    if (!name | name.length < 2) {
      showToast('请输入正确的姓名');
      return;
    }
    
    // 验证身份证号
    const idRegex = /^[1-9]\d{5}(18|19|20)\d{2}(0[1-9]|1[0-2])(0[1-9]|[12]\d|3[01])\d{3}[\dXx]$/;
    if (!idRegex.test(idNumber)) {
      showToast('请输入正确的身份证号');
      return;
    }
    
    onNext({ realName: name, idNumber });
  };
  
  return (
    <div className="p-4 space-y-4">
      <div>
        <label className="block text-gray-600 mb-2">真实姓名</label>
        <input
          type="text"
          value={name}
          onChange={e => setName(e.target.value)}

// 步骤3：人脸识const Step3FaceRecognition = ({ 
  data, 
  onNext 
}: StepProps) => {
  const videoRef = useRef<HTMLVideoElement>(null);
  const [isReady, setIsReady] = useState(false);
  const [isCapturing, setIsCapturing] = useState(false);
  
  // 启动摄像useEffect(() => {
    startCamera();
    return () => stopCamera();
  }, []);
  
  const startCamera = async () => {
    try {
      const stream = await navigator.mediaDevices.getUserMedia({
        video: { facingMode: 'user' }
      });
      if (videoRef.current) {
        videoRef.current.srcObject = stream;
        setIsReady(true);
      }
    } catch (error) {
      showToast('无法访问摄像头，请检查权);
    }
  };
  
  const stopCamera = () => {
    const stream = videoRef.current.srcObject as MediaStream;
    stream.getTracks().forEach(track => track.stop());
  };
  
  // 拍照
  const capturePhoto = async () => {
    setIsCapturing(true);
    
    try {
      const video = videoRef.current;
      if (!video) return;
      
      const canvas = document.createElement('canvas');
      canvas.width = video.videoWidth;
      canvas.height = video.videoHeight;
      const ctx = canvas.getContext('2d');
      ctx.drawImage(video, 0, 0);
      
      const dataUrl = canvas.toDataURL('image/jpeg');
      // 上传照片
      const url = await uploadImage(dataUrl);
      
      // 人脸识别验证
      const result = await verifyFace({
        idCardFront: data.idCardFront,
        facePhoto: url,
      });
      
      if (result.success) {
        onNext({ facePhoto: url });
      } else {
        showToast('人脸识别失败，请重试');
      }
    } catch (error) {
      showToast('识别失败，请重试');
    } finally {
      setIsCapturing(false);
    }
  };
  
  return (
    <div className="p-4">
      <div className="relative aspect-[3/4] bg-black rounded-xl overflow-hidden">
        <video
          ref={videoRef}
          autoPlay
          playsInline
          className="w-full h-full object-cover"
        />
        
        {/* 人脸*/}
        <div className="absolute inset-0 flex items-center justify-center">
          <div className="w-48 h-64 border-2 border-white/50 rounded-full" />
        </div>
        
        {/* 提示文字 */}
        <div className="absolute bottom-4 left-0 right-0 text-center">
          <p className="text-white text-sm">
            {isCapturing '正在识别...' : '请将面部置于框内'}
          </p>
        </div>
      </div>
      
      <button
        onClick={capturePhoto}
        disabled={!isReady | isCapturing}
        className="w-full py-4 bg-blue-500 text-white rounded-xl font-bold mt-4 disabled:opacity-50"
      >
        {isCapturing '识别..' : '开始识}
      </button>
    </div>
  );
};

// 步骤4：提交审const Step4Submit = ({

### 7.1 认证失败

```typescript
// 认证失败处理
const handleVerifyError = (error: VerifyError) => {
  switch (error.code) {
    case 'ID_NUMBER_USED':
      showToast('该身份证号已被使);
      break;
    case 'FACE_MISMATCH':
      showToast('人脸识别不通过，请确保光线充足');
      break;
    case 'ID_CARD_UNCLEAR':
      showToast('身份证照片不清晰，请重新上传');
      break;
    case 'INFO_MISMATCH':
      showToast('填写信息与证件信息不一);
      break;
    case 'ALREADY_VERIFIED':
      showToast('您已完成实名认证');
      break;
    default:
      showToast('认证失败，请重试');
  }
};
```
