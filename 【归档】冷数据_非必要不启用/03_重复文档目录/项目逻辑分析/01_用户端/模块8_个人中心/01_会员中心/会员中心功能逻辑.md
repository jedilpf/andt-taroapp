### 8.1 支付失败

```typescript
// 支付失败处理
const handlePaymentError = (error: PaymentError) => {
  switch (error.code) {
    case 'PAYMENT_CANCELLED':
      showToast('支付已取);
      break;
    case 'INSUFFICIENT_BALANCE':
      showToast('余额不足，请选择其他支付方式');
      break;
    case 'PAYMENT_TIMEOUT':
      showToast('支付超时，请重试');
      break;
    case 'NETWORK_ERROR':
      showToast('网络异常，请检查网络后重试');
      break;
    default:
      showToast('支付失败，请稍后重试');
  }
};
```
