// 优惠券使用逻辑
const useCoupon = (coupon: Coupon, orderAmount: number) => {
  // 检查状if (coupon.status === 'used') {
    return { success: false, message: '优惠券已使用' };
  }
  if (coupon.status === 'expired' | new Date() > coupon.validTo) {
    return { success: false, message: '优惠券已过期' };
  }
  
  // 检查最低消if (coupon.minAmount && orderAmount < coupon.minAmount) {
    return { 
      success: false, 
      message: `订单金额需{coupon.minAmount}元` 
    };
  }
  
  // 计算优惠金额
  let discountAmount = 0;
  if (coupon.discount) {
    // 折扣discountAmount = orderAmount * (1 - coupon.discount);
  } else {
    // 满减无门槛券
    discountAmount = coupon.amount;
  }
  
  return {
    success: true,
    discountAmount,
    finalAmount: orderAmount - discountAmount,
  };
};
```

### 5.2 倒计```typescript
// 倒计时逻辑
const useCountdown = (targetTime: Date) => {
  const [timeLeft, setTimeLeft] = useState('');
  
  useEffect(() => {
    const timer = setInterval(() => {
      const now = new Date();
      const diff = targetTime.getTime() - now.getTime();
      
      if (diff <= 0) {
        setTimeLeft('00:00:00');
        clearInterval(timer);
        return;
      }
      
      const hours = Math.floor(diff / (1000 * 60 * 60));
      const minutes = Math.floor((diff % (1000 * 60 * 60)) / (1000 * 60));
      const seconds = Math.floor((diff % (1000 * 60)) / 1000);
      
      setTimeLeft(
        `${hours.toString().padStart(2, '0')}:${
          minutes.toString().padStart(2, '0')}:${
          seconds.toString().padStart(2, '0')}`
      );
    }, 1000);
    
    return () => clearInterval(timer);
  }, [targetTime]);
  
  return timeLeft;
};
```
