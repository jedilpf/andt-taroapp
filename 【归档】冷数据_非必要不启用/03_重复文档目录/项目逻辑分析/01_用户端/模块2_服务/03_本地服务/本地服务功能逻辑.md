const [selectedItems, setSelectedItems] = useState<any[]>([]);

const toggleItem = (item: any) => {

  if (index > -1) {
    setSelectedItems(selectedItems.filter((_, i) => i !== index));

    setSelectedItems([...selectedItems, item]);
