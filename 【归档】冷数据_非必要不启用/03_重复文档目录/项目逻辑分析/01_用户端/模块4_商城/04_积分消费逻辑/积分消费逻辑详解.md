## 四、积分获取逻辑（基于代码实现）

### 4.1 签到获取积分

```typescript
// 签到获取积分 (AppContext.tsx)
const checkIn = async () => {
  const today = new Date().toDateString();
  const lastCheckIn = currentUser.lastCheckIn;
  
  // 检查是否已签到
  if (lastCheckIn === today) {
    throw new Error('今日已签);
  }
  
  // 计算连续签到天数
  const yesterday = new Date(Date.now() - 86400000).toDateString();
  const consecutiveDays = lastCheckIn === yesterday 
    (currentUser.consecutiveDays | 0) + 1 
    : 1;
  
  // 计算积分奖励
  const basePoints = 5;
  const bonusPoints = Math.min(consecutiveDays * 5, 95);  // 最多额5积分
  const totalPoints = basePoints + bonusPoints;
  
  // 更新用户数据
  await updateUser({
    points: currentUser.points + totalPoints,
    totalPoints: currentUser.totalPoints + totalPoints,
    lastCheckIn: today,
    consecutiveDays
  });
  
  // 记录积分历史
  await addPointsRecord({
    type: 'EARN',
    amount: totalPoints,
    reason: `签到奖励（连{consecutiveDays}天）`
  });
  
  return {
    points: totalPoints,
    consecutiveDays
  };
};
```

## 五、积分使用逻辑（基于代码实现）

### 5.1 积分抵扣计算

```typescript
// 积分抵扣计算 (AppContext.tsx)
const calculatePointsDeduction = (
  orderAmount: number,
  userPoints: number,
  maxDeductionRate: number = 0.1
) => {
  // 计算最大可抵扣积分（订单金额的10%const maxDeductiblePoints = Math.floor(orderAmount * maxDeductionRate * 100);
  
  // 实际可抵扣积分（取最小值）
  const actualDeductiblePoints = Math.min(userPoints, maxDeductiblePoints);
  
  // 抵扣金额00积分 = 1元）
  const deductionAmount = actualDeductiblePoints / 100;
  
  // 剩余支付金额
  const remainingAmount = Math.max(0.01, orderAmount - deductionAmount);
  
  return {
    maxDeductiblePoints,
    actualDeductiblePoints,
    deductionAmount,
    remainingAmount,
    canUsePoints: actualDeductiblePoints >= 100  // 至少100积分才能使用
  };
};
```

// 积分选择器组interface PointsSelectorProps {
  orderAmount: number;
  userPoints: number;
  onPointsChange: (points: number) => void;
}

const PointsSelector: React.FC<PointsSelectorProps> = ({
  orderAmount,
  userPoints,
  onPointsChange
}) => {
  const [usePoints, setUsePoints] = useState(false);
  const [pointsToUse, setPointsToUse] = useState(0);
  
  // 计算可抵扣信const deduction = calculatePointsDeduction(orderAmount, userPoints);
  
  // 切换使用积分
  const toggleUsePoints = () => {
    if (!usePoints) {
      // 开启积分抵扣，默认使用最大setPointsToUse(deduction.actualDeductiblePoints);
      onPointsChange(deduction.actualDeductiblePoints);
    } else {
      // 关闭积分抵扣
      setPointsToUse(0);
      onPointsChange(0);
    }
    setUsePoints(!usePoints);
  };
  
  // 调整积分数量
  const adjustPoints = (delta: number) => {
    const newPoints = Math.max(0, Math.min(pointsToUse + delta, deduction.actualDeductiblePoints));
    setPointsToUse(newPoints);
    onPointsChange(newPoints);
  };
  
  return (
    <div className="bg-white rounded-xl p-4 mb-3">
      <div className="flex items-center justify-between mb-3">
        <div className="flex items-center gap-2">
          <Gift className="text-orange-500" size={20} />
          <span className="font-bold text-slate-900">积分抵扣</span>
        </div>
        <label className="relative inline-flex items-center cursor-pointer">
          <input
            type="checkbox"
            checked={usePoints}
            onChange={toggleUsePoints}
            disabled={!deduction.canUsePoints}
            className="sr-only peer"
          />
          <div className="w-11 h-6 bg-gray-200 peer-focus:outline-none rounded-full peer peer-checked:after:translate-x-full peer-checked:after:border-white after:content-[''] after:absolute after:top-[2px] after:left-[2px] after:bg-white after:rounded-full after:h-5 after:w-5 after:transition-all peer-checked:bg-orange-500"></div>
        </label>
      </div>
      
      {usePoints && (
        <div className="space-y-3">
          <div className="flex items-center justify-between text-sm">
            <span className="text-slate-500">可用积分</span>
            <span className="font-bold text-orange-500">{userPoints}积分</span>
          </div>
          
          <div className="flex items-center justify-between text-sm">
            <span className="text-slate-500">最多可抵扣</span>
            <span className="font-bold text-slate-900">¥{deduction.deductionAmount.toFixed(2)}</span>
          </div>
          
          <div className="flex items-center gap-3">
            <button
              onClick={() => adjustPoints(-100)}
              disabled={pointsToUse <= 100}

            >
              <Minus size={16} />
            </button>
            
            <div className="flex-1 text-center">
              <div className="text-lg font-bold text-slate-900">{pointsToUse}积分</div>
              <div className="text-xs text-slate-500">= ¥{(pointsToUse / 100).toFixed(2)}</div>
            </div>
            
            <button
              onClick={() => adjustPoints(100)}
              disabled={pointsToUse >= deduction.actualDeductiblePoints}
              className="w-8 h-8 rounded-full bg-slate-100 flex items-center justify-center disabled:opacity-50"
            >
              <Plus size={16} />
            </button>
          </div>
        </div>
      )}
      
      {!deduction.canUsePoints && (
        <div className="text-xs text-slate-400 mt-2">
          积分不足100，无法使</div>
      )}
    </div>
  );
};

// 积分使用限制检const validatePointsUsage = (
  points: number,
  orderAmount: number,
  userPoints: number
): { valid: boolean; message: string } => {
  // 最低使用积if (points < 100) {
    return { valid: false, message: '最低使00积分' };
  }
  
  // 积分余额检if (points > userPoints) {
    return { valid: false, message: '积分余额不足' };
  }
  
  // 最高抵扣比例检const maxDeduction = Math.floor(orderAmount * 0.1 * 100);
  if (points > maxDeduction) {
    return { valid: false, message: `最多可使用${maxDeduction}积分` };
  }
  
  // 最低支付金额检const remainingAmount = orderAmount - points / 100;
  if (remainingAmount < 0.01) {
    return { valid: false, message: '最低支付金额.01' };
  }
  
  return { valid: true, message: '' };
};
