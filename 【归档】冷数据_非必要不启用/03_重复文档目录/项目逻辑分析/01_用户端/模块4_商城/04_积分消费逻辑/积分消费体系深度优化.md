// 任务完成处理
const completeTask = async (
  userId: string,
  action: string,
  target: string
) => {
  // 查询可完成的任务
  const tasks = await getActiveTasks(action, target);
  
  for (const task of tasks) {
    // 检查任务是否可完成
    const progress = await getTaskProgress(userId, task.id);
    
    if (progress.completed | progress.claimed) {
      continue;
    }
    
    // 更新进度
    progress.progress += 1;
    
    // 检查是否完if (progress.progress >= task.condition.count) {
      progress.completed = true;
    }
    
    await saveTaskProgress(progress);
    
    // 发送通知
    if (progress.completed && !progress.claimed) {
      await sendNotification(userId, {
        type: 'TASK_COMPLETE',
        taskName: task.name,
        reward: task.reward
      });
    }
  }
};

// 领取任务奖励
const claimTaskReward = async (userId: string, taskId: string) => {
  const progress = await getTaskProgress(userId, taskId);
  const task = await getTask(taskId);
  
  if (!progress.completed) {
    throw new Error('任务未完);
  }
  
  if (progress.claimed) {
    throw new Error('奖励已领);
  }
  
  // 发放奖励
  if (task.reward.type === 'POINTS') {
    await addPoints(userId, task.reward.amount, `任务奖励{task.name}`);
  } else if (task.reward.type === 'COUPON') {
    await claimCoupon(userId, task.reward.couponId);
  }
  
  // 更新状progress.claimed = true;
  await saveTaskProgress(progress);
  
  return task.reward;
};

// 活动积分计算
const calculateActivityPoints = async (
  order: Order,
  user: User
): Promise<number> => {
  // 获取进行中的活动
  const activities = await getActiveActivities();
  
  let bonusPoints = 0;
  
  for (const activity of activities) {
    // 检查是否在适用范围
    if (!isInScope(activity.scope, order, user)) {
      continue;
    }
    
    // 处理积分翻倍活if (activity.rules.type === 'POINTS_MULTIPLE') {
      const basePoints = Math.floor(order.totalAmount);
      const multiple = activity.rules.params.multiple | 2;
      bonusPoints += basePoints * (multiple - 1);
    }
    
    // 处理消费返积分活if (activity.rules.type === 'CONSUMPTION_REWARD') {
      const threshold = activity.rules.params.threshold;
      const rewardRate = activity.rules.params.rate;
      
      if (order.totalAmount >= threshold) {
        bonusPoints += Math.floor(order.totalAmount * rewardRate);
      }
    }
  }
  
  return bonusPoints;
};

### 5.3 积分过期处理逻辑

```typescript
// 检查并处理积分过期
const processPointsExpiration = async (userId: string) => {
  const now = Date.now();
  
  // 获取所有即将过期的积分批次
  const expiringBatches = await getPointsBatches(userId, {
    status: 'ACTIVE',
    expireAt: { $lte: now }
  });
  
  if (expiringBatches.length === 0) {
    return;
  }
  
  // 计算过期积分总数
  const expiredPoints = expiringBatches.reduce(
    (sum, batch) => sum + batch.remainingPoints, 
    0
  );
  
  // 更新批次状for (const batch of expiringBatches) {
    batch.status = 'EXPIRED';
    batch.remainingPoints = 0;
    await savePointsBatch(batch);
  }
  
  // 记录过期
  await addPointsExpireRecord(userId, {
    expiredPoints,
    expiredAmount: expiredPoints / 100,
    expiredAt: now
  });
  
  // 更新用户积分
  await updateUser(userId, {
    points: -expiredPoints,
    totalExpired: expiredPoints
  });
  
  // 发送通知
  await sendNotification(userId, {
    type: 'POINTS_EXPIRED',
    points: expiredPoints,
    value: expiredPoints / 100
  });
};

// 积分续期（使用时顺延const extendPointsExpiration = async (
  userId: string,
  pointsToUse: number
) => {
  // 获取即将过期的积分批const expiringBatches = await getPointsBatches(userId, {
    status: { $in: ['ACTIVE', 'PARTIAL'] },
    expireAt: { $gt: Date.now() }
  });
  
  // 按过期时间排expiringBatches.sort((a, b) => a.expireAt - b.expireAt);
  
  let remaining = pointsToUse;
  
  // 优先使用即将过期的积for (const batch of expiringBatches) {
    if (remaining <= 0) break;
    
    const used = Math.min(batch.remainingPoints, remaining);
    batch.remainingPoints -= used;
    batch.lastUsedAt = Date.now();
    
    // 顺延过期时间
    if (batch.remainingPoints > 0) {
      batch.expireAt = Date.now() + 365 * 24 * 60 * 60 * 1000; // 再延一} else {
      batch.status = 'USED';
    }
    
    remaining -= used;
    await savePointsBatch(batch);
  }
};

// 过期提醒
const sendExpirationReminder = async (userId: string) => {
  const now = Date.now();
  const reminderDays = [30, 15, 7, 1];
  
  for (const days of reminderDays) {
    const threshold = now + days * 24 * 60 * 60 * 1000;
    
    // 获取该时段过期的积分
    const expiringBatches = await getPointsBatches(userId, {
      expireAt: {
        $gt: threshold - 24 * 60 * 60 * 1000,
        $lte: threshold
      }
    });
    
    const expiringPoints = expiringBatches.reduce(
      (sum, batch) => sum + batch.remainingPoints,
      0
    );
    
    if (expiringPoints > 0) {
      await sendNotification(userId, {
        type: 'POINTS_EXPIRE_REMINDER',
        days,
        points: expiringPoints,
        value: expiringPoints / 100
      });
    }
  }
};

    deduct: {
      maxRate: 0.1,
      minPoints: 100
    }
  }
};

// 计算服务订单积分
const calculateServicePoints = (
  serviceType: string,
  orderAmount: number,
  options: {
    convertedFromInspection: boolean;
    hasReview: boolean;
  }
): ServicePointsCalculation => {
  const rule = servicePointsRules[serviceType];
  
  // 基础积分
  const basePoints = Math.floor(orderAmount * rule.earn.rate);
  
  // 转化奖励（仅免费检测转化）
  let conversionBonus = 0;
  if (serviceType === 'REPAIR' && options.convertedFromInspection) {
    conversionBonus = rule.earn.conversionBonus;
  }
  
  // 评价奖励
  const reviewBonus = options.hasReview rule.earn.reviewBonus : 0;
  
  // 总积const totalPoints = basePoints + conversionBonus + reviewBonus;
  
  // 抵扣计算
  const maxDeductible = Math.floor(orderAmount * rule.deduct.maxRate * 100);
  const defaultDeductible = Math.min(maxDeductible, currentUser.points);
  
  return {
    orderId: '',
    serviceType,
    orderAmount,
    basePoints,
    conversionBonus,
    reviewBonus,
    totalPoints,
    maxDeductible,
    defaultDeductible
  };
};
```
