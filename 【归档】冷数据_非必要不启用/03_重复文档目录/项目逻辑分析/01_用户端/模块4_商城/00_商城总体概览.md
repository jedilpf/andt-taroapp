### 4.3 积分抵扣计算逻辑

```typescript
interface PointsDeduction {
  orderId: string;
  orderAmount: number;  // 订单金额
  userPoints: number;  // 用户积分余额
  maxDeductionRate: number;  // 最大抵扣比例（默认10%// 计算结果
  maxDeductiblePoints: number;  // 最大可抵扣积分
  actualDeductiblePoints: number;  // 实际可抵扣积deductionAmount: number;  // 抵扣金额（元remainingAmount: number;  // 剩余支付金额
}

const calculatePointsDeduction = (
  orderAmount: number,
  userPoints: number,
  maxDeductionRate: number = 0.1
): PointsDeduction => {
  // 计算最大可抵扣积分
  const maxDeductiblePoints = Math.floor(orderAmount * maxDeductionRate * 100);
  
  // 实际可抵扣积分（取最小值）
  const actualDeductiblePoints = Math.min(userPoints, maxDeductiblePoints);
  
  // 抵扣金额00积分=1元）
  const deductionAmount = actualDeductiblePoints / 100;
  
  // 剩余支付金额
  const remainingAmount = Math.max(0.01, orderAmount - deductionAmount);
  
  return {
    orderId: '',
    orderAmount,
    userPoints,
    maxDeductionRate,
    maxDeductiblePoints,
    actualDeductiblePoints,
    deductionAmount,
    remainingAmount
  };
};

// 示例：订单金00元，用户000积分
const result = calculatePointsDeduction(100, 2000);
// maxDeductiblePoints: 1000积分00元0%×100// actualDeductiblePoints: 1000积分
// deductionAmount: 10// remainingAmount: 90```

### 4.4 积分支付流程

```typescript
// 积分支付处理
const handlePointsPayment = async (
  orderId: string,
  pointsToUse: number
) => {
  // 1. 验证积分余额
  if (currentUser.points < pointsToUse) {
    throw new Error('积分余额不足');
  }
  
  // 2. 扣除积分
  await spendPoints(pointsToUse);
  
  // 3. 更新订单
  await updateOrder(orderId, {
    pointsUsed: pointsToUse,
    pointsDeduction: pointsToUse / 100
  });
  
  // 4. 如果还有剩余金额，跳转支付页const remainingAmount = orderAmount - pointsToUse / 100;
  if (remainingAmount > 0.01) {
    // 跳转支付页面
    navigate('/payment', { state: { amount: remainingAmount } });
  } else {
    // 支付完成
    await completeOrder(orderId);
  }
};
```

// 添加商品到购物车
const addToCart = async (product: Product, quantity: number = 1) => {
  const existingItem = cart.items.find(item => item.productId === product.id);
  
  if (existingItem) {
    // 更新数量
    existingItem.quantity += quantity;
    if (existingItem.quantity > product.stock) {
      existingItem.quantity = product.stock;
    }
  } else {
    // 添加新商cart.items.push({
      id: generateId(),
      productId: product.id,
      product,
      quantity,
      selected: true,
      addedAt: Date.now()
    });
  }
  
  await saveCart(cart);
  updateCartTotal();
};

// 更新购物车统const updateCartTotal = () => {
  const selectedItems = cart.items.filter(item => item.selected);
  
  cart.totalQuantity = cart.items.reduce((sum, item) => sum + item.quantity, 0);
  cart.selectedQuantity = selectedItems.reduce((sum, item) => sum + item.quantity, 0);
  cart.totalAmount = cart.items.reduce((sum, item) => sum + item.product.price * item.quantity, 0);
  cart.selectedAmount = selectedItems.reduce((sum, item) => sum + item.product.price * item.quantity, 0);
  
  // 计算积分抵扣
  const deduction = calculatePointsDeduction(cart.selectedAmount, currentUser.points);
  cart.pointsDeduction = deduction.deductionAmount;
  cart.finalAmount = deduction.remainingAmount;
};

### 6.2 优惠券使用逻辑

```typescript
// 计算优惠券优惠金const calculateCouponDiscount = (
  coupon: Coupon,
  orderAmount: number,
  zone: string
): number => {
  // 检查最低消if (orderAmount < coupon.minAmount) {
    return 0;
  }
  
  // 检查适用范围
  if (coupon.scope === 'ZONE' && zone && !coupon.scopeIds.includes(zone)) {
    return 0;
  }
  
  // 计算优惠金额
  switch (coupon.type) {
    case 'CASH':
      return coupon.discount;
    case 'PERCENT':
      return orderAmount * (1 - coupon.discount);
    case 'FIXED':
      return coupon.discount;
    default:
      return 0;
  }
};

// 应用优惠const applyCoupon = async (couponId: string) => {
  const coupon = await getCoupon(couponId);
  
  // 验证优惠if (coupon.status !== 'ACTIVE') {
    throw new Error('优惠券已失效');
  }
  
  if (Date.now() < coupon.startTime | Date.now() > coupon.endTime) {
    throw new Error('优惠券不在有效期);
  }
  
  // 计算优惠金额
  const discount = calculateCouponDiscount(coupon, cart.selectedAmount);
  
  // 更新购物cart.appliedCoupon = couponId;
  cart.finalAmount = cart.selectedAmount - discount - cart.pointsDeduction;
  
  await saveCart(cart);
};
