## 五、话费充值逻辑

### 5.1 运营商识```typescript
// 运营商识const identifyOperator = (phoneNumber: string): 'MOBILE' | 'UNICOM' | 'TELECOM' | null => {
  const cleaned = phoneNumber.replace(/\D/g, '');
  
  if (cleaned.length !== 11) return null;
  
  const prefix = cleaned.substring(0, 3);
  
  // 移动号段
  const mobilePrefixes = ['134', '135', '136', '137', '138', '139', '150', '151', '152', '157', '158', '159', '182', '183', '184', '187', '188'];
  
  // 联通号const unicomPrefixes = ['130', '131', '132', '155', '156', '185', '186', '145', '176'];
  
  // 电信号段
  const telecomPrefixes = ['133', '153', '180', '181', '189', '177'];
  
  if (mobilePrefixes.includes(prefix)) return 'MOBILE';
  if (unicomPrefixes.includes(prefix)) return 'UNICOM';
  if (telecomPrefixes.includes(prefix)) return 'TELECOM';
  
  return null;
};
```

### 5.3 充值流```typescript
// 话费充值流const rechargePhone = async () => {
  // 验证手机if (!phoneNumber | phoneNumber.length !== 11) {
    showToast('请输入正确的手机);
    return;
  }
  
  // 验证运营const operator = identifyOperator(phoneNumber);
  if (!operator) {
    showToast('无法识别运营);
    return;
  }
  
  // 验证金额
  if (!selectedAmount | selectedAmount < 10) {
    showToast('请选择充值金);
    return;
  }
  
  setLoading(true);
  
  try {
    // 创建充值订const order = await api.createPhoneRechargeOrder({
      phoneNumber,
      operator,
      amount: selectedAmount,
      pointsUsed: pointsToUse
    });
    
    // 调用支付
    const payment = await api.payment({
      orderId: order.id,
      method: paymentMethod,
      amount: selectedAmount - (pointsToUse / 100)
    });
    
    if (payment.success) {
      // 支付成功，等待充值结await waitForRechargeResult(order.id);
      
      showToast('充值成);
      navigate('/recharge/success', { 
        state: { 
          phoneNumber,
          amount: selectedAmount,
          bonus: bonusAmount
        } 
      });
    }
  } catch (error) {
    showToast('充值失败，请重);
  } finally {
    setLoading(false);
  }
};
```

## 七、物业费充值逻辑

### 7.1 物业费计```typescript
// 物业费计const calculatePropertyFee = (
  months: number,
  amountPerMonth: number,
  pointsToUse: number
) => {
  const totalAmount = months * amountPerMonth;
  
  // 积分抵扣（最0%const maxPointsDeduction = Math.floor(totalAmount * 0.1 * 100);
  const actualPointsDeduction = Math.min(pointsToUse, maxPointsDeduction);
  const deductionAmount = actualPointsDeduction / 100;
  
  const finalAmount = totalAmount - deductionAmount;
  
  return {
    totalAmount,
    pointsDeduction: deductionAmount,
    finalAmount: Math.max(0.01, finalAmount)
  };
};
```

### 7.2 物业费充值流```typescript
// 物业费充值流const rechargePropertyFee = async () => {
  if (!selectedMonths | selectedMonths < 1) {
    showToast('请选择充值月);
    return;
  }
  
  setLoading(true);
  
  try {
    // 计算金额
    const fee = calculatePropertyFee(
      selectedMonths,
      propertyInfo.amountPerMonth,
      pointsToUse
    );
    
    // 创建充值订const order = await api.createPropertyFeeOrder({
      propertyId: propertyInfo.id,
      propertyName: propertyInfo.name,
      unitNumber: propertyInfo.unitNumber,
      months: selectedMonths,
      amountPerMonth: propertyInfo.amountPerMonth,
      totalAmount: fee.totalAmount,
      pointsUsed: pointsToUse,
      validFrom: Date.now(),
      validTo: Date.now() + selectedMonths * 30 * 24 * 60 * 60 * 1000
    });
    
    // 调用支付
    const payment = await api.payment({
      orderId: order.id,
      method: paymentMethod,
      amount: fee.finalAmount
    });
    
    if (payment.success) {
      showToast('充值成);
      navigate('/property/success', { 
        state: { 
          months: selectedMonths,
          amount: fee.totalAmount,
          validTo: order.validTo
        } 
      });
    }
  } catch (error) {
    showToast('充值失败，请重);
  } finally {
    setLoading(false);
  }
};
```

### 8.2 积分抵扣计算

```typescript
// 充值类积分抵扣计算
const calculateRechargePointsDeduction = (
  amount: number,
  userPoints: number,
  type: 'PHONE' | 'UTILITY' | 'PROPERTY'
) => {
  if (type === 'PROPERTY') {
    // 物业费：5000积分 = 1个月
    const monthsCanPay = Math.floor(userPoints / 5000);
    return {
      pointsUsed: monthsCanPay * 5000,
      deductionAmount: monthsCanPay
    };
  } else {
    // 其他00积分 = 1元，最高抵0%
    const maxDeduction = Math.floor(amount * 0.1 * 100);
    const actualDeduction = Math.min(userPoints, maxDeduction);
    
    return {
      pointsUsed: actualDeduction,
      deductionAmount: actualDeduction / 100
    };
  }
};
```
