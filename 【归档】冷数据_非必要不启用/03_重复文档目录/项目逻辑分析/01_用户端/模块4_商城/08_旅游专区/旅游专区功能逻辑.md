// 景点详情const AttractionDetailPage: React.FC<{ attractionId: string }> = ({ attractionId }) => {
  const [attraction, setAttraction] = useState<Attraction | null>(null);
  const [selectedTicket, setSelectedTicket] = useState<string | null>(null);
  const [visitDate, setVisitDate] = useState<Date | null>(null);
  const [quantity, setQuantity] = useState(1);
  
  // 计算总价
  const totalPrice = useMemo(() => {
    if (!selectedTicket) return 0;
    const ticket = attraction.ticketTypes.find(t => t.id === selectedTicket);
    return (ticket.price | 0) * quantity;
  }, [selectedTicket, quantity, attraction]);
  
  const handleBooking = () => {
    if (!selectedTicket | !visitDate) {
      showToast('请选择票种和游玩日);
      return;
    }
    
    navigate('/travel/booking', {
      state: {
        attractionId,
        ticketTypeId: selectedTicket,
        visitDate,
        quantity,
        totalPrice
      }
    });
  };
  
  return (
    <div className="min-h-screen bg-slate-50">
      {/* 景点图片轮播 */}
      <div className="relative h-64">
        <img src={attraction.images[0]} className="w-full h-full object-cover" />
        <div className="absolute bottom-0 left-0 right-0 bg-gradient-to-t from-black/60 to-transparent p-4">
          <h1 className="text-white text-xl font-bold">{attraction.name}</h1>
          <div className="flex items-center gap-2 text-white/80 text-sm mt-1">
            <Star className="text-yellow-400" size={14} />
            <span>{attraction.rating}</span>
            <span>|</span>
            <span>{attraction.reviewCount}条评/span>
          </div>
        </div>
      </div>
      
      {/* 基本信息 */}
      <div className="bg-white p-4 mb-3">
        <div className="flex items-center gap-2 text-sm text-slate-500 mb-2">
          <MapPin size={14} />
          <span>{attraction.address}</span>
        </div>
        <div className="flex items-center gap-2 text-sm text-slate-500">
          <Clock size={14} />
          <span>开放时间：{attraction.openTime}</span>
        </div>
      </div>
      
      {/* 票种选择 */}
      <div className="bg-white p-4 mb-3">
        <h2 className="font-bold mb-3">选择票种</h2>
        <div className="space-y-3">
          {attraction.ticketTypes.map(ticket => (
            <div
              key={ticket.id}
              onClick={() => setSelectedTicket(ticket.id)}
              className={`p-3 rounded-xl border-2 ${
                selectedTicket === ticket.id 
                  'border-purple-500 bg-purple-50' 
                  : 'border-slate-200'
              }`}
            >
              <div className="flex justify-between items-center">
                <div>
                  <div className="font-bold">{ticket.name}</div>
                  <div className="text-xs text-slate-500">{ticket.description}</div>
                </div>
                <div className="text-right">
                  <div className="text-purple-500 font-bold">¥{ticket.price}</div>
                </div>
              </div>
            </div>
          ))}
        </div>
      </div>
      
      {/* 日期选择 */}
      <div className="bg-white p-4 mb-3">
        <h2 className="font-bold mb-3">选择日期</h2>
        <DatePicker
          selected={visitDate}
          onChange={setVisitDate}
          minDate={new Date()}
          className="w-full"
        />
      </div>
      
      {/* 数量选择 */}
      <div className="bg-white p-4 mb-3">
        <h2 className="font-bold mb-3">购买数量</h2>
        <div className="flex items-center gap-3">
          <button
            onClick={() => setQuantity(Math.max(1, quantity - 1))}
            className="w-8 h-8 rounded-full bg-slate-100 flex items-center justify-center"
          >
            <Minus size={16} />
          </button>
          <span className="text-lg font-bold">{quantity}</span>
          <button
            onClick={() => setQuantity(quantity + 1)}
            className="w-8 h-8 rounded-full bg-slate-100 flex items-center justify-center"
          >
            <Plus size={16} />
          </button>
        </div>
      </div>
      
      {/* 底部预订*/}
      <div className="fixed bottom-0 left-0 right-0 bg-white p-4 flex items-center justify-between">
        <div>
          <div className="text-sm text-slate-500">合计</div>
          <div className="text-xl font-black text-red-500">¥{totalPrice}</div>

          onClick={handleBooking}
          className="px-8 py-3 bg-purple-500 text-white rounded-xl font-bold"
        >
          立即预订

// 预订信息填写const TravelBookingPage: React.FC = () => {
  const { attractionId, ticketTypeId, visitDate, quantity, totalPrice } = useLocation().state;
  
  const [guests, setGuests] = useState([{ name: '', phone: '', idType: 'ID_CARD', idNumber: '' }]);
  const [contactName, setContactName] = useState('');
  const [contactPhone, setContactPhone] = useState('');
  
  const handleSubmit = async () => {
    // 验证信息
    if (!contactName | !contactPhone) {
      showToast('请填写联系人信息');
      return;
    }
    
    // 创建订单
    const order = await api.createTravelOrder({
      type: 'ATTRACTION',
      itemId: attractionId,
      ticketTypeId,
      visitDate,
      quantity,
      guests,
      contactName,
      contactPhone,
      total: totalPrice
    });
    
    // 跳转支付

  };
  
  return (
    <div className="min-h-screen bg-slate-50 pb-20">
      {/* 游客信息 */}
      <div className="bg-white p-4 mb-3">
        <h2 className="font-bold mb-3">游客信息</h2>
        {guests.map((guest, index) => (
          <div key={index} className="space-y-3 mb-4">
            <input
              placeholder="姓名"
              value={guest.name}
              onChange={e => updateGuest(index, 'name', e.target.value)}
              className="w-full p-3 border rounded-xl"
            />
            <input
              placeholder="手机value={guest.phone}
              onChange={e => updateGuest(index, 'phone', e.target.value)}
              className="w-full p-3 border rounded-xl"
            />
            <input
              placeholder="身份证号"
              value={guest.idNumber}
              onChange={e => updateGuest(index, 'idNumber', e.target.value)}

        ))}
      </div>
      
      {/* 联系人信*/}
      <div className="bg-white p-4 mb-3">
        <h2 className="font-bold mb-3">联系人信/h2>

          placeholder="联系人姓value={contactName}
          onChange={e => setContactName(e.target.value)}

        />

          placeholder="联系人手机号"
          value={contactPhone}
          onChange={e => setContactPhone(e.target.value)}

        />

      {/* 提交按钮 */}

        <button

          className="w-full py-3 bg-purple-500 text-white rounded-xl font-bold"

          提交订单 ¥{totalPrice}

### 7.2 退票流```typescript
// 退const cancelOrder = async (orderId: string) => {
  const order = await api.getTravelOrder(orderId);
  
  // 检查是否可退
  const visitDate = new Date(order.visitDate);
  const now = new Date();
  const hoursBeforeVisit = (visitDate.getTime() - now.getTime()) / (1000 * 60 * 60);
  
  if (hoursBeforeVisit < 24) {
    showToast('游玩4小时内不可退);
    return;
  }
  
  // 申请退const result = await api.cancelTravelOrder(orderId);
  
  if (result.success) {
    showToast('退票成功，退款将-3个工作日内到);
    navigate('/travel/orders');
  }
};
