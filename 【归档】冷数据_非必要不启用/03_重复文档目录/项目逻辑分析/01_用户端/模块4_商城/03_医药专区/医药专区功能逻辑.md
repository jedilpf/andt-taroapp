**处方审核流程**:
```typescript
interface Prescription {
  id: string;
  userId: string;
  images: string[];
  status: 'PENDING' | 'APPROVED' | 'REJECTED';
  doctorName: string;
  hospital: string;
  diagnosis: string;
  medicines: {
    name: string;
    dosage: string;
    frequency: string;
    duration: string;
  }[];
  validUntil: number;
  reviewedBy: string;
  reviewedAt: number;
  rejectReason: string;
}

// 处方上传
const uploadPrescription = async (images: File[]): Promise<Prescription> => {
  const formData = new FormData();
  images.forEach((img, i) => formData.append(`image${i}`, img));
  
  const response = await fetch('/api/prescription/upload', {
    method: 'POST',
    body: formData
  });
  return response.json();
};

// 处方审核
const verifyPrescription = async (prescriptionId: string): Promise<{
  status: 'APPROVED' | 'REJECTED';
  reason: string;
}> => {
  const response = await fetch(`/api/prescription/verify/${prescriptionId}`);
  return response.json();
};

// 购买处方const buyPrescriptionMedicine = async (
  productId: string,
  prescriptionId: string
) => {
  const prescription = await verifyPrescription(prescriptionId);
  
  if (prescription.status === 'APPROVED') {
    // 允许购买
    createOrder({
      type: 'Medicine',
      title: '处方,
      prescriptionId,
      // ...
    });
  } else {
    // 拒绝购买
    throw new Error(prescription.reason | '处方审核未通过');
  }
};
```

  notes: string;

// 创建用药提醒
const createReminder = async (reminder: Omit<MedicationReminder, 'id' | 'completedDates'>) => {
  const response = await fetch('/api/medication/reminder', {

// 标记已服const markAsTaken = async (reminderId: string, date: number = Date.now()) => {
