// 购物车操const addToCart = (product: Product, quantity: number = 1) => {
  const existingItem = cart.items.find(item => item.productId === product.id);
  if (existingItem) {

  } else {
    cart.items.push({
      productId: product.id,
      product,
      quantity,
      selected: true,
      addedAt: Date.now()
    });
  }

### 10.3 配送时效方**问题ID**: Q003  
**问题**: 如何实现半日达配送？如何计算配送时效？  
**解决方案**:

#### Phase-1方案（固定时效）
- 显示固定配送时半日- 不计算实际配送时#### Phase-2方案（智能配送）

**配送时效计*:
```typescript
interface DeliveryTime {
  earliest: number;  // 最早送达时间
  latest: number;    // 最晚送达时间
  slots: DeliverySlot[];
}

interface DeliverySlot {
  startTime: number;
  endTime: number;
  available: boolean;
  price: number;  // 配送费
}

// 配送时效计const calculateDeliveryTime = (
  orderTime: number,
  location: Location,
  productType: string
): DeliveryTime => {
  const now = new Date(orderTime);
  const hour = now.getHours();
  
  // 半日达规if (hour < 11) {
    // 上午下单，下午送达
    return {
      earliest: setHours(now, 14).getTime(),
      latest: setHours(now, 18).getTime(),
      slots: [
        { startTime: 14, endTime: 16, available: true, price: 0 },
        { startTime: 16, endTime: 18, available: true, price: 0 }
      ]
    };
  } else if (hour < 23) {
    // 下午/晚上下单，次日上午送达
    const tomorrow = addDays(now, 1);
    return {
      earliest: setHours(tomorrow, 9).getTime(),
      latest: setHours(tomorrow, 12).getTime(),
      slots: [
        { startTime: 9, endTime: 10, available: true, price: 0 },
        { startTime: 10, endTime: 12, available: true, price: 0 }
      ]
    };
  }
  
  // 其他情况
  return {
    earliest: addDays(now, 1).getTime(),
    latest: addDays(now, 2).getTime(),
    slots: []
  };
};
```
