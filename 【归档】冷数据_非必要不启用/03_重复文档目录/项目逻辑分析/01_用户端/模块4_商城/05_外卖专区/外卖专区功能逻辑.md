## 七、配送费计算逻辑

```typescript
// 配送费计算
const calculateDeliveryFee = (
  distance: number,
  orderAmount: number,
  restaurantId: string
): number => {
  // 满额免配送费
  if (orderAmount >= 30) {
    return 0;
  }
  
  // 基础配送费
  const baseFee = 3;
  
  // 距离加价
  const distanceFee = distance > 3000 Math.floor((distance - 3000) / 1000) : 0;
  
  return baseFee + distanceFee;
};

// 起送价检const checkMinOrderAmount = (
  orderAmount: number,
  minOrderAmount: number
): boolean => {
  return orderAmount >= minOrderAmount;
};
```

### 8.2 配送追踪UI

```typescript
// 配送追踪组const DeliveryTracker: React.FC<{ orderId: string }> = ({ orderId }) => {
  const [tracking, setTracking] = useState<DeliveryTracking | null>(null);
  
  // 实时更新配送状useEffect(() => {
    const unsubscribe = subscribeDeliveryTracking(orderId, (data) => {
      setTracking(data);
    });
    
    return () => unsubscribe();
  }, [orderId]);
  
  return (
    <div className="bg-white rounded-xl p-4">
      {/* 骑手信息 */}
      {tracking.rider && (
        <div className="flex items-center gap-3 mb-4 pb-4 border-b">
          <img
            src={tracking.rider.avatar}
            className="w-12 h-12 rounded-full"
          />
          <div className="flex-1">
            <div className="font-bold">{tracking.rider.name}</div>
            <div className="text-sm text-slate-500">
              评分 {tracking.rider.rating} | 预计 {tracking.estimatedArrival}分钟送达
            </div>
          </div>
          <a href={`tel:${tracking.rider.phone}`} className="text-orange-500">
            <Phone size={24} />
          </a>
        </div>
      )}
      
      {/* 时间*/}
      <div className="space-y-3">
        {tracking.timeline.map((item, index) => (
          <div key={index} className="flex items-start gap-3">
            <div className={`w-2 h-2 rounded-full mt-2 ${
              index === 0 'bg-orange-500' : 'bg-slate-300'
            }`} />
            <div className="flex-1">
              <div className="font-bold text-slate-900">{item.description}</div>
              <div className="text-xs text-slate-400">{formatTime(item.time)}</div>
            </div>
          </div>
        ))}
      </div>
    </div>
  );
};
```
