// 添加商品到购物车
const addToCart = async (
  product: Product,
  quantity: number = 1,
  specs: { name: string; value: string }[]
) => {
  // 检查库if (product.stock < quantity) {
    throw new Error('库存不足');
  }
  
  // 查找是否已存在相同商const existingItem = cart.items.find(item => 
    item.productId === product.id && 
    JSON.stringify(item.specs) === JSON.stringify(specs)
  );
  
  if (existingItem) {
    // 更新数量
    const newQuantity = existingItem.quantity + quantity;
    if (newQuantity > product.stock) {
      throw new Error('库存不足');
    }
    existingItem.quantity = newQuantity;
  } else {
    // 添加新商cart.items.push({
      id: generateId(),
      productId: product.id,
      product: {
        id: product.id,
        title: product.title,
        price: product.price,
        originalPrice: product.originalPrice,
        image: product.image,
        stock: product.stock,
        zone: product.zone,
        category: product.category
      },
      quantity,
      selected: true,
      addedAt: Date.now(),
      specs
    });
  }
  
  // 更新购物车统updateCartStats();
  
  // 保存购物await saveCart(cart);
  
  // 显示添加成功提示
  showToast('已添加到购物);
};
```

### 5.5 更新购物车统```typescript
// 更新购物车统计信const updateCartStats = () => {
  const selectedItems = cart.items.filter(i => i.selected);
  
  // 计算数量
  cart.totalQuantity = cart.items.reduce((sum, i) => sum + i.quantity, 0);
  cart.selectedQuantity = selectedItems.reduce((sum, i) => sum + i.quantity, 0);
  
  // 计算金额
  cart.totalAmount = cart.items.reduce(
    (sum, i) => sum + i.product.price * i.quantity, 
    0
  );
  cart.selectedAmount = selectedItems.reduce(
    (sum, i) => sum + i.product.price * i.quantity, 
    0
  );
  
  // 计算会员折扣
  const memberLevel = currentUser.memberLevel | 1;
  const memberDiscountRate = getMemberDiscountRate(memberLevel);
  cart.memberDiscount = cart.selectedAmount * (1 - memberDiscountRate);
  
  // 计算积分抵扣
  const deduction = calculatePointsDeduction(
    cart.selectedAmount - cart.memberDiscount,
    currentUser.points | 0
  );
  cart.pointsDeduction = deduction.deductionAmount;
  
  // 计算最终金cart.subtotal = cart.selectedAmount;
  cart.shippingFee = calculateShippingFee(cart.selectedAmount);
  cart.discount = cart.memberDiscount + cart.pointsDeduction + (cart.appliedCoupon.discount | 0);
  cart.finalAmount = Math.max(
    0.01,
    cart.subtotal + cart.shippingFee - cart.discount
  );
};
```

### 8.1 价格计算流程

```
商品总价
    ├── 会员折扣（按等级└── 普通会员：无折└── 黄金会员8└── 铂金会员5└── 钻石会员2优惠后价├── 优惠券抵└── 满减└── 折扣└── 立减优惠券后价格
    ├── 积分抵扣
    └── 最高抵0%
    └── 100积分 = 1积分后价├── 运费
    └── 满额免运└── 会员免运最终实付金```

### 8.2 价格计算代码

```typescript
// 价格计算
const calculateFinalPrice = () => {
  let amount = cart.selectedAmount;
  
  // 1. 会员折扣
  const memberLevel = currentUser.memberLevel | 1;
  const memberDiscountRate = getMemberDiscountRate(memberLevel);
  const memberDiscount = amount * (1 - memberDiscountRate);
  amount -= memberDiscount;
  
  // 2. 优惠券抵let couponDiscount = 0;
  if (cart.appliedCoupon) {
    couponDiscount = calculateCouponDiscount(
      cart.appliedCoupon,
      amount
    );
    amount -= couponDiscount;
  }
  
  // 3. 积分抵扣
  const pointsDeduction = calculatePointsDeduction(
    amount,
    currentUser.points | 0
  );
  const actualPointsDeduction = Math.min(
    cart.pointsToUse / 100,
    pointsDeduction.deductionAmount
  );
  amount -= actualPointsDeduction;
  
  // 4. 运费
  const shippingFee = calculateShippingFee(
    cart.selectedAmount,
    memberLevel
  );
  
  // 5. 最终金const finalAmount = Math.max(0.01, amount + shippingFee);
  
  return {
    subtotal: cart.selectedAmount,
    memberDiscount,
    couponDiscount,
    pointsDeduction: actualPointsDeduction,
    shippingFee,
    finalAmount
  };
};
```
