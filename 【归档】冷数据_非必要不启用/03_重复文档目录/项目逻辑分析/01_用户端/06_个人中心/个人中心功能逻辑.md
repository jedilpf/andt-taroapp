```typescript
const calculateUserLevel = (growthValue: number): UserLevel => {
  if (growthValue >= 50000) return { level: 5, name: '黑钻会员', discount: 0.85 };
  if (growthValue >= 30000) return { level: 4, name: '铂金会员', discount: 0.9 };
  if (growthValue >= 15000) return { level: 3, name: '黄金会员', discount: 0.95 };
  if (growthValue >= 5000) return { level: 2, name: '白银会员', discount: 0.98 };
  return { level: 1, name: '普通会, discount: 1.0 };
};

const awardGrowthValue = async (
  userId: string,
  action: string,
  amount: number
): Promise<number> => {
  const rule = GROWTH_RULES.find(r => r.action === action);
  if (!rule) return 0;
  
  let growthValue = rule.growthValue;
  
  if (action === 'ORDER_COMPLETE' && amount) {
    growthValue = Math.floor(amount);
  }
  
  // 检查每日上if (rule.dailyLimit) {
    const todayGrowth = await getTodayGrowth(userId, action);
    if (todayGrowth >= rule.dailyLimit) {
      return 0;
    }
    growthValue = Math.min(growthValue, rule.dailyLimit - todayGrowth);
  }
  
  // 检查每月上if (rule.monthlyLimit) {
    const monthGrowth = await getMonthGrowth(userId, action);
    if (monthGrowth >= rule.monthlyLimit) {
      return 0;
    }
    growthValue = Math.min(growthValue, rule.monthlyLimit - monthGrowth);
  }
  
  // 发放成长await addGrowthValue(userId, growthValue, action);
  
  // 检查是否升await checkLevelUp(userId);
  
  return growthValue;
};
```

```typescript
const submitIdentityVerification = async (
```

### 11.3 实名认证安全性方**问题ID**: Q003

**解决方案**:

#### Phase-1方案（基础认证- 身份证号+姓名验证

**安全认证流程**:
```
用户输入信息

    调用公安联网核验API
    ├── 姓名身份证号比对
```

```typescript
interface RealNameVerification {
  id: string;
  userId: string;
  realName: string;
  idNumber: string;        // 加密存储
  idFrontImage: string;
  idBackImage: string;
  faceImage: string;

    passed: boolean;
    score: number;
    checkTime: Date;
  };
  
  livenessResult: {
    passed: boolean;
    score: number;
    actionType: string;  // 眨眼、张嘴、摇};

  createdAt: Date;
}

const submitRealNameVerification = async (
  verification: CreateRealNameRequest
): Promise<RealNameVerification> => {
  // 1. 加密身份证号
  const encryptedIdNumber = encrypt(verification.idNumber);
  
  // 2. 创建认证记录
  const newVerification = await createRealNameVerification({
    ...verification,
    idNumber: encryptedIdNumber
  });
  
  // 3. 调用公安联网核验
  const policeCheckResult = await policeService.verify({
    name: verification.realName,
    idNumber: verification.idNumber,
    faceImage: verification.faceImage
  });
  
  // 4. 活体检const livenessResult = await livenessService.detect(
    verification.faceImage
  );
  
  // 5. 判断认证结果
  const passed = policeCheckResult.passed && livenessResult.passed;
  
  // 6. 更新认证状await updateRealNameVerification(newVerification.id, {
    policeCheckResult,
    livenessResult,
    status: passed 'APPROVED' : 'REJECTED'
  });
  
  // 7. 更新用户状if (passed) {
    await updateUser(verification.userId, {
      identityStatus: 'VERIFIED',
      realName: verification.realName,
      idNumber: encryptedIdNumber
    });
  }
  
  // 8. 通知用户
  await notifyUser(verification.userId, {
    title: passed '实名认证成功' : '实名认证失败',
    content: passed 
      '您的实名认证已通过' 
      : '实名认证未通过，请检查信息是否正});
  
  return newVerification;
};

const encrypt = (data: string): string => {
  // 使用AES-256加密
  const key = process.env.ENCRYPTION_KEY;
  const encrypted = crypto.AES.encrypt(data, key).toString();
  return encrypted;
};

const decrypt = (encryptedData: string): string => {
  // 解密
  const key = process.env.ENCRYPTION_KEY;
  const decrypted = crypto.AES.decrypt(encryptedData, key).toString(crypto.enc.Utf8);
  return decrypted;
};
```
