const earnPoints = (amount: number, isFirstPurchaseBonus: boolean) => {
  if (currentUser) {
    const updatedUser = { 
      ...currentUser, 
      points: currentUser.points + amount,
      isFirstPurchase: isFirstPurchaseBonus false : currentUser.isFirstPurchase
    };
    setCurrentUser(updatedUser);
    sessionStorage.setItem('currentUser', JSON.stringify(updatedUser));
  }
};

const spendPoints = (points: number) => {
  if (currentUser) {
    const updatedUser = { ...currentUser, points: currentUser.points - points };
    setCurrentUser(updatedUser);
    sessionStorage.setItem('currentUser', JSON.stringify(updatedUser));
  }
};

const exchangePropertyFee = async (
  exchange: CreatePropertyExchangeRequest
): Promise<PropertyExchange> => {
  // 1. 检查积分余const userPoints = await getUserPoints(exchange.userId);
  if (userPoints < exchange.points) {
    throw new Error('积分不足');
  }
  
  // 2. 创建兑换记录
  const newExchange = await createPropertyExchange({
    ...exchange,
    status: 'PENDING'
  });
  
  // 3. 扣除积分
  await deductPoints(exchange.userId, exchange.points, 'PROPERTY_EXCHANGE');
  
  try {

    const result = await propertyService.recharge({
      propertyName: exchange.propertyName,
      ownerName: exchange.ownerName,
      phone: exchange.phone,
      amount: exchange.amount,
      transactionId: newExchange.id
    });
    
    // 5. 更新兑换状await updatePropertyExchange(newExchange.id, {
      status: 'SUCCESS',

    });
    
    // 6. 通知用户
    await notifyUser(exchange.userId, {

    });

    });
    
    throw error;
  }
};

const detectAbnormalBehavior = async (userId: string) => {

  const patterns = {

  const now = new Date();
  
  // 获取未过期的积分总和
