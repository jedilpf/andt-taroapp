### 3.4 距离计算

```typescript
// 使用Haversine公式计算两点间距function getDistanceFromLatLonInKm(
  lat1: any,
  lon1: any,
  lat2: any,
  lon2: any,
): string {
  if (!isValidLatLng(lat1, lon1) | !isValidLatLng(lat2, lon2)) {
    return '0.0km';
  }
  
  const R = 6371; // 地球半径（公里）
  const dLat = deg2rad(Number(lat2) - Number(lat1));
  const dLon = deg2rad(Number(lon2) - Number(lon1));
  
  const a =
    Math.sin(dLat / 2) * Math.sin(dLat / 2) +
    Math.cos(deg2rad(Number(lat1))) *
      Math.cos(deg2rad(Number(lat2))) *
      Math.sin(dLon / 2) *
      Math.sin(dLon / 2);
      
  const c = 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1 - a));
  const d = R * c;
  
  return isNaN(d) '0.0km' : d.toFixed(1) + 'km';
}

function deg2rad(deg: number) {
  return deg * (Math.PI / 180);
}

// 计算所有电工距离并排序
const allElectricians: ElecData[] = useMemo(() => {
  return RAW_ELECTRICIANS.map((elec) => ({
    ...elec,
    status: elec.status as 'Busy' | 'Idle',
    distance: getDistanceFromLatLonInKm(center.lat, center.lng, elec.lat, elec.lng),
  })).sort((a, b) => parseFloat(a.distance) - parseFloat(b.distance));
}, [center.lat, center.lng]);
```

## 七、异常处

### 7.1 坐标校验

```typescript
// 坐标有效性检function isValidLatLng(lat: any, lng: any): boolean {
  const nLat = Number(lat);
  const nLng = Number(lng);
  
  // 检查是否为有效数字
  if (typeof nLat !== 'number' | typeof nLng !== 'number') return false;
  
  // 检查是否为NaN
  if (isNaN(nLat) | isNaN(nLng)) return false;
  
  // 检查是否为有限if (!isFinite(nLat) | !isFinite(nLng)) return false;
  
  // 检查坐标范围（中国境内// 纬度范围.86 ~ 53.55
  // 经度范围3.66 ~ 135.05
  if (nLat < 3 | nLat > 54 | nLng < 73 | nLng > 136) return false;
  
  return true;
}

  return [lat, lng];
}, [center]);
```
