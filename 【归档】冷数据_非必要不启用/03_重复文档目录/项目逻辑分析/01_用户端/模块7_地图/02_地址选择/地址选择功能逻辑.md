### 5.1 地址数据结构

```typescript
// 地址数据结构
interface Address {
  id: string;           // 地址ID
  tag: string;          // 标签：家/公司/学校
  address: string;      // 详细地址
  detail: string;      // 门牌name: string;         // 联系人姓phone: string;        // 联系电话
  gender: string;      // 称呼：先女士
  lat: number;          // 纬度
  lng: number;          // 经度
  isDefault: boolean;  // 是否默认地址
}

// 位置数据结构
interface Location {
  lat: number;      // 纬度
  lng: number;      // 经度
  address: string;  // 地址描述
}

// POI数据结构
interface POI {
  id: string;           // POI ID
  name: string;         // 名称
  address: string;      // 地址
  lat: number;          // 纬度
  lng: number;          // 经度
  distance: string;    // 距离
  type: string;        // 类型
  tel: string;         // 电话
}
```

### 5.2 地址存储

```typescript
// AppContext中的地址管理
interface AppState {
  // ... 其他状addresses: Address[];
  
  // 地址操作方法
  addAddress: (address: Address) => void;
  updateAddress: (address: Address) => void;
  deleteAddress: (id: string) => void;
  setDefaultAddress: (id: string) => void;
}

// 地址操作实现
const addAddress = (address: Address) => {
  setAddresses(prev => [...prev, address]);
};

const updateAddress = (address: Address) => {
  setAddresses(prev => prev.map(a => a.id === address.id address : a));
};

const deleteAddress = (id: string) => {
  setAddresses(prev => prev.filter(a => a.id !== id));
};

const setDefaultAddress = (id: string) => {
  setAddresses(prev => prev.map(a => ({
    ...a,
    isDefault: a.id === id
  })));
};
```
