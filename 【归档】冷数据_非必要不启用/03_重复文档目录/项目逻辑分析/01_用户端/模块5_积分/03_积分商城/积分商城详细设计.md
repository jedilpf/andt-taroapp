interface PointsProduct {
  id: string;

  name: string;
  description: string;

```typescript
// 积分商城页面
const PointsMallPage: React.FC = () => {
  const { currentUser } = useApp();
  const [products, setProducts] = useState<PointsProduct[]>([]);
  const [categories, setCategories] = useState<ProductCategory[]>([]);
  const [currentCategory, setCurrentCategory] = useState<string>('all');
  const [sortBy, setSortBy] = useState<'default' | 'points_asc' | 'points_desc' | 'sales'>('default');
  const [loading, setLoading] = useState(false);
  const [page, setPage] = useState(1);
  const [hasMore, setHasMore] = useState(true);
  
  // 获取商品列表
  useEffect(() => {
    fetchProducts();
  }, [currentCategory, sortBy]);
  
  const fetchProducts = async (pageNum = 1) => {
    setLoading(true);
    
    try {
      const result = await api.getPointsProducts({
        category: currentCategory,
        sortBy,
        page: pageNum,
        pageSize: 20
      });
      
      if (pageNum === 1) {
        setProducts(result.products);
      } else {
        setProducts([...products, ...result.products]);
      }
      
      setHasMore(result.hasMore);
      setPage(pageNum);
    } catch (error) {
      showToast('加载失败，请重试');
    } finally {
      setLoading(false);
    }
  };
  
  // 加载更多
  const loadMore = () => {
    if (!loading && hasMore) {
      fetchProducts(page + 1);
    }
  };
  
  return (
    <div className="min-h-screen bg-slate-50">
      {/* 顶部区域 */}
      <div className="bg-gradient-to-r from-orange-500 to-orange-600 text-white p-4">
        <div className="flex items-center justify-between mb-4">
          <h1 className="text-xl font-bold">积分商城</h1>
          <Link to="/points-mall/orders" className="text-sm">
            兑换记录
          </Link>
        </div>
        
        {/* 积分余额 */}
        <div className="bg-white/20 rounded-xl p-4 backdrop-blur">
          <div className="flex items-center justify-between">
            <div>
              <div className="text-sm opacity-80">我的积分</div>
              <div className="text-3xl font-black">{currentUser.points | 0}</div>
            </div>
            <div className="text-right">
              <div className="text-sm opacity-80">约等/div>
              <div className="text-xl font-bold">¥{((currentUser.points | 0) / 100).toFixed(2)}</div>
            </div>
          </div>
        </div>
      </div>
      
      {/* 搜索区域 */}
      <div className="bg-white p-3">
        <div className="relative">
          <Search className="absolute left-3 top-1/2 -translate-y-1/2 text-slate-400" size={18} />
          <input
            type="text"
            placeholder="搜索商品"
            className="w-full pl-10 pr-4 py-2 bg-slate-100 rounded-lg text-sm"
          />
        </div>
      </div>
      
      {/* 分类区域 */}
      <div className="bg-white border-b">
        <div className="flex overflow-x-auto gap-4 p-3">
          <button
            onClick={() => setCurrentCategory('all')}
            className={`flex-shrink-0 px-4 py-2 rounded-full text-sm font-bold ${
              currentCategory === 'all'
                'bg-orange-500 text-white'
                : 'bg-slate-100 text-slate-600'
            }`}
          >
            全部
          </button>
          {categories.map(category => (
            <button
              key={category.id}
              onClick={() => setCurrentCategory(category.id)}
              className={`flex-shrink-0 px-4 py-2 rounded-full text-sm font-bold ${
                currentCategory === category.id
                  'bg-orange-500 text-white'
                  : 'bg-slate-100 text-slate-600'
              }`}
            >
              {category.name}
            </button>
          ))}
        </div>
      </div>
      
      {/* 筛选排*/}
      <div className="bg-white border-b sticky top-0 z-10">
        <div className="flex gap-4 p-3">
          {[
            { key: 'default', label: '综合' },
            { key: 'points_asc', label: '积分},
            { key: 'points_desc', label: '积分},
            { key: 'sales', label: '销}
          ].map(item => (
            <button
              key={item.key}
              onClick={() => setSortBy(item.key as any)}
              className={`text-sm font-bold ${
                sortBy === item.key 'text-orange-500' : 'text-slate-600'
              }`}
            >
              {item.label}
            </button>
          ))}
        </div>
      </div>
      
      {/* 商品列表 */}
      <div className="p-3">
        <div className="grid grid-cols-2 gap-3">
          {products.map(product => (
            <ProductCard key={product.id} product={product} />
          ))}
        </div>
        
        {/* 加载更多 */}
        {hasMore && (
          <div className="text-center py-4">
            <button
              onClick={loadMore}
              disabled={loading}
              className="text-sm text-slate-500"
            >
              {loading '加载..' : '加载更多'}
            </button>
          </div>
        )}
        
        {/* 没有更多 */}
        {!hasMore && products.length > 0 && (
          <div className="text-center py-4 text-sm text-slate-400">
            没有更多商品</div>
        )}
        
        {/* 空状*/}
        {!loading && products.length === 0 && (
          <div className="text-center py-12">
            <Package className="mx-auto text-slate-300 mb-3" size={48} />
            <div className="text-slate-400">暂无商品</div>
          </div>
        )}
      </div>
    </div>
  );
};

// 商品卡片组件
const ProductCard: React.FC<{ product: PointsProduct }> = ({ product }) => {
  const { currentUser } = useApp();
  const canExchange = currentUser.points >= product.points;
  
  return (
    <div className="bg-white rounded-xl overflow-hidden shadow-sm">
      {/* 商品图片 */}
      <div className="relative aspect-square">
        <img
          src={product.mainImage}
          alt={product.name}
          className="w-full h-full object-cover"
        />
        
        {/* 标签 */}
        {product.tags.length > 0 && (
          <div className="absolute top-2 left-2">
            {product.tags.map(tag => (
              <span
                key={tag}
                className="inline-block px-2 py-1 bg-orange-500 text-white text-xs rounded mr-1"
              >
                {tag}
              </span>
            ))}
          </div>
        )}
        
        {/* 库存不足 */}
        {product.stock < 10 && product.stock > 0 && (
          <div className="absolute top-2 right-2 px-2 py-1 bg-red-500 text-white text-xs rounded">
            仅剩{product.stock}</div>
        )}
        
        {/* 已售*/}
        {product.stock === 0 && (
          <div className="absolute inset-0 bg-black/50 flex items-center justify-center">
            <span className="text-white font-bold">已售/span>
          </div>
        )}
      </div>
      
      {/* 商品信息 */}
      <div className="p-3">
        <h3 className="font-bold text-sm text-slate-900 mb-2 line-clamp-2">
          {product.name}
        </h3>
        
        {/* 积分价格 */}
        <div className="flex items-center justify-between mb-2">
          <div className="flex items-baseline gap-1">
            <span className="text-orange-500 font-black text-lg">{product.points}</span>
            <span className="text-xs text-slate-500">积分</span>
          </div>
          {product.originalPrice && (
            <span className="text-xs text-slate-400 line-through">
              ¥{product.originalPrice}
            </span>
          )}
        </div>
        
        {/* 兑换按钮 */}
        <button
          disabled={!canExchange | product.stock === 0}
          className={`w-full py-2 rounded-lg text-sm font-bold ${
            !canExchange | product.stock === 0
              'bg-slate-200 text-slate-400'
              : 'bg-orange-500 text-white'
          }`}
        >
          {product.stock === 0 '已售: canExchange '立即兑换' : '积分不足'}
        </button>
      </div>
    </div>
  );
};

    fetchProductDetail();

// 兑换订单
interface ExchangeOrder {
  id: string;
  orderNo: string;

  userName: string;
  userPhone: string;
  
  // 商品信息
  product: {
    id: string;
    name: string;
    image: string;
    type: 'PHYSICAL' | 'VIRTUAL' | 'COUPON' | 'SPECIAL';
  };
  
  // 数量和积quantity: number;

  shippingPoints: number;

    district: string;
  };
  
  // 物流信息
  logistics: {
    company: string;
    trackingNo: string;
    status: string;
    updatedAt: number;
  };
  
  // 虚拟商品信息
  virtualInfo: {
    cardNo: string;
    password: string;
    expireAt: number;
  };

    couponId: string;
    couponCode: string;
    value: number;
    minAmount: number;
    expireAt: number;
  };
  
  // 状status: 'PENDING' | 'PROCESSING' | 'SHIPPED' | 'COMPLETED' | 'CANCELLED';

  createdAt: number;

  remark: string;
}

```typescript
// 商品添加/编辑表单
interface ProductFormData {
  // 基本信息
  name: string;
  description: string;
  images: string[];
  mainImage: string;
  
  // 分类
  categoryId: string;
  
  // 积分价格
  points: number;
  originalPrice: number;
  
  // 库存
  stock: number;
  
  // 限制
  limits: {
    maxPerUser: number;
    minMemberLevel: number;
    dailyLimit: number;
  };
  
  // 配shipping: {
    required: boolean;
    fee: number;
    freeShippingLevel: number;
  };
  
  // 时间
  startTime: number;
  endTime: number;
  
  // 标签
  tags: string[];
  
  // 排序
  sortOrder: number;
}

// 商品表单组件
const ProductForm: React.FC<{
  product: PointsProduct;
  onSubmit: (data: ProductFormData) => void;
}> = ({ product, onSubmit }) => {
  const [formData, setFormData] = useState<ProductFormData>(
    product | {
      name: '',
      description: '',
      images: [],
      mainImage: '',
      categoryId: '',
      points: 0,
      stock: 0,
      limits: {
        maxPerUser: 1,
        minMemberLevel: 1,
        dailyLimit: 0
      },
      shipping: {
        required: false,
        fee: 0,
        freeShippingLevel: 0
      },
      startTime: Date.now(),
      endTime: Date.now() + 365 * 24 * 60 * 60 * 1000,
      tags: [],
      sortOrder: 0
    }
  );
  
  return (
    <form onSubmit={handleSubmit}>
      {/* 基本信息 */}
      <div className="mb-4">
        <label className="block text-sm font-bold mb-2">商品名称</label>
        <input
          type="text"
          value={formData.name}
          onChange={(e) => setFormData({ ...formData, name: e.target.value })}
          className="w-full px-3 py-2 border rounded-lg"
          required
        />
      </div>
      
      {/* 商品图片 */}
      <div className="mb-4">
        <label className="block text-sm font-bold mb-2">商品图片</label>
        <ImageUploader
          images={formData.images}
          onChange={(images) => setFormData({ ...formData, images })}
          maxCount={9}
        />
      </div>
      
      {/* 积分价格 */}
      <div className="mb-4">
        <label className="block text-sm font-bold mb-2">积分价格</label>
        <input
          type="number"
          value={formData.points}
          onChange={(e) => setFormData({ ...formData, points: Number(e.target.value) })}
          className="w-full px-3 py-2 border rounded-lg"
          min={0}
          required
        />
      </div>
      
      {/* 库存 */}
      <div className="mb-4">
        <label className="block text-sm font-bold mb-2">库存数量</label>
        <input
          type="number"
          value={formData.stock}
          onChange={(e) => setFormData({ ...formData, stock: Number(e.target.value) })}
          className="w-full px-3 py-2 border rounded-lg"
          min={0}
          required
        />
      </div>
      
      {/* 限制设置 */}
      <div className="mb-4">
        <label className="block text-sm font-bold mb-2">兑换限制</label>
        <div className="grid grid-cols-2 gap-3">
          <div>
            <label className="text-xs text-slate-500">每人最多兑/label>
            <input
              type="number"
              value={formData.limits.maxPerUser}
              onChange={(e) => setFormData({
                ...formData,
                limits: { ...formData.limits, maxPerUser: Number(e.target.value) }
              })}
              className="w-full px-3 py-2 border rounded-lg"
              min={1}
            />
          </div>
          <div>
            <label className="text-xs text-slate-500">最低会员等/label>
            <select
              value={formData.limits.minMemberLevel}
              onChange={(e) => setFormData({
                ...formData,
                limits: { ...formData.limits, minMemberLevel: Number(e.target.value) }
              })}
              className="w-full px-3 py-2 border rounded-lg"
            >
              <option value={1}>普通会/option>
              <option value={2}>黄金会员</option>
              <option value={3}>铂金会员</option>
              <option value={4}>钻石会员</option>
            </select>
          </div>
        </div>
      </div>
      
      {/* 提交按钮 */}
      <button
        type="submit"
        className="w-full py-3 bg-orange-500 text-white rounded-xl font-bold"
      >
        {product '保存修改' : '添加商品'}
      </button>
    </form>
  );
};
