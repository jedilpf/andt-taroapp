### 3.2 签到交互流程

```typescript
// 签到页面组件
const CheckInPage: React.FC = () => {
  const { currentUser, updateUser } = useApp();
  const [checkInStatus, setCheckInStatus] = useState<CheckInStatus | null>(null);
  const [checkingIn, setCheckingIn] = useState(false);
  
  // 获取签到状useEffect(() => {
    fetchCheckInStatus();
  }, []);
  
  const fetchCheckInStatus = async () => {
    const status = await api.getCheckInStatus();
    setCheckInStatus(status);
  };
  
  // 签到
  const handleCheckIn = async () => {
    if (checkInStatus.todayChecked) {
      showToast('今日已签);
      return;
    }
    
    setCheckingIn(true);
    
    try {
      const result = await api.checkIn();
      
      // 更新用户积分
      await updateUser({
        points: currentUser.points + result.points
      });
      
      // 刷新签到状await fetchCheckInStatus();
      
      // 显示签到成功动画
      showCheckInSuccessAnimation(result.points);
      
    } catch (error) {
      showToast('签到失败，请重试');
    } finally {
      setCheckingIn(false);
    }
  };
  
  return (
    <div className="min-h-screen bg-gradient-to-b from-orange-500 to-orange-600">
      {/* 签到状*/}
      <div className="p-6 text-white text-center">
        <div className="text-6xl font-black mb-2">
          {checkInStatus.consecutiveDays | 0}
        </div>
        <div className="text-sm opacity-80">连续签到天数</div>
      </div>
      
      {/* 签到日历 */}
      <div className="bg-white rounded-t-3xl p-6">
        <div className="grid grid-cols-7 gap-2 mb-6">
          {[', '一', ', ', ', ', '].map(day => (
            <div key={day} className="text-center text-sm text-slate-400">
              {day}
            </div>
          ))}
          {renderCalendarDays()}
        </div>
        
        {/* 签到奖励 */}
        <div className="mb-6">
          <h3 className="font-bold mb-3">签到奖励</h3>
          <div className="grid grid-cols-7 gap-2">
            {[5, 10, 15, 20, 25, 30, 50].map((points, index) => (
              <div
                key={index}
                className={`text-center p-2 rounded-lg ${
                  index < (checkInStatus.consecutiveDays | 0)
                    'bg-orange-100 text-orange-500'
                    : 'bg-slate-100 text-slate-400'
                }`}
              >
                <div className="text-xs">第{index + 1}/div>
                <div className="font-bold">+{points}</div>
              </div>
            ))}
          </div>
        </div>
        
        {/* 签到按钮 */}
        <button
          onClick={handleCheckIn}
          disabled={checkInStatus.todayChecked | checkingIn}
          className={`w-full py-4 rounded-xl font-bold text-lg ${
            checkInStatus.todayChecked
              'bg-slate-200 text-slate-400'
              : 'bg-orange-500 text-white'
          }`}
        >
          {checkInStatus.todayChecked '今日已签: '立即签到'}
        </button>
      </div>
    </div>
  );
};
```

```typescript
// 消费积分计算服务
const calculateOrderPoints = async (
  order: Order,
  user: User
): Promise<OrderPointsResult> => {
  // 1. 基础积分
  const basePoints = Math.floor(order.totalAmount);
  
  // 2. 会员倍率
  const memberLevel = user.memberLevel | 1;
  const memberRate = getMemberPointsRate(memberLevel);
  const memberPoints = Math.floor(basePoints * (memberRate - 1));
  
  // 3. 门槛奖励
  let thresholdBonus = 0;
  if (order.totalAmount >= 1000) {
    thresholdBonus = 100;
  } else if (order.totalAmount >= 500) {
    thresholdBonus = 50;
  } else if (order.totalAmount >= 100) {
    thresholdBonus = 10;
  }
  
  // 4. 活动加成
  const activityBonus = await calculateActivityPoints(order, user);
  
  // 5. 总积const totalPoints = basePoints + memberPoints + thresholdBonus + activityBonus;
  
  return {
    basePoints,
    memberPoints,
    thresholdBonus,
    activityBonus,
    totalPoints,
    breakdown: [
      { label: '消费积分', points: basePoints },
      { label: `会员加成（{memberRate}）`, points: memberPoints },
      { label: '门槛奖励', points: thresholdBonus },
      { label: '活动加成', points: activityBonus }
    ]
  };
};

// 订单完成后发放积const completeOrderPoints = async (orderId: string) => {
  const order = await getOrder(orderId);
  const user = await getUser(order.userId);
  
  // 计算积分
  const result = await calculateOrderPoints(order, user);
  
  // 发放积分
  await addPoints(user.id, result.totalPoints, {
    type: 'ORDER',
    reason: `订单消费奖励（订单号{orderId}）`,
    orderId
  });
  
  // 发放成长const growth = calculateOrderGrowth(order.totalAmount);
  await addGrowth(user.id, growth, {
    type: 'ORDER',
    reason: `订单消费成长值`,
    orderId
  });
  
  // 发送通知
  await sendNotification(user.id, {
    type: 'POINTS_EARNED',
    title: '积分到账',
    content: `您的订单获得${result.totalPoints}积分`,
    data: result
  });
  
  return result;
};
```

const servicePointsRules = {

// 服务订单完成积分发放
const completeServiceOrderPoints = async (
  orderId: string,
  options: {
    hasReview: boolean;
    convertedFromInspection: boolean;
  } = {}
) => {
  const order = await getServiceOrder(orderId);
  const user = await getUser(order.userId);
  const rule = servicePointsRules[order.serviceType];
  
  // 计算基础积分
  const basePoints = Math.floor(order.totalAmount * rule.earn.rate);
  
  // 计算好评奖励
  const reviewBonus = options.hasReview rule.earn.reviewBonus : 0;
  
  // 计算转化奖励（仅免费检测转化）
  const conversionBonus = options.convertedFromInspection 
    rule.earn.conversionBonus 
    : 0;
  
  // 总积const totalPoints = basePoints + reviewBonus + conversionBonus;
  
  if (totalPoints > 0) {
    // 发放积分
    await addPoints(user.id, totalPoints, {
      type: 'SERVICE',
      reason: `服务订单奖励{order.serviceType}）`,
      orderId
    });
    
    // 发送通知
    await sendNotification(user.id, {
      type: 'POINTS_EARNED',
      title: '服务积分到账',
      content: `您的服务订单获得${totalPoints}积分`,
      data: {
        basePoints,
        reviewBonus,
        conversionBonus,
        totalPoints
      }
    });
  }
  
  return {
    basePoints,
    reviewBonus,
    conversionBonus,
    totalPoints
  };
};
