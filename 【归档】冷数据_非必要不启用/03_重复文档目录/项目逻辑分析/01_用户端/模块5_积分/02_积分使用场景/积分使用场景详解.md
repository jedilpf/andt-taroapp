interface PointsDeductionProps {
  orderAmount: number;         // 订单金额
  userPoints: number;          // 用户积分
  onPointsChange: (points: number) => void;
}

const PointsDeduction: React.FC<PointsDeductionProps> = ({
  orderAmount,
  userPoints,
  onPointsChange
}) => {
  const [usePoints, setUsePoints] = useState(false);
  const [pointsToUse, setPointsToUse] = useState(0);
  
  // 计算最大可抵扣积分
  const maxDeductiblePoints = Math.floor(orderAmount * 0.1 * 100);
  
  // 计算实际可抵扣积const actualDeductiblePoints = Math.min(userPoints, maxDeductiblePoints);
  
  // 计算抵扣金额
  const deductionAmount = pointsToUse / 100;
  
  // 计算剩余支付金额
  const remainingAmount = Math.max(0.01, orderAmount - deductionAmount);
  
  // 处理开关变const handleToggle = (checked: boolean) => {
    setUsePoints(checked);
    if (checked) {
      // 默认使用最大可抵扣积分
      const defaultPoints = Math.min(actualDeductiblePoints, userPoints);
      setPointsToUse(defaultPoints);
      onPointsChange(defaultPoints);
    } else {
      setPointsToUse(0);
      onPointsChange(0);
    }
  };
  
  // 处理积分变化
  const handlePointsChange = (value: number) => {
    const validPoints = Math.min(value, actualDeductiblePoints);
    setPointsToUse(validPoints);
    onPointsChange(validPoints);
  };
  
  return (
    <div className="bg-white rounded-xl p-4 mb-3">
      {/* 开*/}
      <div className="flex items-center justify-between mb-3">
        <div className="flex items-center gap-2">

          <span className="font-bold text-slate-900">积分抵扣</span>
        </div>
        <Switch
          checked={usePoints}
          onChange={handleToggle}
          disabled={userPoints < 100}
        />
      </div>
      
      {/* 积分信息 */}
      {usePoints && (
        <>
          <div className="flex justify-between text-sm text-slate-500 mb-2">
            <span>可用积分：{userPoints}</span>
            <span>最多可使用：{actualDeductiblePoints}积分</span>

              className="flex-1 px-3 py-2 border rounded-lg"
              min={100}
              max={actualDeductiblePoints}

### 3.3 积分抵扣计算逻辑

```typescript
// 积分抵扣计算服务
const calculatePointsDeduction = (
  orderAmount: number,
  userPoints: number,
  pointsToUse: number,
  config: DeductionConfig = defaultConfig
): DeductionResult => {
  const { rate, maxRate, minPoints } = config;
  
  // 验证最低积if (pointsToUse < minPoints) {
    return {
      success: false,
      error: `最低使{minPoints}积分`,
      deductionAmount: 0,
      remainingAmount: orderAmount
    };
  }
  
  // 验证积分余额
  if (pointsToUse > userPoints) {
    return {
      success: false,
      error: '积分余额不足',
      deductionAmount: 0,
      remainingAmount: orderAmount
    };
  }
  
  // 计算最大可抵扣积分
  const maxDeductiblePoints = Math.floor(orderAmount * maxRate * rate);
  
  // 验证最高抵扣比if (pointsToUse > maxDeductiblePoints) {
    return {
      success: false,
      error: `最多可使用${maxDeductiblePoints}积分`,
      deductionAmount: 0,
      remainingAmount: orderAmount
    };
  }
  
  // 计算抵扣金额
  const deductionAmount = pointsToUse / rate;
  
  // 计算剩余支付金额
  const remainingAmount = Math.max(0.01, orderAmount - deductionAmount);
  
  return {
    success: true,
    error: '',
    pointsUsed: pointsToUse,
    deductionAmount,
    remainingAmount,
    maxDeductiblePoints
  };
};

// 默认配置
const defaultConfig = {
  rate: 100,        // 100积分 = 1maxRate: 0.1,     // 最高抵0%
  minPoints: 100    // 最00积分
};

### 4.3 积分兑换流程

```typescript
// 积分兑换服务
const exchangePointsProduct = async (
  productId: string,
  userId: string,
  quantity: number = 1,
  shippingInfo: ShippingInfo
): Promise<ExchangeResult> => {
  // 1. 获取商品信息
  const product = await getPointsProduct(productId);
  
  if (!product) {
    return { success: false, error: '商品不存};
  }
  
  // 2. 检查商品状if (product.status !== 'ACTIVE') {
    return { success: false, error: '商品已下};
  }
  
  // 3. 检查库if (product.stock < quantity) {
    return { success: false, error: '库存不足' };
  }
  
  // 4. 获取用户信息
  const user = await getUser(userId);
  
  // 5. 检查会员等if (user.memberLevel < product.limit.minMemberLevel) {
    return { success: false, error: '会员等级不足' };
  }
  
  // 6. 检查兑换次const exchangedCount = await getUserExchangedCount(userId, productId);
  if (exchangedCount + quantity > product.limit.maxPerUser) {
    return { success: false, error: `每人最多兑{product.limit.maxPerUser}件` };
  }
  
  // 7. 计算所需积分
  const totalPoints = product.points * quantity + product.shipping.fee;
  
  // 8. 检查积分余if (user.points < totalPoints) {
    return { success: false, error: '积分不足' };
  }
  
  // 9. 扣除积分
  await deductPoints(userId, totalPoints, {
    type: 'EXCHANGE',
    reason: `积分兑换{product.name}`,
    productId
  });
  
  // 10. 扣减库存
  await updateProductStock(productId, -quantity);
  
  // 11. 创建兑换订单
  const order = await createExchangeOrder({
    userId,
    productId,
    quantity,
    totalPoints,
    shippingInfo: product.shipping.required shippingInfo : null,
    status: product.type === 'COUPON' 'COMPLETED' : 'PENDING'
  });
  
  // 12. 发放优惠券（如果是优惠券商品if (product.type === 'COUPON') {
    await issueCoupon(userId, product.id);
  }
  
  // 13. 发送通知
  await sendNotification(userId, {
    type: 'EXCHANGE_SUCCESS',
    title: '兑换成功',
    content: `您已成功兑换${product.name}，消{totalPoints}积分`
  });
  
  return {
    success: true,
    orderId: order.id,
    totalPoints,
    product
  };
};
```

    <div className="bg-white rounded-xl p-4 mb-3">

### 7.2 积分转赠规则

```typescript
// 积分转赠规则
const transferRules = {
  // 单次转赠限制
  minPoints: 100,           // 最低转00积分
  maxPoints: 10000,         // 最高转0000积分
  
  // 每日限制
  dailyMaxPoints: 50000,    // 每日最多转0000积分
  dailyMaxCount: 10,        // 每日最多转0// 手续feeRate: 0.05,            // 5%手续// 会员限制
  minMemberLevel: 2,        // 黄金会员以上才能转赠
  
  // 其他限制
  requireFriend: true,      // 需要好友关minRegisterDays: 30       // 注册0};

// 积分转赠服务
const transferPoints = async (
  fromUserId: string,
  toUserId: string,
  points: number,
  message: string
): Promise<TransferResult> => {
  // 1. 验证转赠数量
  if (points < transferRules.minPoints) {
    return { success: false, error: `最低转{transferRules.minPoints}积分` };
  }
  
  if (points > transferRules.maxPoints) {
    return { success: false, error: `最高转{transferRules.maxPoints}积分` };
  }
  
  // 2. 获取用户信息
  const fromUser = await getUser(fromUserId);
  const toUser = await getUser(toUserId);
  
  // 3. 验证会员等级
  if (fromUser.memberLevel < transferRules.minMemberLevel) {
    return { success: false, error: '黄金会员以上才能转赠积分' };
  }
  
  // 4. 验证注册时间
  const registerDays = Math.floor((Date.now() - fromUser.createdAt) / (1000 * 60 * 60 * 24));
  if (registerDays < transferRules.minRegisterDays) {
    return { success: false, error: `注册{transferRules.minRegisterDays}天才能转赠` };
  }
  
  // 5. 验证好友关系
  if (transferRules.requireFriend) {
    const isFriend = await checkFriendship(fromUserId, toUserId);
    if (!isFriend) {
      return { success: false, error: '只能转赠给好};
    }
  }
  
  // 6. 验证积分余额
  if (fromUser.points < points) {
    return { success: false, error: '积分余额不足' };
  }
  
  // 7. 验证每日限制
  const todayTransfers = await getTodayTransfers(fromUserId);
  if (todayTransfers.count >= transferRules.dailyMaxCount) {
    return { success: false, error: `每日最多转{transferRules.dailyMaxCount}次` };
  }
  
  if (todayTransfers.totalPoints + points > transferRules.dailyMaxPoints) {
    return { success: false, error: `每日最多转{transferRules.dailyMaxPoints}积分` };
  }
  
  // 8. 计算手续const fee = Math.floor(points * transferRules.feeRate);
  const actualPoints = points - fee;
  
  // 9. 扣除转赠人积await deductPoints(fromUserId, points, {
    type: 'TRANSFER_OUT',
    reason: `转赠{toUser.name}`,
    relatedUserId: toUserId
  });
  
  // 10. 增加接收人积await addPoints(toUserId, actualPoints, {
    type: 'TRANSFER_IN',
    reason: `来自${fromUser.name}的转赠`,
    relatedUserId: fromUserId
  });
  
  // 11. 记录转赠
  await createTransferRecord({
    fromUserId,
    toUserId,
    points,
    fee,
    actualPoints,
    message
  });
  
  // 12. 发送通知
  await sendNotification(toUserId, {
    type: 'POINTS_RECEIVED',
    title: '收到积分转赠',
    content: `${fromUser.name}转赠给您${actualPoints}积分`,
    data: { points: actualPoints, fromUser: fromUser.name, message }
  });
  
  return {
    success: true,
    points,
    fee,
    actualPoints
  };
};
```

### 8.2 积分抽奖规则

```typescript
// 积分抽奖规则
const lotteryRules = {
  // 抽奖消costPerDraw: 100,         // 每次消00积分
  
  // 奖品配置
  prizes: [
    { id: '1', name: '1000积分', probability: 0.1, points: 1000 },
    { id: '2', name: '500积分', probability: 0.2, points: 500 },
    { id: '3', name: '100积分', probability: 0.3, points: 100 },
    { id: '4', name: '50积分', probability: 0.3, points: 50 },
    { id: '5', name: '谢谢参与', probability: 0.1, points: 0 }
  ],
  
  // 每日限制
  dailyMaxDraws: 10,        // 每日最多抽0// 保底机制
  guarantee: {
    enabled: true,
    draws: 10,              // 10次必minPrize: '100积分'
  }
};

// 积分抽奖服务
const drawLottery = async (userId: string): Promise<LotteryResult> => {
  // 1. 获取用户信息
  const user = await getUser(userId);
  
  // 2. 验证积分余额
  if (user.points < lotteryRules.costPerDraw) {
    return { success: false, error: '积分不足' };
  }
  
  // 3. 验证每日限制
  const todayDraws = await getTodayDraws(userId);
  if (todayDraws >= lotteryRules.dailyMaxDraws) {
    return { success: false, error: `每日最多抽{lotteryRules.dailyMaxDraws}次` };
  }
  
  // 4. 扣除积分
  await deductPoints(userId, lotteryRules.costPerDraw, {
    type: 'LOTTERY',
    reason: '积分抽奖'
  });
  
  // 5. 计算中奖结果
  const prize = calculatePrize(userId, todayDraws);
  
  // 6. 发放奖品
  if (prize.points > 0) {
    await addPoints(userId, prize.points, {
      type: 'LOTTERY',
      reason: `抽奖获得{prize.name}`
    });
  }
  
  // 7. 记录抽奖
  await createLotteryRecord({
    userId,
    prize: prize.id,
    points: prize.points,
    cost: lotteryRules.costPerDraw
  });
  
  return {
    success: true,
    prize,
    cost: lotteryRules.costPerDraw
  };
};

// 计算中奖结果
const calculatePrize = (userId: string, todayDraws: number): Prize => {
  // 检查保底机if (lotteryRules.guarantee.enabled && todayDraws >= lotteryRules.guarantee.draws - 1) {
    const guaranteePrize = lotteryRules.prizes.find(
      p => p.name === lotteryRules.guarantee.minPrize
    );
    if (guaranteePrize) {
      return guaranteePrize;
    }
  }
  
  // 正常概率计算
  const random = Math.random();
  let cumulative = 0;
  
  for (const prize of lotteryRules.prizes) {
    cumulative += prize.probability;
    if (random < cumulative) {
      return prize;
    }
  }
  
  return lotteryRules.prizes[lotteryRules.prizes.length - 1];
};
