# 积分规则配置详解

## 一、规则配置概| 属性 | 内容 | 属性 | 内容 
|------|------
| 模块名称 | 积分规则配置 
| 配置方式 | 后台管理 + 数据库配
| Phase-1优先级 | **P0（核心）** 
| 核心价值 | 灵活配置、动态调整、业务支

```typescript
    start: '00:00';              // 签到开始时end: '23:59';                // 签到结束时间
  };

  breakRule: {
```

```typescript
// 积分过期规则配置
interface ExpirationRules {
  // 基础配置
  enabled: boolean;
  version: string;
  
  // 有效期规validity: {
    type: 'FIXED' | 'ROLLING';   // 固定期限 | 滚动期限
    days: number;                // 有效天数
  };
  
  // 过期时间计算
  calculation: {
    type: 'EARN_DATE' | 'END_OF_YEAR';  // 获得日期 | 年底
    graceDays: number;           // 宽限天数
  };
  
  // 提醒配置
  reminder: {
    enabled: boolean;
    days: number[];              // 提前提醒天数
    channels: ('PUSH' | 'SMS' | 'EMAIL')[];
  };
  
  // 会员特权
  memberPrivilege: {
    enabled: boolean;
    privileges: {
      1: { validityDays: 365 },
      2: { validityDays: 365 },
      3: { validityDays: 730 },
      4: { permanent: true }     // 永久有效
    };
  };
  
  // 续期规则
  extension: {
    enabled: boolean;
    days: number;                // 续期天数
    cost: number;                // 续期消耗积};
  
  // 过期处理
  handling: {
    autoDeduct: boolean;         // 自动扣除
    notification: boolean;       // 发送通知
    archive: boolean;            // 归档记录

}

    enabled: true,
```

```typescript
class GrayReleaseExecutor {
  async startRelease(strategy: GrayReleaseStrategy): Promise<void> {
    await this.validateStrategy(strategy);
    await this.initializeMonitoring(strategy);
    await this.applyRuleChange(strategy);
    await this.startGrayTraffic(strategy);
    await this.enableAlertMonitoring(strategy);
  }

  async updateTrafficRatio(strategy: GrayReleaseStrategy, ratio: number): Promise<void> {
    const currentRatio = await this.getCurrentRatio(strategy.id);
    
    if (strategy.method.type === 'PERCENTAGE' && strategy.method.percentage.gradual) {
      const step = ratio - currentRatio;
      await this.gradualAdjust(strategy, step);
    } else {
      await this.directAdjust(strategy, ratio);
    }
    
    await this.updateMetrics(strategy.id, ratio);
  }

  async autoRollback(strategy: GrayReleaseStrategy): Promise<void> {
    await this.stopGrayTraffic(strategy);
    await this.restoreOriginalRule(strategy);
    await this.notifyRollback(strategy);
    await this.logRollback(strategy);
  }

  async completeRelease(strategy: GrayReleaseStrategy): Promise<void> {
    await this.applyFullRule(strategy);
    await this.stopGrayTraffic(strategy);
    await this.updateStrategyStatus(strategy, 'COMPLETED');
    await this.generateReport(strategy);
  }
}
```

```typescript
    const user = await this.getUser(userId);
    const role = await this.getRole(user.roleId);
    return this.hasPermission(role, permission);
  }

  checkPermissionDecorator(permission: string) {
    return (target: any, propertyKey: string, descriptor: PropertyDescriptor) => {
      const originalMethod = descriptor.value;
      
      descriptor.value = async function(...args: any[]) {
        const userId = getCurrentUserId();
        const hasPermission = await permissionService.checkPermission(userId, permission);
        
        if (!hasPermission) {
          throw new PermissionDeniedError(`缺少权限: ${permission}`);
        }
        
        return originalMethod.apply(this, args);
      };
      
      return descriptor;
    };
  }

  async logOperation(params: {
    userId: string;
    action: string;
    resource: string;
    resourceId: string;
    oldValue: any;
    newValue: any;
    ip: string;
    userAgent: string;
  }): Promise<void> {
    await auditLogRepository.create({
      id: generateId(),
      ...params,
      createdAt: new Date()
    });
  }
}

class RulesService {
  @PermissionService.checkPermissionDecorator('rules.config.publish')
  async publishRule(ruleId: string): Promise<void> {}

  @PermissionService.checkPermissionDecorator('rules.config.delete')
  async deleteRule(ruleId: string): Promise<void> {}
}
```
