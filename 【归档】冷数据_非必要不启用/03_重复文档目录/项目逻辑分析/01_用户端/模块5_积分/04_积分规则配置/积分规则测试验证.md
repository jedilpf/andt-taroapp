// 签到规则单元测试
describe('CheckInRules', () => {
  let checkInService: CheckInService;
  
  beforeEach(() => {
    checkInService = new CheckInService(defaultCheckInRules);
  });
  
  describe('积分计算', () => {
    test('天签到应获得5积分', () => {
      const result = checkInService.calculatePoints(1);
      expect(result).toBe(5);
    });
    
    test('天签到应获得50积分', () => {
      const result = checkInService.calculatePoints(7);
      expect(result).toBe(50);
    });
    
    test('连续签到应正确计, () => {
      const results = [];
      for (let day = 1; day <= 7; day++) {
        results.push(checkInService.calculatePoints(day));
      }
      expect(results).toEqual([5, 10, 15, 20, 25, 30, 50]);
    });
  });
  
  describe('会员加成', () => {
    test('普通会员应无加, () => {
      const result = checkInService.calculatePointsWithMember(1, 1);
      expect(result).toBe(5);
    });
    
    test('黄金会员应有1.2倍加, () => {
      const result = checkInService.calculatePointsWithMember(1, 2);
      expect(result).toBe(6);  // 5 * 1.2 = 6
    });
    
    test('铂金会员应有1.5倍加, () => {
      const result = checkInService.calculatePointsWithMember(1, 3);
      expect(result).toBe(8);  // Math.floor(5 * 1.5) = 7
    });
    
    test('钻石会员应有2倍加, () => {
      const result = checkInService.calculatePointsWithMember(1, 4);
      expect(result).toBe(10);  // 5 * 2 = 10
    });
  });
  
  describe('断签规则', () => {
    test('断签后应重置连续天数', () => {
      const status = {
        consecutiveDays: 5,
        lastCheckInDate: '2024-01-10'
      };
      
      const result = checkInService.handleBreak(status, '2024-01-12');
      expect(result.consecutiveDays).toBe(1);
    });
    
    test('宽限期内不应重置', () => {
      const rules = {
        ...defaultCheckInRules,
        breakRule: { type: 'RESET', gracePeriod: 1 }
      };
      
      const service = new CheckInService(rules);
      const status = {
        consecutiveDays: 5,
        lastCheckInDate: '2024-01-10'
      };
      
      const result = service.handleBreak(status, '2024-01-11');
      expect(result.consecutiveDays).toBe(6);
    });
  });
  
  describe('补签规则', () => {
    test('补签应扣除积, () => {
      const userPoints = 100;
      const result = checkInService.makeUp(userPoints, 1);
      
      expect(result.success).toBe(true);
      expect(result.pointsDeducted).toBe(50);
      expect(result.remainingPoints).toBe(50);
    });
    
    test('积分不足应失, () => {
      const userPoints = 30;
      const result = checkInService.makeUp(userPoints, 1);
      
      expect(result.success).toBe(false);
      expect(result.error).toBe('积分不足');
    });
    
    test('超过最大补签天数应失败', () => {
      const userPoints = 1000;
      const result = checkInService.makeUp(userPoints, 5);
      
      expect(result.success).toBe(false);
      expect(result.error).toBe('超过最大补签天);
    });
  });
  
  describe('边界条件', () => {
    test('连续签到超过7天应循环', () => {
      const result = checkInService.calculatePoints(8);
      expect(result).toBe(5);  // = });
    
    test('连续签到0天应返回0', () => {
      const result = checkInService.calculatePoints(0);
      expect(result).toBe(0);
    });
    
    test('负数天数应返', () => {
      const result = checkInService.calculatePoints(-1);
      expect(result).toBe(0);
    });
  });
});

// 性能测试
describe('Performance Tests', () => {
  let consumeService: ConsumePointsService;
  
  beforeEach(() => {
    consumeService = new ConsumePointsService(defaultConsumePointsRules);
  });
  
  test('积分计算性能应满足要, () => {
    const iterations = 10000;
    const startTime = Date.now();
    
    for (let i = 0; i < iterations; i++) {
      consumeService.calculateFullPoints({
        amount: Math.random() * 10000,
        memberLevel: Math.floor(Math.random() * 4) + 1,
        category: 'SUPERMARKET'
      });
    }
    
    const endTime = Date.now();
    const duration = endTime - startTime;
    const avgTime = duration / iterations;
    
    console.log(`总耗时: ${duration}ms`);
    console.log(`平均耗时: ${avgTime}ms`);
    
    // 平均耗时应小ms
    expect(avgTime).toBeLessThan(1);
  });
  
  test('批量积分计算性能应满足要, async () => {
    const orders = Array(1000).fill(0).map((_, i) => ({
      amount: Math.random() * 10000,
      memberLevel: Math.floor(Math.random() * 4) + 1,
      category: 'SUPERMARKET'
    }));
    
    const startTime = Date.now();
    
    const results = await Promise.all(
      orders.map(order => consumeService.calculateFullPoints(order))
    );
    
    const endTime = Date.now();
    const duration = endTime - startTime;
    
    console.log(`批量计算${orders.length}个订单耗时: ${duration}ms`);
    
    // 批量计算1000个订单应小于100ms
    expect(duration).toBeLessThan(100);
    expect(results.length).toBe(1000);
  });
});
