# 积分规则监控告警

## 一、监控告警概| 属性 | 内容 | 属性 | 内容 
|------|------
| 模块名称 | 积分规则监控告警体系 
| 监控范围 | 积分发放、使用、过期、异常等全链路监
| Phase-1优先级 | **P0（核心）** 
| 核心价值 | 实时监控、快速响应、风险预警、数据驱

## 三、核心监控指

### 3.1 业务指标监控

```typescript
// 业务监控指标
interface BusinessMetrics {
  // 积分发放指标
  pointsIssued: {
    // 实时指标
    realtime: {
      total: number;              // 总发放量
      bySource: {                 // 按来源统CHECKIN: number;
        ORDER: number;
        TASK: number;
        ACTIVITY: number;
        SERVICE: number;
      };
      rate: number;               // 发放速率（积分钟};
    
    // 统计指标
    statistics: {
      hourly: number[];           // 每小时发放量
      daily: number[];            // 每日发放weekly: number[];           // 每周发放monthly: number[];          // 每月发放};
    
    // 趋势指标
    trend: {
      dayOverDay: number;         // 环比
      weekOverWeek: number;       // 同比
      monthOverMonth: number;     // 月同};
  };
  
  // 积分使用指标
  pointsUsed: {
    realtime: {
      total: number;
      byScene: {
        MALL: number;
        RECHARGE: number;
        SERVICE: number;
        TRANSFER: number;
        LOTTERY: number;
      };
      rate: number;
    };
    
    statistics: {
      hourly: number[];
      daily: number[];
      weekly: number[];
      monthly: number[];
    };
    
    trend: {
      dayOverDay: number;
      weekOverWeek: number;
      monthOverMonth: number;
    };
  };
  
  // 积分过期指标
  pointsExpired: {
    realtime: {
      total: number;
      count: number;              // 过期用户};
    
    statistics: {
      daily: number[];
      weekly: number[];
      monthly: number[];
    };
  };
  
  // 积分余额指标
  pointsBalance: {
    total: number;                // 总积分余avgPerUser: number;           // 人均积分
    distribution: {               // 分布情况
      range: string;
      count: number;
      percentage: number;
    }[];
  };
  
  // 用户活跃指标
  userActivity: {
    dailyActiveUsers: number;     // 日活用户
    weeklyActiveUsers: number;    // 周活用户
    monthlyActiveUsers: number;   // 月活用户
    
    checkInRate: number;          // 签到taskCompletionRate: number;   // 任务完成pointsUsageRate: number;      // 积分使用};
  
  // 转化指标
  conversion: {
    pointsToOrderRate: number;    // 积分转化订单pointsRedemptionRate: number; // 积分兑换repeatPurchaseRate: number;   // 复购memberUpgradeRate: number;    // 会员升级};
}
```

      code: string;
      message: string;
      count: number;
      lastOccurrence: number;
    }[];
  };

  resources: {
    // CPU

      usage: number;              // CPU使用load: number;               // 系统负载

    memory: {
      usage: number;              // 内存使用available: number;          // 可用内存

    };
    
    // Redis
    redis: {
      connections: number;        // 连接memory: number;             // 内存使用
      hitRate: number;            // 命中ops: number;                // 操作};
  };
  
  // 可用性指availability: {
    uptime: number;               // 运行时间
    availability: number;         // 可用性（百分比）
    downtime: number;             // 宕机时间
    incidents: number;            // 故障次数
  };
}

## 七、监控API

```
# 监控数据
GET    /api/monitor/metrics                    # 获取监控指标
GET    /api/monitor/metrics/:name              # 获取指定指标

# 告警管理
GET    /api/alerts                             # 获取告警列表
GET    /api/alerts/:id                         # 获取告警详情
POST   /api/alerts/:id/acknowledge             # 确认告警
POST   /api/alerts/:id/close                   # 关闭告警
GET    /api/alerts/statistics                  # 获取告警统计

# 看板数据
GET    /api/dashboard/realtime                 # 获取实时看板数据
GET    /api/dashboard/history                  # 获取历史看板数据
GET    /api/dashboard/export                   # 导出看板数据
