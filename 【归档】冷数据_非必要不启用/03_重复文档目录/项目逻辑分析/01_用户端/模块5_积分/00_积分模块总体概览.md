## 四、积分价值体

### 4.1 积分价值定```
┌─────────────────────────────────────────────────────────────积分价值体├─────────────────────────────────────────────────────────────基础价值：                                                   - 100积分 = 1元人民币                                       - 积分不可逆向兑换现金                                       - 积分仅限平台内使使用价值：                                                   - 商城购物：最高抵扣订单金0%                              - 积分商城：直接兑换商- 充值缴费：100积分 = 1- 服务抵扣：最高抵扣服务费10%                                获取成本- 签到获取：免费（用户时间成本- 消费获取：消= 1积分                                  - 任务获取：完成任务获- 活动获取：参与活动获└─────────────────────────────────────────────────────────────```

### 4.2 积分流转平衡

```
积分发放 ─────────────────────────────────────积分消├── 签到发放（约30%├── 商城抵扣（约50%├── 消费发放（约40%├── 积分商城（约30%├── 任务发放（约20%├── 充值抵扣（5%└── 活动发放（约10%└── 服务抵扣（约5%目标平衡- 月发放量 月消耗量 × 1.2（保持适度增长- 积分池总量控制在合理范- 避免积分通胀或通缩
```

// 用户积分账户
interface UserPointsAccount {
  userId: string;
  points: number;           // 当前积分余额
  totalEarned: number;      // 累计获得积分
  totalUsed: number;        // 累计使用积分
  totalExpired: number;     // 累计过期积分
  frozenPoints: number;     // 冻结积分
  
  // 会员信息
  memberLevel: number;      // 会员等级
  memberGrowth: number;     // 成长// 时间信息
  createdAt: number;
  updatedAt: number;
}

// 积分批次
interface PointsBatch {
  id: string;
  userId: string;
  batchNo: string;          // 批次// 积分信息
  points: number;           // 批次积分总数
  remainingPoints: number;  // 剩余积分
  
  // 来源信息
  source: {
    type: 'CHECKIN' | 'ORDER' | 'TASK' | 'ACTIVITY' | 'SERVICE' | 'TRANSFER';
    reason: string;
    orderId: string;
    taskId: string;
    activityId: string;
  };
  
  // 时间信息
  earnedAt: number;         // 获得时间
  expireAt: number;         // 过期时间
  lastUsedAt: number;       // 最后使用时// 状status: 'ACTIVE' | 'PARTIAL' | 'EXPIRED' | 'USED';
}

// 积分记录
interface PointsRecord {
  id: string;
  userId: string;
  
  // 变动信息
  type: 'EARN' | 'SPEND' | 'EXPIRE' | 'FREEZE' | 'UNFREEZE' | 'TRANSFER_IN' | 'TRANSFER_OUT';
  amount: number;           // 变动数量
  balance: number;          // 变动后余// 来源信息
  source: {
    type: string;
    reason: string;
    batchId: string;
    orderId: string;
    taskId: string;
  };
  
  // 时间信息
  timestamp: number;
  
  // 关联信息
  relatedUserId: string;   // 转赠对象
  relatedOrderId: string;
}

// 签到记录
interface CheckInRecord {
  id: string;
  userId: string;
  date: string;             // "2026-03-30"
  
  // 签到信息
  consecutiveDays: number;  // 连续签到天数
  points: number;           // 获得积分
  
  // 时间信息
  timestamp: number;
}

// 任务进度
interface TaskProgress {
  id: string;
  userId: string;
  taskId: string;
  
  // 进度信息
  progress: number;         // 当前进度
  target: number;           // 目标completed: boolean;       // 是否完成
  claimed: boolean;         // 是否已领取奖// 时间信息
  updatedAt: number;
}

// 会员成长记录
interface GrowthRecord {
  id: string;
  userId: string;
  
  // 成长值信amount: number;           // 成长值变balance: number;          // 变动后成长// 来源
  source: {
    type: 'ORDER' | 'CHECKIN' | 'REVIEW' | 'INVITE' | 'COMPLETE_PROFILE';
    reason: string;
    orderId: string;
  };
  
  // 时间信息
  timestamp: number;
}
