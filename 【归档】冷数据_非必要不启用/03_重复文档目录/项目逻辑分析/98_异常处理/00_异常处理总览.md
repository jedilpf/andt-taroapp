| 等级 | 名称 | 影响范围 | 响应时间 | 处理方式

### 5.2 异常处理代码示例

```typescript
// 异常处理统一封装
class ErrorHandler {
  static handle(error: ApiError, context: ErrorContext) {
    const errorConfig = this.getErrorConfig(error.code);
    
    // 1. 记录日志
    this.logError(error, context);
    
    // 2. 用户提示
    if (errorConfig.showToUser) {
      this.showUserMessage(errorConfig);
    }
    
    // 3. 自动处理
    if (errorConfig.autoHandle) {
      this.autoHandle(error, errorConfig);
    }
    
    // 4. 告警通知
    if (errorConfig.needAlert) {
      this.sendAlert(error, context);
    }
    
    // 5. 返回处理结果
    return {
      handled: true,
      userAction: errorConfig.userAction,
      retryable: errorConfig.retryable
    };
  }
}
```
