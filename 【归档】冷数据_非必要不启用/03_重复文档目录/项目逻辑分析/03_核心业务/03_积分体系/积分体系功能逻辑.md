### 4.2 积分抵扣计算规则

```typescript
// 积分抵扣计算
const calculatePointsDeduction = (
    orderAmount: number,
    userPoints: number
): { pointsToUse: number; deduction: number; remainingAmount: number } => {
    // 1. 计算最大可抵扣积分
    // 规则：保留最低支付.01
    const maxDeductiblePoints = Math.min(
        userPoints,
        Math.floor((orderAmount - 0.01) * 100)
    );
    
    // 2. 计算抵扣金额
    const deduction = maxDeductiblePoints / 100;
    
    // 3. 计算剩余支付金额
    const remainingAmount = Math.max(orderAmount - deduction, 0.01);
    
    return {
        pointsToUse: maxDeductiblePoints,
        deduction,
        remainingAmount
    };
};
```

            read: false
        }, ...prev]);
    }
};

// 签到积分计算
const calculateCheckInPoints = (consecutiveDays: number): number => {
    if (consecutiveDays >= 5) {
        return 20; // 封顶
    }
    return 5 + (consecutiveDays - 1); // ..

// 签到处理
const handleCheckIn = async () => {
    const today = new Date().toDateString();
    const lastCheckIn = localStorage.getItem('lastCheckIn');
    
    // 判断是否已签if (lastCheckIn === today) {
        showToast('今日已签到，明天再来吧！');
        return;
    }
    
    // 计算连续签到天数
    let consecutiveDays = 1;
    const lastDate = new Date(lastCheckIn | 0);
    const yesterday = new Date();
    yesterday.setDate(yesterday.getDate() - 1);
    
    if (lastDate.toDateString() === yesterday.toDateString()) {
        consecutiveDays = (parseInt(localStorage.getItem('consecutiveDays') | '0') + 1);
    }
    
    // 计算积分
    const points = calculateCheckInPoints(consecutiveDays);
    
    // 发放积分
    earnPoints(points);
    
    // 更新签到记录
    localStorage.setItem('lastCheckIn', today);
    localStorage.setItem('consecutiveDays', consecutiveDays.toString());
    
    // 显示成功提示
    showToast(`签到成功！获${points} 积分`);
};

### 10.2 过期提醒实现

```typescript
// 积分过期提醒
const checkPointsExpiration = async (userId: string) => {
    const records = await getPointsRecords(userId);
    const now = Date.now();
    
    records.forEach(record => {
        if (!record.expiredAt) return;
        
        const daysUntilExpiration = Math.ceil(
            (record.expiredAt - now) / (24 * 60 * 60 * 1000)
        );
        
        // 过期0天提if (daysUntilExpiration === 30) {
            sendNotification(userId, {
                title: '积分即将过期',
                content: `您有${record.amount}积分将在30天后过期，请尽快使用`,
                type: 'SYSTEM'
            });
        }
        
        // 过期天提if (daysUntilExpiration === 7) {
            sendNotification(userId, {
                title: '积分过期提醒',
                content: `您有${record.amount}积分将在7天后过期，请尽快使用`,
                type: 'SYSTEM'
            });
        }
        
        // 过期天提if (daysUntilExpiration === 1) {
            sendNotification(userId, {
                title: '积分明天过期',
                content: `您有${record.amount}积分将在明天过期，请尽快使用`,
                type: 'SYSTEM'
            });
        }
    });
};

): Promise<boolean> => {
    // 1. 检查获取频const recentRecords = await getRecentPointsRecords(userId, 60 * 60 * 1000); // 最小时
    const totalRecent = recentRecords
        .filter(r => r.source === source)
        .reduce((sum, r) => sum + r.amount, 0);
    
    if (totalRecent > 1000) {
        logSuspiciousActivity(userId, '高频获取积分');
        return false;
    }
    
    // 2. 检查IP地址
    const currentIP = await getCurrentIP();
    const lastIP = await getLastLoginIP(userId);
    if (currentIP !== lastIP) {
        logSuspiciousActivity(userId, 'IP地址变更');
    }
    
    // 3. 检查设const currentDevice = await getCurrentDevice();
    const lastDevice = await getLastLoginDevice(userId);
    if (currentDevice !== lastDevice) {
        logSuspiciousActivity(userId, '设备变更');
    }
    
    return true;
};

    userId: string,
    amount: number,
    source: string
) => {
    const isSafe = await checkPointsFraud(userId, source, amount);
    
    if (isSafe) {
