## 七、状态更新实

### 7.1 Context 更新方法

```typescript
// AppContext.tsx 中的订单更新方法
const updateOrder = (orderId: string, updates: Partial<Order>) => {
  setOrders(prev => prev.map(o => {
    if (o.id === orderId) {
      const updated = { ...o, ...updates };
      
      // 状态变更时记录时间if (updates.status) {
        updated.timeline = [
          ...o.timeline,
          { status: updates.status, time: Date.now() }
        ];
        
        // 发送消息通知
        setMessages(m => [{
          id: Date.now().toString(),
          type: 'ORDER',
          title: '订单状态更,
          content: `您的订单 [${o.title}] 已变更为: ${updates.status}`,
          time: Date.now(),
          read: false,
          link: `/user/order/${o.id}`
        }, ...m]);
      }
      
      return updated;
    }
    return o;
  }));
};
```

### 8.2 状态校```typescript
// 状态流转校const validateStatusTransition = (
  currentStatus: OrderStatus, 
  targetStatus: OrderStatus
): boolean => {
  const validTransitions: Record<OrderStatus, OrderStatus[]> = {
    [OrderStatus.PENDING]: [
      OrderStatus.ACCEPTED, 
      OrderStatus.CANCELLED
    ],
    [OrderStatus.ACCEPTED]: [
      OrderStatus.ARRIVED, 
      OrderStatus.CANCELLED
    ],
    [OrderStatus.ARRIVED]: [
      OrderStatus.IN_PROGRESS
    ],
    [OrderStatus.IN_PROGRESS]: [
      OrderStatus.COMPLETED
    ],
    [OrderStatus.COMPLETED]: [
      OrderStatus.PAID, 
      OrderStatus.CANCELLED
    ],
    [OrderStatus.PAID]: [], // 终[OrderStatus.CANCELLED]: [], // 终};
  
  return validTransitions[currentStatus].includes(targetStatus) false;
};

## 十一、异常处

### 11.1 状态冲```typescript
// 状态冲突处const handleStatusConflict = (error: StatusConflictError) => {
  switch (error.code) {
    case 'ORDER_ALREADY_ACCEPTED':
      showToast('订单已被其他师傅接单');
      break;
    case 'ORDER_ALREADY_CANCELLED':
      showToast('订单已取消，无法操作');

    case 'INVALID_STATUS_TRANSITION':
      showToast('状态流转不合法');
      break;
    default:
      showToast('操作失败，请刷新后重);
  }
};

// 乐观锁机const updateOrderWithLock = async (
  orderId: string,

  expectedVersion: number
) => {
  const response = await fetch(`/api/orders/${orderId}`, {
    method: 'PUT',
    body: JSON.stringify({

      _version: expectedVersion,
    }),
  });
  
  if (response.status === 409) {

    return;
  }
  
  return response.json();
