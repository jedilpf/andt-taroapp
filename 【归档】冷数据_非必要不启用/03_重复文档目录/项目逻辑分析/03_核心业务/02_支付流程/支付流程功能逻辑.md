### 4.3 积分抵扣

**适用场景**: 所有支付场景（可叠加其他方式）

```typescript
// 积分抵扣逻辑
const calculatePointsDeduction = (
    orderAmount: number,
    userPoints: number
): { pointsToUse: number; deduction: number; remainingAmount: number } => {
    // 1. 计算最大可抵扣积分
    const maxDeductiblePoints = Math.min(
        userPoints,
        Math.floor((orderAmount - 0.01) * 100) // 保留最低支付.01
    );
    
    // 2. 计算抵扣金额
    const deduction = maxDeductiblePoints / 100;
    
    // 3. 计算剩余支付金额
    const remainingAmount = orderAmount - deduction;
    
    return {
        pointsToUse: maxDeductiblePoints,
        deduction,
        remainingAmount: Math.max(remainingAmount, 0.01)
    };
};

// 积分抵扣支付
const handlePointsPayment = async (
    orderId: string,
    pointsToUse: number
) => {
    try {
        // 1. 验证积分余额
        if (currentUser.points < pointsToUse) {
            showToast('积分余额不足');
            return;
        }
        
        // 2. 扣除积分
        spendPoints(pointsToUse);
        
        // 3. 更新订单
        updateOrder(orderId, {
            pointsUsed: pointsToUse,
            pointsDeduction: pointsToUse / 100
        });
        
        // 4. 如果还有剩余金额，跳转支付页const remainingAmount = orderAmount - pointsToUse / 100;
        if (remainingAmount > 0.01) {
            navigate('/payment', { state: { amount: remainingAmount } });
        } else {
            // 支付完成
            await completeOrder(orderId);
        }
    } catch (error) {
        showToast('积分抵扣失败，请重试');
    }
};
```

## 八、异常处

### 8.1 支付失败处理

**场景**: 用户支付时遇到错**处理方案**:
```typescript
// 支付失败处理
const handlePaymentFailure = (
    orderId: string,
    error: PaymentError
) => {
    // 记录失败日志
    logPaymentError(orderId, error);
    
    // 根据错误类型提供解决方案
    switch (error.code) {
        case 'INSUFFICIENT_BALANCE':
            showToast('余额不足，请选择其他支付方式');
            showPaymentMethods(orderId);
            break;
            
        case 'NETWORK_ERROR':
            showToast('网络异常，请检查网络后重试');
            showRetryButton(orderId);
            break;
            
        case 'PAYMENT_TIMEOUT':
            showToast('支付超时，订单已保留30分钟');
            startPaymentCountdown(orderId, 30 * 60);
            break;
            
        case 'PAYMENT_LIMIT_EXCEEDED':
            showToast('支付金额超限，请联系客服');
            showCustomerService();
            break;
            
        case 'ACCOUNT_FROZEN':
            showToast('账户已冻结，请联系客服解);
            showCustomerService();
            break;
            
        default:
            showToast('支付失败，请重试或联系客);
            showRetryOrCustomerService(orderId);
    }
};
```

### 11.2 积分抵扣计算示例

```typescript
// 积分抵扣计算函数
const calculatePointsDeduction = (
    orderAmount: number,
    userPoints: number
): { pointsToUse: number; deduction: number; remainingAmount: number } => {
    // 1. 计算最大可抵扣积分
    // 规则：保留最低支付.01
    const maxDeductiblePoints = Math.min(
        userPoints,
        Math.floor((orderAmount - 0.01) * 100)
    );
    
    // 2. 计算抵扣金额
    const deduction = maxDeductiblePoints / 100;
    
    // 3. 计算剩余支付金额
    const remainingAmount = Math.max(orderAmount - deduction, 0.01);
    
    return {
        pointsToUse: maxDeductiblePoints,
        deduction,
        remainingAmount
    };
};

// 示例1：积分充const result1 = calculatePointsDeduction(80, 10000);
// 结果：{ pointsToUse: 7900, deduction: 79, remainingAmount: 1 }

// 示例2：积分不const result2 = calculatePointsDeduction(80, 2580);
// 结果：{ pointsToUse: 2580, deduction: 25.8, remainingAmount: 54.2 }

// 示例3：积分刚const result3 = calculatePointsDeduction(80, 7900);
// 结果：{ pointsToUse: 7900, deduction: 79, remainingAmount: 1 }
