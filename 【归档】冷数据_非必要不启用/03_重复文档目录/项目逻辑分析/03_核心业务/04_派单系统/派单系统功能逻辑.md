// 综合推荐算法
const calculateRecommendScore = (
  order: Order,
  electrician: ElectricianProfile
): number => {
  // 1. 距离权重0%const distance = calculateDistance(
    electrician.location,
    order.location
  );
  const distanceScore = Math.max(0, 100 - distance * 10); // 10km内满// 2. 评分权重0%const ratingScore = electrician.rating * 20; // 5分制00// 3. 价格权重0%const priceScore = order.priceEstimate.min / 10; // 价格越高分数越高
  
  // 4. 在线状态权重（10%const onlineScore = electrician.workStatus === WorkStatus.ONLINE 100 : 0;
  
  // 总分计算
  const totalScore = 
    distanceScore * 0.4 +
    ratingScore * 0.3 +
    priceScore * 0.2 +
    onlineScore * 0.1;
  
  return totalScore;
};
```

**问题**: 多个电工同时抢单，如何保证订单不被重复分配？

```typescript
// 乐观锁实const grabOrderWithOptimisticLock = async (
  orderId: string,
  electricianId: string
): Promise<{ success: boolean; message: string }> => {
  // 1. 查询订单当前状const order = await getOrderById(orderId);
  
  // 2. 检查订单状if (order.status !== OrderStatus.PENDING) {
    return { success: false, message: '订单已被抢走' };
  }
  
  // 3. 尝试更新（带版本号校验）
  const result = await updateOrderWithVersion(
    orderId,
    {
      status: OrderStatus.ACCEPTED,
      electricianId,
      electricianName: await getElectricianName(electricianId)
    },
    order.version // 乐观锁版本号
  );
  
  // 4. 返回结果
  if (result.affected > 0) {
    return { success: true, message: '抢单成功' };
  } else {
    return { success: false, message: '手慢了，订单已被抢走' };
  }
};

```
订单状态流转：

PENDING（待接单电工抢单
ACCEPTED（已接单电工到达现场
ARRIVED（已到达开始施IN_PROGRESS（施工中施工完成
COMPLETED（已完成用户支付
PAID（已支付异常路径PENDING ──(超时/用户取消)──CANCELLED
ACCEPTED ──(电工取消)──CANCELLED

const updateOrder = (orderId: string, updates: Partial<Order>) => {
  setOrders(prev => prev.map(o => {
    if (o.id === orderId) {
      const updated = { ...o, ...updates };
      
      // 状态变更时发送通知

        setMessages(m => [{

          type: 'ORDER',

          read: false,

      }
      
      return updated;
    }
    return o;
  }));
};

// 订单超时检const checkOrderTimeout = async () => {
  const pendingOrders = await getPendingOrders();
  const now = Date.now();
  
  pendingOrders.forEach(order => {
    const elapsed = now - order.createdAt;
    const timeout = order.type === 'Repair' 30 * 60 * 1000 : 24 * 60 * 60 * 1000;
    
    if (elapsed > timeout) {
      // 自动取消订单

// Haversine公式计算距离
const calculateDistance = (
  loc1: Location,
  loc2: Location
): number => {
  const R = 6371; // 地球半径（公里）
  const dLat = toRad(loc2.lat - loc1.lat);
  const dLng = toRad(loc2.lng - loc1.lng);
  
  const a = 
    Math.sin(dLat / 2) * Math.sin(dLat / 2) +
    Math.cos(toRad(loc1.lat)) * Math.cos(toRad(loc2.lat)) *
    Math.sin(dLng / 2) * Math.sin(dLng / 2);
  
  const c = 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1 - a));
  const distance = R * c;
  
  return distance;
};

const toRad = (deg: number): number => {
  return deg * (Math.PI / 180);
};
```
