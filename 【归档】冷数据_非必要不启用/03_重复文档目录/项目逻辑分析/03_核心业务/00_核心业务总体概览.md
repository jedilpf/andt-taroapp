// 订单状态更新方const updateOrder = (orderId: string, updates: Partial<Order>) => {
    setOrders(prev => prev.map(o => {
        if (o.id === orderId) {
            const updated = { ...o, ...updates };
            // 状态变更时发送通知
            if (updates.status) {
                setMessages(m => [{
                    id: Date.now().toString(),
                    type: 'ORDER',
                    title: '订单状态更,
                    content: `您的订单 [${o.title}] 已变更为: ${updates.status}`,
                    time: Date.now(),
                    read: false,
                    link: `/user/order/${o.id}`
                }, ...m]);
            }
            return updated;
        }
        return o;
    }));
};

// 积分获取方法
const earnPoints = (amount: number, isFirstPurchaseBonus = false) => {
    if (currentUser) {
        // 计算总积分：基础积分 + 首单奖励(1000const totalAdd = Math.floor(amount) + (isFirstPurchaseBonus 1000 : 0);
        const updatedUser = { 
            ...currentUser, 
            points: currentUser.points + totalAdd,
            isFirstPurchase: isFirstPurchaseBonus false : currentUser.isFirstPurchase 
        };
        setCurrentUser(updatedUser);
        sessionStorage.setItem('currentUser', JSON.stringify(updatedUser));
        
        // 发送积分变动通知
        setMessages(prev => [{
            id: Date.now().toString(),
            type: 'SYSTEM',
            title: '积分变动通知',
            content: `恭喜！您获得${totalAdd} 积分奖励。`,
            time: Date.now(),
            read: false
        }, ...prev]);
    }
};
