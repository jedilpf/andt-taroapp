### 3. 订单创建页 - pages/order/create.tsx

```typescript
import { View, Form, Input, Picker, Image } from '@tarojs/components'
import { inject, observer } from 'mobx-react'
import { Component } from 'react'
import Taro from '@tarojs/taro'
import './create.scss'

@inject('orderStore', 'userStore')
@observer
class CreateOrder extends Component {
  state = {
    faultType: '',
    description: '',
    images: [],
    address: null,
    serviceTime: ''
  }

  onSubmit = async () => {
    const { faultType, description, images, address, serviceTime } = this.state
    
    if (!faultType) {
      return Taro.showToast({ title: '请选择故障类型', icon: 'none' })
    }
    if (!address) {
      return Taro.showToast({ title: '请选择服务地址', icon: 'none' })
    }

    Taro.showLoading({ title: '提交中...' })
    
    try {
      await this.props.orderStore.createOrder({
        faultType,
        description,
        images,
        address,
        serviceTime
      })
      
      Taro.hideLoading()
      Taro.showToast({ title: '提交成功' })
      
      setTimeout(() => {
        Taro.navigateTo({ url: '/pages/order/list' })
      }, 1500)
    } catch (error) {
      Taro.hideLoading()
      Taro.showToast({ title: error.message, icon: 'none' })
    }
  }

  chooseImage = () => {
    Taro.chooseImage({
      count: 3,
      sizeType: ['compressed'],
      sourceType: ['album', 'camera']
    }).then(res => {
      this.setState({ images: res.tempFilePaths })
    })
  }

  render() {
    const faultTypes = ['跳闸/断电', '开关故障', '插座故障', '灯具故障', '线路问题', '其他']
    
    return (
      <View className='create-order-page'>
        <Form onSubmit={this.onSubmit}>
          <Picker
            mode='selector'
            range={faultTypes}
            onChange={e => this.setState({ faultType: faultTypes[e.detail.value] })}
          >
            <View className='form-item'>
              <View className='label'>故障类型</View>
              <View className='value'>{this.state.faultType || '请选择'}</View>
            </View>
          </Picker>
          
          <View className='form-item'>
            <View className='label'>故障描述</View>
            <Input
              className='input'
              placeholder='请描述故障情况'
              value={this.state.description}
              onInput={e => this.setState({ description: e.detail.value })}
            />
          </View>
          
          <View className='form-item'>
            <View className='label'>上传照片</View>
            <View className='image-list'>
              {this.state.images.map((img, index) => (
                <Image key={index} src={img} className='image' />
              ))}
              {this.state.images.length < 3 && (
                <View className='add-image' onClick={this.chooseImage}>+</View>
              )}
            </View>
          </View>
          
          <View className='submit-btn' onClick={this.onSubmit}>提交订单</View>
        </Form>
      </View>
    )
  }
}

export default CreateOrder
```

```typescript
    currentTab: 0,
    tabs: ['全部', '待接单', '进行中', '已完成']
  }

  componentDidMount() {
    this.loadOrders()
  }

    const { currentTab, tabs } = this.state
    const status = currentTab === 0 ? '' : tabs[currentTab]
    this.props.orderStore.getOrders({ status })
  }

  onTabChange = (index) => {
    this.setState({ currentTab: index }, this.loadOrders)
  }

  render() {
    const { orders } = this.props.orderStore
    const { currentTab, tabs } = this.state
    
    return (
      <View className='order-list-page'>
        <View className='tabs'>
          {tabs.map((tab, index) => (
            <View
              key={index}
              className={`tab-item ${currentTab === index ? 'active' : ''}`}
              onClick={() => this.onTabChange(index)}
            >
              {tab}
            </View>
          ))}
        </View>
        
        <ScrollView scrollY className='scroll-view'>
          {orders.map(order => (
```

```java
package com.andiantong.service.impl;

import com.andiantong.dto.OrderCreateDTO;
import com.andiantong.dto.OrderQueryDTO;
import com.andiantong.entity.Order;
import com.andiantong.enums.OrderStatus;
import com.andiantong.exception.BusinessException;
import com.andiantong.mapper.OrderMapper;
import com.andiantong.service.OrderService;
import com.andiantong.service.PaymentService;
import com.andiantong.service.PushService;
import com.andiantong.vo.OrderVO;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Service;
import org.springframework.transaction.annotation.Transactional;
import java.time.LocalDateTime;
import java.util.List;
import java.util.stream.Collectors;

@Service
public class OrderServiceImpl implements OrderService {

    @Autowired
    private OrderMapper orderMapper;

    @Autowired
    private PaymentService paymentService;

    @Autowired
    private PushService pushService;

    @Override
    @Transactional
    public OrderVO createOrder(OrderCreateDTO dto) {
        Order order = new Order();
        order.setUserId(dto.getUserId());
        order.setFaultType(dto.getFaultType());
        order.setDescription(dto.getDescription());
        order.setImages(String.join(",", dto.getImages()));
        order.setAddress(dto.getAddress());
        order.setServiceTime(dto.getServiceTime());
        order.setStatus(OrderStatus.PENDING.getCode());
        order.setCreateTime(LocalDateTime.now());
        
        orderMapper.insert(order);
        
        pushService.notifyElectricians(order);
        
        return convertToVO(order);
    }

    @Override
    public List<OrderVO> getOrderList(OrderQueryDTO query) {
        List<Order> orders = orderMapper.selectList(query);
        return orders.stream()
                .map(this::convertToVO)
                .collect(Collectors.toList());
    }

    @Override
    public OrderVO getOrderDetail(Long id) {
        Order order = orderMapper.selectById(id);
        if (order == null) {
            throw new BusinessException("订单不存在");
        }
        return convertToVO(order);
    }

    @Override
    @Transactional
    public void cancelOrder(Long id) {
        Order order = orderMapper.selectById(id);
        if (order == null) {
            throw new BusinessException("订单不存在");
        }
        if (order.getStatus() != OrderStatus.PENDING.getCode()) {
            throw new BusinessException("当前状态不可取消");
        }
        
        order.setStatus(OrderStatus.CANCELLED.getCode());
        order.setUpdateTime(LocalDateTime.now());
        orderMapper.updateById(order);
    }

    @Override
    @Transactional
    public Object payOrder(Long id) {
        Order order = orderMapper.selectById(id);
        if (order == null) {
            throw new BusinessException("订单不存在");
        }
        if (order.getStatus() != OrderStatus.COMPLETED.getCode()) {
            throw new BusinessException("订单未完成，无法支付");
        }
        
        return paymentService.createPayment(order);
    }

    private OrderVO convertToVO(Order order) {
        OrderVO vo = new OrderVO();
        vo.setId(order.getId());
        vo.setFaultType(order.getFaultType());
        vo.setDescription(order.getDescription());
        vo.setAddress(order.getAddress());
        vo.setStatus(order.getStatus());
        vo.setStatusText(OrderStatus.getText(order.getStatus()));
        vo.setAmount(order.getAmount());
        vo.setCreateTime(order.getCreateTime());
        return vo;
    }
}
```

```java
    @Transactional
    public Object createPayment(Order order) {
        Payment payment = new Payment();
        payment.setOrderId(order.getId());
        payment.setUserId(order.getUserId());
        payment.setAmount(order.getAmount());
        payment.setStatus(PaymentStatus.PENDING.getCode());
        payment.setCreateTime(LocalDateTime.now());
        
        paymentMapper.insert(payment);
        
        Map<String, Object> params = new HashMap<>();
        params.put("body", "安电通服务-" + order.getFaultType());
        params.put("out_trade_no", payment.getId().toString());
        params.put("total_fee", String.valueOf(order.getAmount() * 100));
        params.put("notify_url", wechatConfig.getNotifyUrl());
        
        return wechatPayUtil.createOrder(params);
    }

    @Override
    @Transactional
    public void handleCallback(Map<String, Object> data) {
        String orderId = (String) data.get("out_trade_no");
        String transactionId = (String) data.get("transaction_id");
        
        Payment payment = paymentMapper.selectById(Long.parseLong(orderId));
        if (payment == null) {
            throw new BusinessException("支付记录不存在");
        }
        
        payment.setStatus(PaymentStatus.SUCCESS.getCode());
        payment.setTransactionId(transactionId);
        payment.setPayTime(LocalDateTime.now());
        payment.setUpdateTime(LocalDateTime.now());
        
        paymentMapper.updateById(payment);
    }
}
```
